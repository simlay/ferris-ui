1/// Create a new type to represent a class.
2///
3/// This is similar to an `@interface` declaration in Objective-C.
4///
5/// It is useful for things like `objc2-foundation`, which needs to create
6/// interfaces to existing, externally defined classes like `NSString`,
7/// `NSURL` and so on, but can also be useful for users that have custom
8/// classes written in Objective-C that they want to access from Rust.
9///
10///
11/// # Specification
12///
13/// The syntax is similar enough to Rust syntax that if you invoke the macro
14/// with parentheses (as opposed to curly brackets), [`rustfmt` will be able to
15/// format the contents][rustfmt-macros] (so e.g. as `extern_class!( ... );`).
16///
17/// The macro creates an opaque struct containing the superclass (which means
18/// that auto traits are inherited from the superclass), and implements the
19/// following traits for it to allow easier usage as an Objective-C object:
20///
21/// - [`RefEncode`][crate::RefEncode]
22/// - [`Message`][crate::Message]
23/// - [`ClassType`][crate::ClassType]
24/// - [`Deref<Target = $superclass>`][core::ops::Deref]
25/// - [`DerefMut`][core::ops::DerefMut]
26/// - [`AsRef<$inheritance_chain>`][AsRef]
27/// - [`AsMut<$inheritance_chain>`][AsMut]
28/// - [`Borrow<$inheritance_chain>`][core::borrow::Borrow]
29/// - [`BorrowMut<$inheritance_chain>`][core::borrow::BorrowMut]
30///
31/// The macro allows specifying zero-sized fields like [`PhantomData`] on the
32/// struct.
33///
34/// You can add most attributes to the class, including `#[cfg(...)]`,
35/// `#[derive(...)]` and doc comments (but not ABI-modifying attributes like
36/// `#[repr(...)]`).
37///
38/// [rustfmt-macros]: https://github.com/rust-lang/rustfmt/discussions/5437
39/// [`PhantomData`]: core::marker::PhantomData
40///
41///
42/// ## `ClassType` implementation
43///
44/// The syntax of this macro neatly documents that it implements the
45/// [`ClassType`] trait for you, though to do so you need to provide it the
46/// following:
47/// - The [`Super`] class.
48///
49///   Due to Rust trait limitations, specifying e.g. the superclass `NSData`
50///   would not give you the ability to convert via. `AsRef` to `NSObject`.
51///   Therefore, you may optionally specify additional parts of the
52///   inheritance chain using an `#[inherits(...)]` attribute.
53/// - The class' [`Mutability`].
54/// - Optionally, the class' [`NAME`] - if not specified, this will default to
55///   the struct name.
56///
57/// You may add `#[cfg(...)]` attributes to the `ClassType` impl, and then it
58/// will work as expected. No other attributes are supported.
59///
60/// [`ClassType`]: crate::ClassType
61/// [`Super`]: crate::ClassType::Super
62/// [`Mutability`]: crate::ClassType::Mutability
63/// [`NAME`]: crate::ClassType::NAME
64///
65///
66/// # Safety
67///
68/// This macro implements the three unsafe traits [`RefEncode`], [`Message`]
69/// and [`ClassType`] for you, and while it can ensure most of the required
70/// properties in those, it cannot ensure all of them.
71///
72/// In particular, when writing `unsafe` on `impl ClassType`, you must ensure
73/// that:
74/// 1. [`ClassType::Super`] is correct.
75/// 2. [`ClassType::Mutability`] is correct.
76///
77///    See [`ClassType`'s safety section][ClassType#safety] for further
78///    details on what this entails.
79///
80/// [`RefEncode`]: crate::encode::RefEncode
81/// [`Message`]: crate::Message
82/// [`ClassType::Super`]: crate::ClassType::Super
83/// [`ClassType::Mutability`]: crate::ClassType::Mutability
84/// [ClassType#safety]: crate::ClassType#safety
85///
86///
87/// # Examples
88///
89/// Create a new type to represent the `NSFormatter` class (for demonstration,
90/// `objc2_foundation::NSFormatter` exist for exactly this purpose).
91///
92/// ```
93/// # #[cfg(not_available)]
94/// use objc2_foundation::{NSCoding, NSCopying, NSObjectProtocol};
95/// # use objc2::runtime::NSObjectProtocol;
96/// use objc2::rc::Retained;
97/// use objc2::runtime::NSObject;
98/// use objc2::{extern_class, msg_send_id, mutability, ClassType};
99///
100/// extern_class!(
101///     /// An example description.
102///     #[derive(PartialEq, Eq, Hash)] // Uses the superclass' implementation
103///     // Specify the class and struct name to be used
104///     pub struct NSFormatter;
105///
106///     // Specify the superclass, in this case `NSObject`
107///     unsafe impl ClassType for NSFormatter {
108///         type Super = NSObject;
109///         type Mutability = mutability::InteriorMutable;
110///         // Optionally, specify the name of the class, if it differs from
111///         // the struct name.
112///         // const NAME: &'static str = "NSFormatter";
113///     }
114/// );
115///
116/// // Note: We have to specify the protocols for the superclasses as well,
117/// // since Rust doesn't do inheritance.
118/// unsafe impl NSObjectProtocol for NSFormatter {}
119/// # #[cfg(not_available)]
120/// unsafe impl NSCopying for NSFormatter {}
121/// # #[cfg(not_available)]
122/// unsafe impl NSCoding for NSFormatter {}
123///
124/// fn main() {
125///     // Provided by the implementation of `ClassType`
126///     let cls = NSFormatter::class();
127///
128///     // `NSFormatter` implements `Message`:
129///     let obj: Retained<NSFormatter> = unsafe { msg_send_id![cls, new] };
130/// }
131/// ```
132///
133/// Represent the `NSDateFormatter` class, using the `NSFormatter` type we
134/// declared previously to specify as its superclass.
135///
136/// ```
137/// # #[cfg(not_available)]
138/// use objc2_foundation::{NSCoding, NSCopying, NSObjectProtocol};
139/// # use objc2::runtime::NSObjectProtocol;
140/// use objc2::runtime::NSObject;
141/// use objc2::{extern_class, mutability, ClassType};
142/// #
143/// # extern_class!(
144/// #     #[derive(PartialEq, Eq, Hash)]
145/// #     pub struct NSFormatter;
146/// #
147/// #     unsafe impl ClassType for NSFormatter {
148/// #         type Super = NSObject;
149/// #         type Mutability = mutability::InteriorMutable;
150/// #     }
151/// # );
152///
153/// extern_class!(
154///     #[derive(PartialEq, Eq, Hash)]
155///     pub struct NSDateFormatter;
156///
157///     unsafe impl ClassType for NSDateFormatter {
158///         // Specify the correct inheritance chain
159///         #[inherits(NSObject)]
160///         type Super = NSFormatter;
161///         type Mutability = mutability::InteriorMutable;
162///     }
163/// );
164///
165/// // Similarly, we can specify the protocols that this implements here:
166/// unsafe impl NSObjectProtocol for NSFormatter {}
167/// # #[cfg(not_available)]
168/// unsafe impl NSCopying for NSDateFormatter {}
169/// # #[cfg(not_available)]
170/// unsafe impl NSCoding for NSDateFormatter {}
171/// ```
172///
173/// See the source code of `objc2-foundation` for many more examples.
174#[doc(alias = "@interface")]
175#[macro_export]
176macro_rules! extern_class {
177    // No fields
178    (
179        $(#[$m:meta])*
180        $v:vis struct $name:ident;
181
182        $(#[$impl_m:meta])*
183        unsafe impl ClassType for $for:ty {
184            $(#[inherits($($inheritance_rest:ty),+)])?
185            type Super = $superclass:ty;
186            type Mutability = $mutability:ty;
187
188            $(const NAME: &'static str = $name_const:expr;)?
189        }
190    ) => {
191        // Shorthand syntax for the following
192        $crate::extern_class!(
193            $(#[$m])*
194            $v struct $name {}
195
196            $(#[$impl_m])*
197            unsafe impl ClassType for $for {
198                $(#[inherits($($inheritance_rest),+)])?
199                type Super = $superclass;
200                type Mutability = $mutability;
201
202                $(const NAME: &'static str = $name_const;)?
203            }
204        );
205    };
206    (
207        $(#[$m:meta])*
208        $v:vis struct $name:ident {
209            $($field_vis:vis $field:ident: $field_ty:ty,)*
210        }
211
212        $(#[$impl_m:meta])*
213        unsafe impl ClassType for $for:ty {
214            $(#[inherits($($inheritance_rest:ty),+)])?
215            type Super = $superclass:ty;
216            type Mutability = $mutability:ty;
217
218            $(const NAME: &'static str = $name_const:expr;)?
219        }
220    ) => {
221        $crate::__inner_extern_class!(
222            $(#[$m])*
223            $v struct $name<> {
224                __superclass: $superclass,
225                $($field_vis $field: $field_ty,)*
226            }
227
228            $(#[$impl_m])*
229            unsafe impl<> ClassType for $for {
230                $(#[inherits($($inheritance_rest),+)])?
231                type Super = $superclass;
232                type Mutability = $mutability;
233
234                fn as_super(&self) -> &Self::Super {
235                    &self.__superclass
236                }
237
238                fn as_super_mut(&mut self) -> &mut Self::Super {
239                    &mut self.__superclass
240                }
241
242                $(const NAME: &'static str = $name_const;)?
243            }
244        );
245
246        $(#[$impl_m])*
247        const _: () = {
248            if $crate::__macro_helpers::size_of::<$name>() != 0 {
249                $crate::__macro_helpers::panic!($crate::__macro_helpers::concat!(
250                    "the struct ",
251                    $crate::__macro_helpers::stringify!($name),
252                    " is not zero-sized!",
253                ))
254            }
255        };
256    };
257}

260#[macro_export]
261macro_rules! __impl_as_ref_borrow {
262    {
263        impl ($($t:tt)*) for $for:ty {
264            fn as_ref($($self:tt)*) $ref:block
265            fn as_mut($($self_mut:tt)*) $mut:block
266        }
267
268        ()
269    } => {};
270    {
271        impl ($($t:tt)*) for $for:ty {
272            fn as_ref($($self:tt)*) $ref:block
273            fn as_mut($($self_mut:tt)*) $mut:block
274        }
275
276        ($item:ty, $($tail:ty,)*)
277    } => {
278        impl<$($t)*> $crate::__macro_helpers::AsRef<$item> for $for {
279            #[inline]
280            fn as_ref($($self)*) -> &$item $ref
281        }
282
283        impl<$($t)*> $crate::__macro_helpers::AsMut<$item> for $for {
284            #[inline]
285            fn as_mut($($self_mut)*) -> &mut $item $mut
286        }
287
288        // Borrow and BorrowMut are correct, since subclasses behaves
289        // identical to the class they inherit (message sending doesn't care).
290        //
291        // In particular, `Eq`, `Ord` and `Hash` all give the same results
292        // after borrow.
293
294        impl<$($t)*> $crate::__macro_helpers::Borrow<$item> for $for {
295            #[inline]
296            fn borrow($($self)*) -> &$item $ref
297        }
298
299        impl<$($t)*> $crate::__macro_helpers::BorrowMut<$item> for $for {
300            #[inline]
301            fn borrow_mut($($self_mut)*) -> &mut $item $mut
302        }
303
304        $crate::__impl_as_ref_borrow! {
305            impl ($($t)*) for $for {
306                fn as_ref($($self)*) $ref
307                fn as_mut($($self_mut)*) $mut
308            }
309
310            ($($tail,)*)
311        }
312    };
313    // TODO: Expose a generic variant of the macro.
314}

319#[macro_export]
320macro_rules! __inner_extern_class {
321    (
322        $(#[$m:meta])*
323        $v:vis struct $name:ident<$($t_struct:ident $(: $(?$b_sized_struct:ident)? $($b_struct:ident)? $(= $default:ty)?)?),* $(,)?> {
324            $superclass_field:ident: $superclass_field_ty:ty,
325            $($fields:tt)*
326        }
327
328        $(#[$impl_m:meta])*
329        unsafe impl<$($t_for:ident $(: $(?$b_sized_for:ident +)? $b_for:ident)?),* $(,)?> ClassType for $for:ty {
330            $(#[inherits($($inheritance_rest:ty),+ $(,)?)])?
331            type Super = $superclass:ty;
332            type Mutability = $mutability:ty;
333
334            fn as_super(&$as_super_self:ident) -> &Self::Super $as_super:block
335            fn as_super_mut(&mut $as_super_mut_self:ident) -> &mut Self::Super $as_super_mut:block
336
337            $(const NAME: &'static str = $name_const:expr;)?
338        }
339    ) => {
340        $(#[$m])*
341        #[repr(C)]
342        $v struct $name<$($t_struct $(: $(?$b_sized_struct)? $($b_struct)? $(= $default)?)?),*> {
343            $superclass_field: $superclass_field_ty,
344            $($fields)*
345        }
346
347        $crate::__extern_class_impl_traits! {
348            $(#[$impl_m])*
349            unsafe impl ($($t_for $(: $(?$b_sized_for +)? $b_for)?),*) for $for {
350                INHERITS = [$superclass, $($($inheritance_rest,)+)? $crate::runtime::AnyObject];
351
352                fn as_super(&$as_super_self) $as_super
353                fn as_super_mut(&mut $as_super_mut_self) $as_super_mut
354            }
355        }
356
357        $(#[$impl_m])*
358        unsafe impl<$($t_for $(: $(?$b_sized_for +)? $b_for)?),*> ClassType for $for {
359            type Super = $superclass;
360            type Mutability = $mutability;
361            const NAME: &'static $crate::__macro_helpers::str = $crate::__select_name!($name; $($name_const)?);
362
363            #[inline]
364            fn class() -> &'static $crate::runtime::AnyClass {
365                $crate::__macro_helpers::assert_mutability_matches_superclass_mutability::<Self>();
366
367                $crate::__class_inner!(
368                    $crate::__select_name!($name; $($name_const)?),
369                    $crate::__hash_idents!($name)
370                )
371            }
372
373            #[inline]
374            fn as_super(&$as_super_self) -> &Self::Super $as_super
375
376            #[inline]
377            fn as_super_mut(&mut $as_super_mut_self) -> &mut Self::Super $as_super_mut
378        }
379    };
380}

382#[doc(hidden)]
383#[macro_export]
384macro_rules! __extern_class_impl_traits {
385    (
386        $(#[$impl_m:meta])*
387        unsafe impl ($($t:tt)*) for $for:ty {
388            INHERITS = [$superclass:ty $(, $inheritance_rest:ty)*];
389
390            fn as_super(&$as_super_self:ident) $as_super:block
391            fn as_super_mut(&mut $as_super_mut_self:ident) $as_super_mut:block
392        }
393    ) => {
394        // SAFETY:
395        // - The item is FFI-safe with `#[repr(C)]`.
396        // - The encoding is taken from the inner item, and caller verifies
397        //   that it actually inherits said object.
398        // - The rest of the struct's fields are ZSTs, so they don't influence
399        //   the layout.
400        //
401        // Be aware that very rarely, this implementation is wrong because the
402        // class' instances do not have the encoding `Encoding::Object`.
403        //
404        // A known case is that `NSAutoreleasePool` has a different encoding.
405        // This should be fairly problem-free though, since that is still
406        // valid in Objective-C to represent that class' instances as
407        // `NSObject*`.
408        $(#[$impl_m])*
409        unsafe impl<$($t)*> $crate::RefEncode for $for {
410            const ENCODING_REF: $crate::Encoding
411                = <$superclass as $crate::RefEncode>::ENCODING_REF;
412        }
413
414        // SAFETY: This is a newtype wrapper over `AnyObject` (we even ensure
415        // that `AnyObject` is always last in our inheritance tree), so it is
416        // always safe to reinterpret as that.
417        //
418        // That the object must work with standard memory management is
419        // properly upheld by the fact that the superclass is required by
420        // `assert_mutability_matches_superclass_mutability` to implement
421        // `ClassType`, and hence must be a subclass of one of `NSObject`,
422        // `NSProxy` or some other class that ensures this (e.g. the object
423        // itself is not a root class).
424        $(#[$impl_m])*
425        unsafe impl<$($t)*> $crate::Message for $for {}
426
427        // SAFETY: An instance can always be _used_ in exactly the same way as
428        // its superclasses (though not necessarily _constructed_ in the same
429        // way, but `Deref` doesn't allow this).
430        //
431        // Remember; while we (the Rust side) may intentionally be forgetting
432        // which instance we're holding, the Objective-C side will remember,
433        // and will always dispatch to the correct method implementations.
434        //
435        // Any lifetime information that the object may have been holding is
436        // safely kept in the returned reference.
437        //
438        // Generics are discarded (for example in the case of `&NSArray<T>` to
439        // `&NSObject`), but if the generic contained a lifetime, that
440        // lifetime is still included in the returned reference, and is not
441        // erasable by e.g. `ClassType::retain` since `NSObject` does not
442        // allow that.
443        //
444        // Note that you can have two different variables pointing to the same
445        // object, `x: &T` and `y: &T::Target`, and this would be perfectly
446        // safe!
447        $(#[$impl_m])*
448        impl<$($t)*> $crate::__macro_helpers::Deref for $for {
449            type Target = $superclass;
450
451            #[inline]
452            fn deref(&$as_super_self) -> &Self::Target $as_super
453        }
454
455        // SAFETY: Mutability does not change anything in the above
456        // consideration, the lifetime of `&mut Self::Target` is still tied to
457        // `&mut self`.
458        //
459        // Usually we don't want to allow `&mut` of immutable objects like
460        // `NSString`, because their `NSCopying` implementation returns the
461        // same object, and would violate aliasing rules.
462        //
463        // But `&mut NSMutableString` -> `&mut NSString` safe, since the
464        // `NSCopying` implementation of `NSMutableString` is still used on
465        // the `&mut NSString`, and that is guaranteed to return a different
466        // object.
467        $(#[$impl_m])*
468        impl<$($t)*> $crate::__macro_helpers::DerefMut for $for {
469            #[inline]
470            fn deref_mut(&mut $as_super_mut_self) -> &mut Self::Target $as_super_mut
471        }
472
473        $(#[$impl_m])*
474        impl<$($t)*> $crate::__macro_helpers::AsRef<Self> for $for {
475            #[inline]
476            fn as_ref(&self) -> &Self {
477                self
478            }
479        }
480
481        $(#[$impl_m])*
482        impl<$($t)*> $crate::__macro_helpers::AsMut<Self> for $for {
483            #[inline]
484            fn as_mut(&mut self) -> &mut Self {
485                self
486            }
487        }
488
489        // Assume the meta attributes are all `cfg` attributes
490        $(#[$impl_m])*
491        $crate::__impl_as_ref_borrow! {
492            impl ($($t)*) for $for {
493                fn as_ref(&self) {
494                    // Triggers Deref coercion depending on return type
495                    &*self
496                }
497                fn as_mut(&mut self) {
498                    // Triggers Deref coercion depending on return type
499                    &mut *self
500                }
501            }
502
503            ($superclass, $($inheritance_rest,)*)
504        }
505    };
