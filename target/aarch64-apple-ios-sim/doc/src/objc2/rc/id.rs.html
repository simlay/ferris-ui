13/// A reference counted pointer type for Objective-C objects.
14///
15/// [`Retained`] strongly references or "retains" the given object `T`, and
16/// decrements the retain count or "releases" it again when dropped, thereby
17/// ensuring it will be deallocated at the right time.
18///
19/// The type `T` inside `Retained<T>` can be anything that implements
20/// [`Message`].
21///
22/// `T`'s [`ClassType`] implementation (if any) determines whether it is
23/// mutable, and by extension whether `Retained<T>` is mutable.
24///
25/// This can usually be gotten from one of the methods in the framework
26/// crates, but can be created manually with the [`msg_send_id!`] macro (or
27/// even more manually, with the [`Retained::retain`], [`Retained::from_raw`]
28/// or [`Retained::retain_autoreleased`] methods).
29///
30/// [`msg_send_id!`]: crate::msg_send_id
31///
32///
33/// # Comparison to `std` types
34///
35/// `Retained<T>` can be thought of as kind of a weird combination of [`Arc`]
36/// and [`Box`]:
37///
38/// If `T` implements [`IsMutable`] (like it does on `NSMutableString` and
39/// `NSMutableArray<_>`), `Retained<T>` acts like `Box<T>`, and allows mutable
40/// / unique access to the type.
41///
42/// Otherwise, which is the most common case, `Retained<T>` acts like
43/// `Arc<T>`, and allows cloning by bumping the reference count.
44///
45/// [`Arc`]: alloc::sync::Arc
46/// [`Box`]: alloc::boxed::Box
47///
48///
49/// # Forwarding implementations
50///
51/// Since `Retained<T>` is a smart pointer, it [`Deref`]s to `T`, and
52/// similarly implements [`DerefMut`] when mutable.
53///
54/// It also forwards the implementation of a bunch of standard library traits
55/// such as [`PartialEq`], [`AsRef`], and so on, so that it becomes possible
56/// to use e.g. `Retained<NSString>` as if it was `NSString`. (Having
57/// `NSString` directly is not possible since Objective-C objects cannot live
58/// on the stack, but instead must reside on the heap).
59///
60/// Note that because of current limitations in the Rust trait system, some
61/// traits like [`Default`], [`IntoIterator`], [`FromIterator`], [`From`] and
62/// [`Into`] are not directly implementable on `NSString`; for that use-case,
63/// we instead provide the [`DefaultRetained`], [`RetainedIntoIterator`] and
64/// [`RetainedFromIterator`] traits, which make some of the the aforementioned
65/// traits implementable on `Retained`.
66///
67/// [`DefaultRetained`]: crate::rc::DefaultRetained
68/// [`RetainedIntoIterator`]: crate::rc::RetainedIntoIterator
69/// [`RetainedFromIterator`]: crate::rc::RetainedFromIterator
70///
71///
72/// # Memory layout
73///
74/// This is guaranteed to have the same size and alignment as a pointer to the
75/// object, `*const T`.
76///
77/// Additionally, it participates in the null-pointer optimization, that is,
78/// `Option<Retained<T>>` is guaranteed to have the same size as
79/// `Retained<T>`.
80///
81///
82/// # Example
83///
84/// Various usage of `Retained` on an immutable object.
85///
86/// ```
87/// # #[cfg(not_available)]
88/// use objc2_foundation::{NSObject, NSString};
89/// # use objc2::runtime::NSObject;
90/// use objc2::rc::Retained;
91/// use objc2::{ClassType, msg_send_id};
92/// #
93/// # objc2::extern_class!(
94/// #     pub struct NSString;
95/// #
96/// #     unsafe impl ClassType for NSString {
97/// #         type Super = NSObject;
98/// #         // This is wrong, but let's do it for the example
99/// #         type Mutability = objc2::mutability::Immutable;
100/// #     }
101/// # );
102///
103/// // Use `msg_send_id!` to create an `Retained` with correct memory management
104/// //
105/// // SAFETY: The types are correct, and it is safe to call the `new`
106/// // selector on `NSString`.
107/// let string: Retained<NSString> = unsafe { msg_send_id![NSString::class(), new] };
108/// // Or:
109/// // let string = NSString::new();
110///
111/// // Methods on `NSString` is usable via. `Deref`
112/// #[cfg(not_available)]
113/// assert_eq!(string.len(), 0);
114///
115/// // Bump the reference count of the object (possible because the object is
116/// // immutable, would not be possible for `NSMutableString`).
117/// let another_ref: Retained<NSString> = string.clone();
118///
119/// // Convert one of the references to a reference to `NSObject` instead
120/// let obj: Retained<NSObject> = Retained::into_super(string);
121///
122/// // And use the `Debug` impl from that
123/// assert_eq!(format!("{obj:?}"), "");
124///
125/// // Finally, the `Retained`s go out of scope, the reference counts are
126/// // decreased, and the string will deallocate
127/// ```
128#[repr(transparent)]
129#[doc(alias = "id")]
130#[doc(alias = "Id")]
131#[doc(alias = "StrongPtr")]

134    /// A pointer to the contained object. The pointer is always retained.
135    ///
136    /// It is important that this is `NonNull`, since we want to dereference
137    /// it later, and be able to use the null-pointer optimization.
138    ///
139    /// Additionally, covariance is correct because we're either the unique
140    /// owner of `T`, or `T` is immutable.
141    ptr: NonNull<T>,
142    /// Necessary for dropck even though we never actually run T's destructor,
143    /// because it might have a `dealloc` that assumes that contained
144    /// references outlive the type.
145    ///
146    /// See <https://doc.rust-lang.org/nightly/nomicon/phantom-data.html>
147    item: PhantomData<T>,
148    /// Marks the type as !UnwindSafe. Later on we'll re-enable this.
149    ///
150    /// See <https://github.com/rust-lang/rust/issues/93367> for why this is
151    /// required.
152    notunwindsafe: PhantomData<&'static mut ()>,
153}

175impl<T: ?Sized + Message> Retained<T> {
176    /// Construct an [`Retained`] from a pointer that already has +1 retain count.
177    ///
178    /// Returns `None` if the pointer was NULL.
179    ///
180    /// This is useful when you have a retain count that has been handed off
181    /// from somewhere else, usually Objective-C methods like `init`, `alloc`,
182    /// `new`, `copy`, or methods with the `ns_returns_retained` attribute.
183    ///
184    /// If you do not have +1 retain count, such as if your object was
185    /// retrieved from other methods than the ones noted above, use
186    /// [`Retained::retain`] instead.
187    ///
188    ///
189    /// # Safety
190    ///
191    /// You must uphold the same requirements as described in [`Retained::retain`].
192    ///
193    /// Additionally, you must ensure the given object pointer has +1 retain
194    /// count.
195    ///
196    ///
197    /// # Example
198    ///
199    /// Comparing different ways of creating a new `NSObject`.
200    ///
201    /// ```
202    /// use objc2::rc::Retained;
203    /// use objc2::runtime::NSObject;
204    /// use objc2::{msg_send, msg_send_id, ClassType};
205    ///
206    /// // Manually using `msg_send!` and `Retained::from_raw`
207    /// let obj: *mut NSObject = unsafe { msg_send![NSObject::class(), alloc] };
208    /// let obj: *mut NSObject = unsafe { msg_send![obj, init] };
209    /// // SAFETY: `-[NSObject init]` returns +1 retain count
210    /// let obj: Retained<NSObject> = unsafe { Retained::from_raw(obj).unwrap() };
211    ///
212    /// // Or with `msg_send_id!`
213    /// let obj: Retained<NSObject> = unsafe { msg_send_id![NSObject::alloc(), init] };
214    ///
215    /// // Or using the `NSObject::new` method
216    /// let obj = NSObject::new();
217    /// ```
218    #[inline]
219    // Note: We don't take a reference as a parameter since it would be too
220    // easy to accidentally create two aliasing mutable references.
221    pub unsafe fn from_raw(ptr: *mut T) -> Option<Self> {
222        // Should optimize down to a noop.
223        // SAFETY: Upheld by the caller
224        NonNull::new(ptr).map(|ptr| unsafe { Retained::new_nonnull(ptr) })
225    }
226
227    /// Deprecated alias for [`Retained::from_raw`], see that for details.
228    ///
229    ///
230    /// # Safety
231    ///
232    /// Same as [`Retained::from_raw`].
233    #[deprecated = "use the more descriptive name `Retained::from_raw` instead"]
234    #[inline]
235    pub unsafe fn new(ptr: *mut T) -> Option<Self> {
236        // SAFETY: Upheld by caller
237        unsafe { Self::from_raw(ptr) }
238    }
239
240    /// Consumes the `Retained`, returning a raw pointer with +1 retain count.
241    ///
242    /// After calling this function, the caller is responsible for the memory
243    /// previously managed by the `Retained`.
244    ///
245    /// This is effectively the opposite of [`Retained::from_raw`], see that for
246    /// more details on when this function is useful.
247    ///
248    ///
249    /// # Examples
250    ///
251    /// Converting an `Retained` to a pointer and back.
252    ///
253    /// ```
254    /// use objc2::rc::Retained;
255    /// use objc2::runtime::NSObject;
256    ///
257    /// let obj = NSObject::new();
258    /// let ptr = Retained::into_raw(obj);
259    /// // SAFETY: The pointer is valid, and has +1 retain count from above.
260    /// let obj = unsafe { Retained::from_raw(ptr) }.unwrap();
261    /// ```
262    #[inline]
263    pub fn into_raw(this: Self) -> *mut T {
264        ManuallyDrop::new(this).ptr.as_ptr()
265    }
266
267    /// Returns a raw pointer to the object.
268    ///
269    /// The pointer is valid for at least as long as the `Retained` is held.
270    ///
271    /// See [`Retained::as_mut_ptr`] for the mutable equivalent.
272    ///
273    /// This is an associated method, and must be called as `Retained::as_ptr(obj)`.
274    #[inline]
275    pub fn as_ptr(this: &Self) -> *const T {
276        this.ptr.as_ptr()
277    }
278
279    /// Returns a raw mutable pointer to the object.
280    ///
281    /// The pointer is valid for at least as long as the `Retained` is held.
282    ///
283    /// See [`Retained::as_ptr`] for the immutable equivalent.
284    ///
285    /// This is an associated method, and must be called as
286    /// `Retained::as_mut_ptr(obj)`.
287    #[inline]
288    #[allow(unknown_lints)] // New lint below
289    #[allow(clippy::needless_pass_by_ref_mut)]
290    pub fn as_mut_ptr(this: &mut Self) -> *mut T
291    where
292        T: IsMutable,
293    {
294        this.ptr.as_ptr()
295    }
296
297    #[inline]
298    pub(crate) fn consume_as_ptr_option(this: Option<Self>) -> *mut T
299    where
300        T: Sized,
301    {
302        this.map(|this| Retained::into_raw(this))
303            .unwrap_or_else(ptr::null_mut)
304    }
305}

309    /// Convert the type of the given object to another.
310    ///
311    /// This is equivalent to a `cast` between two pointers.
312    ///
313    /// See [`Retained::into_super`] and [`ProtocolObject::from_retained`] for
314    /// safe alternatives.
315    ///
316    /// This is common to do when you know that an object is a subclass of
317    /// a specific class (e.g. casting an instance of `NSString` to `NSObject`
318    /// is safe because `NSString` is a subclass of `NSObject`).
319    ///
320    /// All `'static` objects can safely be cast to [`AnyObject`], since that
321    /// assumes no specific class.
322    ///
323    /// [`AnyObject`]: crate::runtime::AnyObject
324    /// [`ProtocolObject::from_retained`]: crate::runtime::ProtocolObject::from_retained
325    ///
326    ///
327    /// # Safety
328    ///
329    /// You must ensure that the object can be reinterpreted as the given
330    /// type.
331    ///
332    /// If `T` is not `'static`, you must ensure that `U` ensures that the
333    /// data contained by `T` is kept alive for as long as `U` lives.
334    ///
335    /// Additionally, you must ensure that any safety invariants that the new
336    /// type has are upheld.
337    ///
338    /// Note that it is not in general safe to cast e.g. `Retained<NSString>` to
339    /// `Retained<NSMutableString>`, even if you've checked at runtime that the
340    /// object is an instance of `NSMutableString`! This is because
341    /// `Retained<NSMutableString>` assumes the string is unique, whereas it may
342    /// have been cloned while being an `Retained<NSString>`.
343    #[inline]
344    pub unsafe fn cast<U: Message>(this: Self) -> Retained<U> {
345        let ptr = ManuallyDrop::new(this).ptr.cast();
346        // SAFETY: The object is forgotten, so we have +1 retain count.
347        //
348        // Caller verifies that the returned object is of the correct type.
349        unsafe { Retained::new_nonnull(ptr) }
350    }

352    /// Retain the pointer and construct an [`Retained`] from it.
353    ///
354    /// Returns `None` if the pointer was NULL.
355    ///
356    /// This is useful when you have been given a pointer to an object from
357    /// some API, and you would like to ensure that the object stays around
358    /// while you work on it.
359    ///
360    /// For normal Objective-C methods, you may want to use
361    /// [`Retained::retain_autoreleased`] instead, as that is usually more
362    /// performant.
363    ///
364    /// See [`ClassType::retain`] for a safe alternative.
365    ///
366    ///
367    /// # Safety
368    ///
369    /// If the object is mutable, the caller must ensure that there are no
370    /// other pointers or references to the object, such that the returned
371    /// `Retained` pointer is unique.
372    ///
373    /// Additionally, the pointer must be valid as a reference (aligned,
374    /// dereferencable and initialized, see the [`std::ptr`] module for more
375    /// information) or NULL.
376    ///
377    /// Finally, you must ensure that any data that `T` may reference lives
378    /// for at least as long as `T`.
379    ///
380    /// [`std::ptr`]: core::ptr
381    #[doc(alias = "objc_retain")]
382    #[inline]
383    pub unsafe fn retain(ptr: *mut T) -> Option<Retained<T>> {
384        // SAFETY: The caller upholds that the pointer is valid
385        let res: *mut T = unsafe { objc_retain_fast(ptr.cast()) }.cast();
386        debug_assert_eq!(res, ptr, "objc_retain did not return the same pointer");
387        // SAFETY: We just retained the object, so it has +1 retain count
388        unsafe { Self::from_raw(res) }
389    }

391    /// Retains a previously autoreleased object pointer.
392    ///
393    /// This is useful when calling Objective-C methods that return
394    /// autoreleased objects, see [Cocoa's Memory Management Policy][mmRules].
395    ///
396    /// This has exactly the same semantics as [`Retained::retain`], except it can
397    /// sometimes avoid putting the object into the autorelease pool, possibly
398    /// yielding increased speed and reducing memory pressure.
399    ///
400    /// Note: This relies heavily on being inlined right after [`msg_send!`],
401    /// be careful to not accidentally require instructions between these.
402    ///
403    /// [mmRules]: https://developer.apple.com/library/archive/documentation/Cocoa/Conceptual/MemoryMgmt/Articles/mmRules.html
404    /// [`msg_send!`]: crate::msg_send
405    ///
406    ///
407    /// # Safety
408    ///
409    /// Same as [`Retained::retain`].
410    #[doc(alias = "objc_retainAutoreleasedReturnValue")]
411    #[inline]
412    pub unsafe fn retain_autoreleased(ptr: *mut T) -> Option<Retained<T>> {
413        // Add magic nop instruction to participate in the fast autorelease
414        // scheme.
415        //
416        // See `callerAcceptsOptimizedReturn` in `objc-object.h`:
417        // https://github.com/apple-oss-distributions/objc4/blob/objc4-838/runtime/objc-object.h#L1209-L1377
418        //
419        // We will unconditionally emit these instructions, even if they end
420        // up being unused (for example because we're unlucky with inlining,
421        // some other work is done between the objc_msgSend and this, or the
422        // runtime version is too old to support it).
423        //
424        // It may seem like there should be a better way to do this, but
425        // emitting raw assembly is exactly what Clang and Swift does:
426        // swiftc: https://github.com/apple/swift/blob/swift-5.5.3-RELEASE/lib/IRGen/GenObjC.cpp#L148-L173
427        // Clang: https://github.com/llvm/llvm-project/blob/889317d47b7f046cf0e68746da8f7f264582fb5b/clang/lib/CodeGen/CGObjC.cpp#L2339-L2373
428        //
429        // Note that LLVM may sometimes insert extra instructions between the
430        // assembly and the `objc_retainAutoreleasedReturnValue` call,
431        // especially when doing tail calls and it needs to clean up the
432        // function frame. Unsure how to avoid this in a performant manner?
433        // Maybe force not doing tail calls by inserting assembly to do the
434        // call manually?
435        //
436        // Resources:
437        // - https://www.mikeash.com/pyblog/friday-qa-2011-09-30-automatic-reference-counting.html
438        // - https://www.galloway.me.uk/2012/02/how-does-objc_retainautoreleasedreturnvalue-work/
439        // - https://github.com/gfx-rs/metal-rs/issues/222
440        // - https://news.ycombinator.com/item?id=29311736
441        // - https://stackoverflow.com/a/23765612
442        //
443        // SAFETY:
444        // Based on https://doc.rust-lang.org/stable/reference/inline-assembly.html#rules-for-inline-assembly
445        //
446        // We don't care about the value of the register (so it's okay to be
447        // undefined), and its value is preserved.
448        //
449        // nomem: No reads or writes to memory are performed (this `mov`
450        //   operates entirely on registers).
451        // preserves_flags: `mov` doesn't modify any flags.
452        // nostack: We don't touch the stack.
453
454        // Only worth doing on the Apple runtime.
455        // Not supported on TARGET_OS_WIN32.
456        #[cfg(target_vendor = "apple")]
457        {
458            // Supported since macOS 10.7.
459            #[cfg(target_arch = "x86_64")]
460            {
461                // x86_64 looks at the next call instruction.
462                //
463                // This is expected to be a PLT entry - if the user specifies
464                // `-Zplt=no`, a GOT entry will be created instead, and this
465                // will not work.
466            }
467
468            // Supported since macOS 10.8.
469            #[cfg(target_arch = "arm")]
470            unsafe {
471                core::arch::asm!("mov r7, r7", options(nomem, preserves_flags, nostack))
472            };
473
474            // Supported since macOS 10.10.
475            //
476            // On macOS 13.0 / iOS 16.0 / tvOS 16.0 / watchOS 9.0, the runtime
477            // instead checks the return pointer address, so we no longer need
478            // to emit these extra instructions, see this video from WWDC22:
479            // https://developer.apple.com/videos/play/wwdc2022/110363/
480            #[cfg(all(target_arch = "aarch64", not(feature = "unstable-apple-new")))]
481            unsafe {
482                // Same as `mov x29, x29`.
483                core::arch::asm!("mov fp, fp", options(nomem, preserves_flags, nostack))
484            };
485
486            // Supported since macOS 10.12.
487            #[cfg(target_arch = "x86")]
488            unsafe {
489                core::arch::asm!("mov ebp, ebp", options(nomem, preserves_flags, nostack))
490            };
491        }
492
493        // SAFETY: Same as `Retained::retain`, this is just an optimization.
494        let res: *mut T = unsafe { ffi::objc_retainAutoreleasedReturnValue(ptr.cast()) }.cast();
495
496        // Ideally, we'd be able to specify that the above call should never
497        // be tail-call optimized (become a `jmp` instruction instead of a
498        // `call`); Rust doesn't really have a way of doing this currently, so
499        // we emit a `nop` to make such tail-call optimizations less likely to
500        // occur.
501        //
502        // This is brittle! We should find a better solution!
503        #[cfg(all(target_vendor = "apple", target_arch = "x86_64"))]
504        {
505            // SAFETY: Similar to above.
506            unsafe { core::arch::asm!("nop", options(nomem, preserves_flags, nostack)) };
507            // TODO: Possibly more efficient alternative? Also consider PLT.
508            // #![feature(asm_sym)]
509            // core::arch::asm!(
510            //     "mov rdi, rax",
511            //     "call {}",
512            //     sym objc2::ffi::objc_retainAutoreleasedReturnValue,
513            //     inout("rax") obj,
514            //     clobber_abi("C"),
515            // );
516        }
517
518        debug_assert_eq!(
519            res, ptr,
520            "objc_retainAutoreleasedReturnValue did not return the same pointer"
521        );
522
523        // SAFETY: Same as `Retained::retain`.
524        unsafe { Self::from_raw(res) }
525    }

527    /// Autoreleases the [`Retained`], returning a pointer.
528    ///
529    /// The object is not immediately released, but will be when the innermost
530    /// / current autorelease pool (given as a parameter) is drained.
531    ///
532    /// This is useful when defining your own classes and you have some error
533    /// parameter passed as `Option<&mut *mut NSError>`, and you want to
534    /// create and autorelease an error before returning.
535    ///
536    /// See [`Retained::autorelease`] and [`Retained::autorelease_mut`] for alternatives
537    /// that yield safe references.
538    ///
539    /// This is an associated method, and must be called as
540    /// `Retained::autorelease_ptr(obj)`.
541    #[doc(alias = "objc_autorelease")]
542    #[must_use = "if you don't intend to use the object any more, drop it as usual"]
543    #[inline]
544    pub fn autorelease_ptr(this: Self) -> *mut T {
545        let ptr = ManuallyDrop::new(this).ptr.as_ptr();
546        // SAFETY:
547        // - The `ptr` is guaranteed to be valid and have at least one
548        //   retain count.
549        // - Because of the ManuallyDrop, we don't call the Drop
550        //   implementation, so the object won't also be released there.
551        let res: *mut T = unsafe { ffi::objc_autorelease(ptr.cast()) }.cast();
552        debug_assert_eq!(res, ptr, "objc_autorelease did not return the same pointer");
553        res
554    }

556    /// Autoreleases the [`Retained`], returning a reference bound to the pool.
557    ///
558    /// The object is not immediately released, but will be when the innermost
559    /// / current autorelease pool (given as a parameter) is drained.
560    ///
561    /// See [`Retained::autorelease_mut`] for the mutable alternative.
562    ///
563    /// This is an associated method, and must be called as
564    /// `Retained::autorelease(obj, pool)`.
565    #[doc(alias = "objc_autorelease")]
566    #[must_use = "if you don't intend to use the object any more, drop it as usual"]
567    #[inline]
568    #[allow(clippy::needless_lifetimes)]
569    pub fn autorelease<'p>(this: Self, pool: AutoreleasePool<'p>) -> &'p T {
570        let ptr = Self::autorelease_ptr(this);
571        // SAFETY: The pointer is valid as a reference
572        unsafe { pool.ptr_as_ref(ptr) }
573    }

575    /// Autoreleases the [`Retained`], returning a mutable reference bound to the
576    /// pool.
577    ///
578    /// The object is not immediately released, but will be when the innermost
579    /// / current autorelease pool (given as a parameter) is drained.
580    ///
581    /// See [`Retained::autorelease`] for the immutable alternative.
582    ///
583    /// This is an associated method, and must be called as
584    /// `Retained::autorelease_mut(obj, pool)`.
585    #[doc(alias = "objc_autorelease")]
586    #[must_use = "if you don't intend to use the object any more, drop it as usual"]
587    #[inline]
588    #[allow(clippy::needless_lifetimes)]
589    pub fn autorelease_mut<'p>(this: Self, pool: AutoreleasePool<'p>) -> &'p mut T
590    where
591        T: IsMutable,
592    {
593        let ptr = Self::autorelease_ptr(this);
594        // SAFETY:
595        // - The pointer is valid as a reference.
596        // - The object is safe as mutable because of the `T: IsMutable`
597        //   bound + the consumption of unique access to the `Retained`.
598        unsafe { pool.ptr_as_mut(ptr) }
599    }
600
601    #[inline]
602    pub(crate) fn autorelease_return_option(this: Option<Self>) -> *mut T {
603        let ptr: *mut T = this
604            .map(|this| ManuallyDrop::new(this).ptr.as_ptr())
605            .unwrap_or_else(ptr::null_mut);
606
607        // SAFETY: Same as `autorelease_inner`, this is just an optimization.
608        let res: *mut T = unsafe { ffi::objc_autoreleaseReturnValue(ptr.cast()) }.cast();
609        debug_assert_eq!(
610            res, ptr,
611            "objc_autoreleaseReturnValue did not return the same pointer"
612        );
613        res
614    }
615
616    /// Autoreleases and prepares the [`Retained`] to be returned to Objective-C.
617    ///
618    /// The object is not immediately released, but will be when the innermost
619    /// autorelease pool is drained.
620    ///
621    /// This is useful when [declaring your own methods][declare] where you
622    /// will often find yourself in need of returning autoreleased objects to
623    /// properly follow [Cocoa's Memory Management Policy][mmRules].
624    ///
625    /// To that end, you could use [`Retained::autorelease`], but that would require
626    /// you to have an [`AutoreleasePool`] object at hand, which you often
627    /// won't have in such cases. This function doesn't require a `pool`
628    /// object (but as a downside returns a pointer instead of a reference).
629    ///
630    /// This is also more efficient than a normal `autorelease`, since it
631    /// makes a best effort attempt to hand off ownership of the retain count
632    /// to a subsequent call to `objc_retainAutoreleasedReturnValue` /
633    /// [`Retained::retain_autoreleased`] in the enclosing call frame.
634    ///
635    /// This optimization relies heavily on this function being tail called,
636    /// so make sure you only call this function at the end of your method.
637    ///
638    /// [declare]: crate::declare
639    /// [mmRules]: https://developer.apple.com/library/archive/documentation/Cocoa/Conceptual/MemoryMgmt/Articles/mmRules.html
640    ///
641    ///
642    /// # Example
643    ///
644    /// Returning an `Retained` from a declared method (note: the [`declare_class!`]
645    /// macro supports doing this for you automatically).
646    ///
647    /// ```
648    /// use objc2::{class, msg_send_id, sel};
649    /// use objc2::declare::ClassBuilder;
650    /// use objc2::rc::Retained;
651    /// use objc2::runtime::{AnyClass, AnyObject, Sel};
652    ///
653    /// let mut builder = ClassBuilder::new("ExampleObject", class!(NSObject)).unwrap();
654    ///
655    /// extern "C" fn get(cls: &AnyClass, _cmd: Sel) -> *mut AnyObject {
656    ///     let obj: Retained<AnyObject> = unsafe { msg_send_id![cls, new] };
657    ///     Retained::autorelease_return(obj)
658    /// }
659    ///
660    /// unsafe {
661    ///     builder.add_class_method(
662    ///         sel!(get),
663    ///         get as extern "C" fn(_, _) -> _,
664    ///     );
665    /// }
666    ///
667    /// let cls = builder.register();
668    /// ```
669    ///
670    /// [`declare_class!`]: crate::declare_class
671    #[doc(alias = "objc_autoreleaseReturnValue")]
672    #[must_use = "if you don't intend to use the object any more, drop it as usual"]
673    #[inline]
674    pub fn autorelease_return(this: Self) -> *mut T {
675        Self::autorelease_return_option(Some(this))
676    }
677}

686        // SAFETY:
687        // - The casted-to type is a superclass of the type.
688        // - Both types are `'static`, so no lifetime information is lost
689        //   (this could maybe be relaxed a bit, but let's be on the safe side

695// TODO: Add ?Sized bound
696impl<T: Message + IsIdCloneable> Clone for Retained<T> {
697    /// Makes a clone of the shared object.
698    ///
699    /// This increases the object's reference count.
700    #[doc(alias = "objc_retain")]
701    #[doc(alias = "retain")]
702    #[inline]
703    fn clone(&self) -> Self {
704        // SAFETY:
705        // - The object is known to not be mutable due to the `IsIdCloneable`
706        //   bound. Additionally, since the object is already an `Retained`, types
707        //   like `NSObject` and `NSString` that have a mutable subclass is
708        //   also allowed (since even if the object is originally an
709        //   `Retained<NSMutableString>`, by converting it into `Retained<NSObject>` or
710        //   `Retained<NSString>` that fact is wholly forgotten, and the object
711        //   cannot ever be mutated again).
712        // - The pointer is valid.
713        let obj = unsafe { Retained::retain(self.ptr.as_ptr()) };
714        // SAFETY: `objc_retain` always returns the same object pointer, and
715        // the pointer is guaranteed non-null.
716        unsafe { obj.unwrap_unchecked() }
717    }
718}

720/// `#[may_dangle]` (see [this][dropck_eyepatch]) doesn't apply here since we
721/// don't run `T`'s destructor (rather, we want to discourage having `T`s with
722/// a destructor); and even if we did run the destructor, it would not be safe
723/// to add since we cannot verify that a `dealloc` method doesn't access
724/// borrowed data.
725///
726/// [dropck_eyepatch]: https://doc.rust-lang.org/nightly/nomicon/dropck.html#an-escape-hatch
727impl<T: ?Sized> Drop for Retained<T> {
728    /// Releases the retained object.
729    ///
730    /// The contained object's destructor (`Drop` impl, if it has one) is
731    /// never run - override the `dealloc` method instead (which
732    /// `declare_class!` does for you).
733    #[doc(alias = "objc_release")]
734    #[doc(alias = "release")]
735    #[inline]
736    fn drop(&mut self) {
737        // We could technically run the destructor for `T` when it is mutable,
738        // but that would be confusing and inconsistent since we cannot really
739        // guarantee that it is run if the `Retained<T>` is passed to Objective-C.
740
741        // SAFETY: The `ptr` is guaranteed to be valid and have at least one
742        // retain count.
743        unsafe { objc_release_fast(self.ptr.as_ptr().cast()) };
744    }
745}

850/// `Retained<T>` is always `Send` if `T` is `Send + Sync`.
851///
852/// Additionally, for mutable types, `T` doesn't have to be `Sync` (only
853/// requires `T: Send`), since it has unique access to the object.
854//
855// SAFETY:
856// - `T: Send` is required because otherwise you could move the object to
857//   another thread and let `dealloc` get called there.
858// - If `T` is not mutable, `T: Sync` is required because otherwise you could
859//   clone `&Retained<T>`, send it to another thread, and drop the clone last,
860//   making `dealloc` get called on the other thread.

862where
863    T::Mutability: private::SendSyncHelper<T>,
864    private::EquivalentType<T>: Send,
865{
866}

871/// requires `T: Sync`), since it has unique access to the object.

873// SAFETY:
