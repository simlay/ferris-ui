9/// The actual pool object.
10///
11/// It is drained when dropped.
12///
13/// This is not [`Send`], since `objc_autoreleasePoolPop` must be called on
14/// the same thread as `objc_autoreleasePoolPush`.
15///
16/// And this is not [`Sync`], since that would make `AutoreleasePool` `Send`.
17#[derive(Debug)]
18struct Pool {
19    /// This is an opaque handle, and is not guaranteed to be neither a valid
20    /// nor an aligned pointer.
21    context: *mut c_void,
22}
23
24impl Pool {
25    /// Construct a new autorelease pool.
26    ///
27    ///
28    /// # Safety
29    ///
30    /// The caller must ensure that when handing out `AutoreleasePool<'p>` to
31    /// functions that this is the innermost pool.
32    ///
33    /// Additionally, the pools must be dropped in the same order they were
34    /// created.
35    #[inline]
36    unsafe fn new() -> Self {
37        let context = unsafe { ffi::objc_autoreleasePoolPush() };
38        #[cfg(all(debug_assertions, not(feature = "unstable-autoreleasesafe")))]
39        POOLS.with(|c| c.borrow_mut().push(context));
40        Self { context }
41    }
42
43    /// Drains the autoreleasepool.
44    ///
45    /// The [clang documentation] says that `@autoreleasepool` blocks are not
46    /// drained when exceptions occur because:
47    ///
48    /// > Not draining the pool during an unwind is apparently required by the
49    /// > Objective-C exceptions implementation.
50    ///
51    /// We _would_ really like to do this anyway whenever possible, since the
52    /// unwind is probably caused by Rust, and forgetting to pop the pool will
53    /// likely leak memory.
54    ///
55    /// The above statement was true in the past, but since [revision `551.1`]
56    /// of objc4 (ships with MacOS 10.9) the exception is now retained when
57    /// `@throw` is encountered (on __OBJC2__, so e.g. not on macOS 32bit).
58    ///
59    /// So in the future, once we drop support for older versions, we should
60    /// move this to `Drop`.
61    ///
62    /// [clang documentation]: https://clang.llvm.org/docs/AutomaticReferenceCounting.html#autoreleasepool
63    /// [revision `551.1`]: https://github.com/apple-oss-distributions/objc4/blob/objc4-551.1/runtime/objc-exception.mm#L516
64    #[inline]
65    unsafe fn drain(self) {
66        unsafe { ffi::objc_autoreleasePoolPop(self.context) }
67    }
68}
69
70impl Drop for Pool {
71    #[inline]
72    fn drop(&mut self) {
73        #[cfg(all(debug_assertions, not(feature = "unstable-autoreleasesafe")))]
74        POOLS.with(|c| {
75            assert_eq!(
76                c.borrow_mut().pop(),
77                Some(self.context),
78                "popped pool that was not the innermost pool"
79            );
80        });
81    }
82}

84/// An Objective-C autorelease pool.
85///
86/// Autorelease pools are a way to store objects in a certain thread-local
87/// scope, such that they are only released at the end of said scope.
88///
89/// See [`autoreleasepool`] and [`autoreleasepool_leaking`] for how to create
90/// this.
91///
92/// This is not [`Send`] nor [`Sync`], since you can only autorelease a
93/// reference to a pool on the current thread.
94///
95///
96/// # Example
97///
98/// Use the pool as a bound on a function, and release an object to that pool.
99///
100/// ```
101/// use objc2::rc::{autoreleasepool, AutoreleasePool};
102/// use objc2::runtime::NSObject;
103/// use objc2::msg_send;
104///
105/// fn needs_lifetime_from_pool<'p>(pool: AutoreleasePool<'p>) -> &'p NSObject {
106///     let obj = NSObject::new();
107///     // Do action that returns an autoreleased object
108///     let description: *mut NSObject = unsafe { msg_send![&obj, description] };
109///     // Bound the lifetime of the reference to the pool
110///     unsafe { pool.ptr_as_ref(description) }
111/// }
112///
113/// autoreleasepool(|pool| {
114///     let obj = needs_lifetime_from_pool(pool);
115///     println!("{obj:?}");
116/// });
117/// ```
118#[derive(Debug, Copy, Clone)]
119pub struct AutoreleasePool<'pool> {
120    /// A reference to the pool.
121    ///
122    /// The lifetime is covariant, since shortening the lifetime is not a
123    /// problem (the lifetime talks about the pool, and not any data inside
124    /// the pool).
125    ///
126    /// To somewhat prove this, consider the following example using
127    /// `typed-arena` to partially implement the autorelease pool:
128    ///
129    /// ```ignore
130    /// struct Pool(typed_arena::Arena<String>);
131    ///
132    /// pub struct AutoreleasePool<'pool>(&'pool Pool);
133    ///
134    /// impl<'pool> AutoreleasePool<'pool> {
135    ///     pub fn autorelease(self, s: String) -> &'pool str {
136    ///         &*self.0.0.alloc(s)
137    ///     }
138    ///
139    ///     pub fn autorelease_mut(self, s: String) -> &'pool mut str {
140    ///         &mut *self.0.0.alloc(s)
141    ///     }
142    /// }
143    ///
144    /// pub fn autoreleasepool<F, R>(f: F) -> R
145    /// where
146    ///     F: for<'pool> FnOnce(AutoreleasePool<'pool>) -> R
147    /// {
148    ///     let pool = Pool(Default::default());
149    ///     f(AutoreleasePool(&pool))
150    /// }
151    /// ```
152    ///
153    /// Hence assuming `typed-arena` is sound, having covariance here should
154    /// also be sound.
155    #[cfg(all(debug_assertions, not(feature = "unstable-autoreleasesafe")))]
156    inner: Option<&'pool Pool>,
157    /// We use `PhantomData` here to make `AutoreleasePool` a ZST.
158    #[cfg(not(all(debug_assertions, not(feature = "unstable-autoreleasesafe"))))]
159    inner: PhantomData<&'pool Pool>,
160}

163thread_local! {
164    /// We track the thread's pools to verify that object lifetimes are only
165    /// taken from the innermost pool.
166    static POOLS: RefCell<Vec<*mut c_void>> = const { RefCell::new(Vec::new()) };
167}

169impl<'pool> AutoreleasePool<'pool> {
170    fn new(_inner: Option<&'pool Pool>) -> Self {
171        Self {
172            #[cfg(all(debug_assertions, not(feature = "unstable-autoreleasesafe")))]
173            inner: _inner,
174            #[cfg(not(all(debug_assertions, not(feature = "unstable-autoreleasesafe"))))]
175            inner: PhantomData,
176        }
177    }
178
179    /// This will be removed in a future version.
180    #[inline]
181    #[doc(hidden)]
182    pub fn __verify_is_inner(self) {
183        #[cfg(all(debug_assertions, not(feature = "unstable-autoreleasesafe")))]
184        if let Some(pool) = &self.inner {
185            POOLS.with(|c| {
186                assert_eq!(
187                    c.borrow().last(),
188                    Some(&pool.context),
189                    "tried to use lifetime from pool that was not innermost"
190                );
191            });
192        }
193    }
194
195    /// Returns a shared reference to the given autoreleased pointer object.
196    ///
197    /// This is the preferred way to make references from autoreleased
198    /// objects, since it binds the lifetime of the reference to the pool, and
199    /// does some extra checks when debug assertions are enabled.
200    ///
201    /// For the mutable counterpart see [`ptr_as_mut`](#method.ptr_as_mut).
202    ///
203    ///
204    /// # Safety
205    ///
206    /// This is equivalent to `&*ptr`, and shares the unsafety of that, except
207    /// the lifetime is bound to the pool instead of being unbounded.
208    #[inline]
209    pub unsafe fn ptr_as_ref<T: ?Sized>(self, ptr: *const T) -> &'pool T {
210        self.__verify_is_inner();
211        // SAFETY: Checked by the caller
212        unsafe { ptr.as_ref().unwrap_unchecked() }
213    }
214
215    /// Returns a unique reference to the given autoreleased pointer object.
216    ///
217    /// This is the preferred way to make mutable references from autoreleased
218    /// objects, since it binds the lifetime of the reference to the pool, and
219    /// does some extra checks when debug assertions are enabled.
220    ///
221    /// For the shared counterpart see [`ptr_as_ref`](#method.ptr_as_ref).
222    ///
223    ///
224    /// # Safety
225    ///
226    /// This is equivalent to `&mut *ptr`, and shares the unsafety of that,
227    /// except the lifetime is bound to the pool instead of being unbounded.
228    #[inline]
229    pub unsafe fn ptr_as_mut<T: ?Sized>(self, ptr: *mut T) -> &'pool mut T {
230        self.__verify_is_inner();
231        // SAFETY: Checked by the caller
232        unsafe { ptr.as_mut().unwrap_unchecked() }
233    }
234}

254auto_trait! {
255    /// Marks types that are safe to pass across the closure in an
256    /// [`autoreleasepool`].
257    ///
258    /// With the `"unstable-autoreleasesafe"` feature enabled, this is an auto
259    /// trait that is implemented for all types except [`AutoreleasePool`].
260    ///
261    /// Otherwise it is a dummy trait that is implemented for all types; the
262    /// safety invariants are checked with debug assertions instead.
263    ///
264    /// You should not normally need to implement this trait yourself.
265    ///
266    ///
267    /// # Safety
268    ///
269    /// Must not be implemented for types that interract with the autorelease
270    /// pool. So if you reimplement the [`AutoreleasePool`] struct or
271    /// likewise, this should be negatively implemented for that.
272    ///
273    /// This can be accomplished with an `PhantomData<AutoreleasePool<'_>>` if
274    /// the `"unstable-autoreleasesafe"` feature is enabled.
275    ///
276    ///
277    /// # Examples
278    ///
279    /// Most types are [`AutoreleaseSafe`].
280    ///
281    /// ```
282    /// use objc2::rc::{AutoreleasePool, AutoreleaseSafe};
283    /// fn requires_autoreleasesafe<T: AutoreleaseSafe>() {}
284    /// requires_autoreleasesafe::<()>();
285    /// requires_autoreleasesafe::<Box<Vec<i32>>>();
286    /// requires_autoreleasesafe::<fn(AutoreleasePool<'_>)>();
287    /// ```
288    ///
289    /// But [`AutoreleasePool`] isn't (if the `"unstable-autoreleasesafe"`
290    /// feature is enabled).
291    ///
292    #[cfg_attr(feature = "unstable-autoreleasesafe", doc = "```compile_fail")]
293    #[cfg_attr(not(feature = "unstable-autoreleasesafe"), doc = "```")]
294    /// use objc2::rc::AutoreleasePool;
295    /// # use objc2::rc::AutoreleaseSafe;
296    /// # fn requires_autoreleasesafe<T: AutoreleaseSafe>() {}
297    /// requires_autoreleasesafe::<AutoreleasePool<'static>>();
298    /// ```
299    ///
300    /// This also means that trait objects aren't (since they may contain an
301    /// [`AutoreleasePool`] internally):
302    ///
303    #[cfg_attr(feature = "unstable-autoreleasesafe", doc = "```compile_fail")]
304    #[cfg_attr(not(feature = "unstable-autoreleasesafe"), doc = "```")]
305    /// # use objc2::rc::AutoreleaseSafe;
306    /// # fn requires_autoreleasesafe<T: AutoreleaseSafe>() {}
307    /// requires_autoreleasesafe::<&dyn std::io::Write>();
308    /// ```
309    pub unsafe trait AutoreleaseSafe {}
310}

320/// Execute `f` in the context of a new autorelease pool. The pool is drained
321/// after the execution of `f` completes.
322///
323/// This corresponds to `@autoreleasepool` blocks in Objective-C and Swift.
324///
325/// The pool is passed as a parameter to the closure to give you a lifetime
326/// parameter that autoreleased objects can refer to.
327///
328/// Note that this is mostly useful for preventing leaks (as any Objective-C
329/// method may autorelease internally - see also [`autoreleasepool_leaking`]).
330/// If implementing an interface to an object, you should try to return
331/// retained pointers with [`msg_send_id!`] wherever you can instead, since
332/// it is usually more efficient, and having to use this function can be quite
333/// cumbersome for users.
334///
335/// [`msg_send_id!`]: crate::msg_send_id
336///
337///
338/// # Restrictions
339///
340/// The given parameter must not be used in an inner `autoreleasepool` - doing
341/// so will panic with debug assertions enabled, and be a compile error in a
342/// future release.
343///
344/// Note that this means that **this function is currently unsound**, since it
345/// doesn't disallow wrong usage in all cases. Enabling the assertions in
346/// release mode would be prohibitively expensive though, so this is the
347/// least-bad solution.
348///
349/// You can try to compile your crate with the `"unstable-autoreleasesafe"`
350/// crate feature enabled on nightly Rust - if your crate compiles with that,
351/// its autoreleasepool usage is guaranteed to be correct.
352///
353///
354/// # Examples
355///
356/// Basic usage:
357///
358/// ```
359/// use objc2::rc::{autoreleasepool, Retained};
360/// use objc2::runtime::NSObject;
361///
362/// autoreleasepool(|pool| {
363///     // Create `obj` and autorelease it to the pool
364///     let obj = Retained::autorelease(NSObject::new(), pool);
365///     // We now have a reference that we can freely use
366///     println!("{obj:?}");
367///     // `obj` is deallocated when the pool ends
368/// });
369/// // And is no longer usable outside the closure
370/// ```
371///
372/// Fails to compile because `obj` does not live long enough for us to take it
373/// out of the pool:
374///
375/// ```compile_fail
376/// use objc2::rc::{autoreleasepool, Retained};
377/// use objc2::runtime::NSObject;
378///
379/// let obj = autoreleasepool(|pool| {
380///     Retained::autorelease(NSObject::new(), pool)
381/// });
382/// ```
383///
384/// Fails to compile with the `"unstable-autoreleasesafe"` feature enabled, or
385/// panics with debug assertions enabled, because we tried to pass an outer
386/// pool to an inner pool:
387///
388#[cfg_attr(feature = "unstable-autoreleasesafe", doc = "```compile_fail")]
389#[cfg_attr(not(feature = "unstable-autoreleasesafe"), doc = "```should_panic")]
390/// use objc2::rc::{autoreleasepool, Retained};
391/// use objc2::runtime::NSObject;
392///
393/// autoreleasepool(|outer_pool| {
394///     let obj = autoreleasepool(|inner_pool| {
395///         Retained::autorelease(NSObject::new(), outer_pool)
396///     });
397///     // `obj` could wrongly be used here because its lifetime was
398///     // assigned to the outer pool, even though it was released by the
399///     // inner pool already.
400/// });
401/// #
402/// # panic!("Does not panic in release mode, so for testing we make it!");
403/// ```
404///
405/// It is impossible to extend the lifetime of the pool.
406///
407/// ```compile_fail
408/// use std::cell::RefCell;
409/// use objc2::rc::{autoreleasepool, AutoreleasePool};
410///
411/// thread_local! {
412///     static POOL: RefCell<Option<&'static AutoreleasePool<'static>>> = RefCell::new(None);
413/// }
414///
415/// autoreleasepool(|pool| {
416///     POOL.with(|p| {
417///         *p.borrow_mut() = Some(Box::leak(Box::new(pool)))
418///     });
419/// });
420/// ```
421#[doc(alias = "@autoreleasepool")]
422#[doc(alias = "objc_autoreleasePoolPush")]
423#[doc(alias = "objc_autoreleasePoolPop")]
424#[inline]
425pub fn autoreleasepool<T, F>(f: F) -> T
426where
427    for<'pool> F: AutoreleaseSafe + FnOnce(AutoreleasePool<'pool>) -> T,
428{
429    // SAFETY:
430    // - The `AutoreleaseSafe` bound on the closure ensures that no pool from
431    //   a different "level" can be passed down through and used in this one.
432    // - The pools are guaranteed to be dropped in the reverse order they were
433    //   created (since you can't possibly "interleave" closures).
434    //
435    //   This would not work if we e.g. allowed users to create pools on the
436    //   stack, since they could then safely control the drop order.
437    let pool = unsafe { Pool::new() };
438    let res = f(AutoreleasePool::new(Some(&pool)));
439    unsafe { pool.drain() };
440    res
441}

443/// Execute `f` in the context of a "fake" autorelease pool.
444///
445/// This is useful to create a context in which to use autoreleased objects,
446/// without the overhead of actually creating and draining the pool.
447///
448/// Any function boundary in Objective-C is an implicit autorelease pool, so
449/// there you'd do `id obj2 = [obj autorelease]` and be done with it - but we
450/// do this using a closure instead because we need some way to bind the
451/// lifetime of any objects released to the pool.
452///
453///
454/// # Examples
455///
456/// Autorelease an object to an outer pool, from inside an inner, "fake" pool.
457///
458/// ```
459/// use objc2::rc::{autoreleasepool, autoreleasepool_leaking, Retained};
460/// use objc2::runtime::NSObject;
461///
462/// autoreleasepool(|outer_pool| {
463///     let obj = autoreleasepool_leaking(|inner_pool| {
464///         Retained::autorelease(NSObject::new(), outer_pool)
465///     });
466///     // `obj` is still usable here, since the leaking pool doesn't actually
467///     // do anything.
468///     println!("{obj:?}");
469/// });
470///
471/// // But it is not usable here, since the outer pool has been closed
472/// ```
473///
474/// Like [`autoreleasepool`], you can't extend the lifetime of an object to
475/// outside the closure.
476///
477/// ```compile_fail
478/// use objc2::rc::{autoreleasepool_leaking, Retained};
479/// use objc2::runtime::NSObject;
480///
481/// let obj = autoreleasepool_leaking(|pool| {
482///     Retained::autorelease(NSObject::new(), pool)
483/// });
484/// ```
485///
486/// While you can pass an outer pool into this, you still can't pass the pool
487/// from this into [`autoreleasepool`]:
488///
489#[cfg_attr(feature = "unstable-autoreleasesafe", doc = "```compile_fail")]
490#[cfg_attr(not(feature = "unstable-autoreleasesafe"), doc = "```should_panic")]
491/// use objc2::rc::{autoreleasepool, autoreleasepool_leaking, Retained};
492/// use objc2::runtime::NSObject;
493///
494/// autoreleasepool_leaking(|outer_pool| {
495///     let obj = autoreleasepool(|inner_pool| {
496///         Retained::autorelease(NSObject::new(), outer_pool)
497///     });
498/// });
499/// #
500/// # panic!("Does not panic in release mode, so for testing we make it!");
501/// ```
502#[inline]
503pub fn autoreleasepool_leaking<T, F>(f: F) -> T
504where
505    for<'pool> F: FnOnce(AutoreleasePool<'pool>) -> T,
506{
507    // SAFETY: This is effectively what most Objective-C code does; they
508    // assume that there's an autorelease pool _somewhere_ in the call stack
509    // above it, and then use their autoreleased objects for a duration that
510    // is guaranteed to be shorter than that.
511    //
512    // The `AutoreleaseSafe` bound is not required, since we don't actually do
513    // anything inside this; hence if the user know they have the _actual_
514    // innermost pool, they may still safely use it to extend the lifetime
515    // beyond this closure.
516    f(AutoreleasePool::new(None))
517}
