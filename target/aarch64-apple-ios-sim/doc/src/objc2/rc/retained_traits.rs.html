5/// Helper trait to implement [`Default`] on [`Retained`].
6#[doc(alias = "DefaultId")] // Previous name
7pub trait DefaultRetained {
8    /// The default [`Retained`] for a type.
9    ///
10    /// On most objects the implementation would be sending a message to the
11    /// `new` selector.
12    fn default_retained() -> Retained<Self>;
13}
14
15impl<T: ?Sized + DefaultRetained> Default for Retained<T> {
16    #[inline]
17    fn default() -> Self {
18        T::default_retained()
19    }
20}

22/// Helper trait to implement [`IntoIterator`] on [`Retained`].
23///
24/// This should be implemented in exactly the same fashion as if you were
25/// implementing `IntoIterator` for your type normally.
26//
27// Note that [`Box<T>` gets to cheat with regards moves][box-move], so
28// `boxed.into_iter()` is possible, while `obj.into_iter()` is not possible
29// without this helper trait.
30//
31// [box-move]: https://doc.rust-lang.org/reference/expressions.html#moved-and-copied-types
32#[doc(alias = "IdIntoIterator")] // Previous name
33pub trait RetainedIntoIterator {
34    /// The type of the elements being iterated over.
35    type Item;
36
37    /// Which kind of iterator are we turning this into?
38    type IntoIter: Iterator<Item = Self::Item>;
39
40    /// Creates an iterator from an [`Retained`].
41    ///
42    /// You would normally not call this function directly; instead, you'd
43    /// call [`into_iter`](IntoIterator::into_iter) on an [`Retained`].
44    fn retained_into_iter(this: Retained<Self>) -> Self::IntoIter;
45}
46
47// Note: These `IntoIterator` implementations conflict with an `Iterator`
48// implementation for `Retained`.
49//
50// For our case however (in contrast with `Box`), that is the better tradeoff,
51// which I will show with an example:
52//
53// ```
54// let xs = Box::new(vec![]);
55// for x in &xs { // Doesn't compile, `&Box` doesn't implement `IntoIterator`
56//     // ...
57// }
58// ```
59//
60// Here, you're expected to write `xs.iter()` or `&**xs` instead, which is
61// fairly acceptable, since usually people don't wrap things in boxes so much;
62// but in Objective-C, _everything_ is wrapped in an `Retained`, and hence we should
63// attempt to make that common case easier:
64//
65// ```
66// let obj = NSArray::new(); // `Retained<NSArray<_>>`
67// for item in &obj { // Should compile
68//     // ...
69// }
70// ```
71//
72// The loss of the `Iterator` impl is a bit unfortunate, but not a big deal,
73// since there is only one iterator in Objective-C anyhow, `NSEnumerator`, and
74// for that we can make other abstractions instead.
75impl<T: ?Sized + RetainedIntoIterator> IntoIterator for Retained<T> {
76    type Item = <T as RetainedIntoIterator>::Item;
77    type IntoIter = <T as RetainedIntoIterator>::IntoIter;
78
79    #[inline]
80    fn into_iter(self) -> Self::IntoIter {
81        T::retained_into_iter(self)
82    }
83}
84
85impl<'a, T: ?Sized> IntoIterator for &'a Retained<T>
86where
87    &'a T: IntoIterator,
88{
89    type Item = <&'a T as IntoIterator>::Item;
90    type IntoIter = <&'a T as IntoIterator>::IntoIter;
91
92    #[inline]
93    fn into_iter(self) -> Self::IntoIter {
94        (&**self).into_iter()
95    }
96}

98/// Helper trait to implement [`FromIterator`] on [`Retained`].
99///
100/// This should be implemented in exactly the same fashion as if you were
101/// implementing `FromIterator` for your type normally.
102#[doc(alias = "IdFromIterator")] // Previous name
103pub trait RetainedFromIterator<T>: Sized {
104    /// Creates an `Retained` from an iterator.
105    fn retained_from_iter<I>(iter: I) -> Retained<Self>
106    where
107        I: IntoIterator<Item = T>;
108}
109
110impl<T, U: RetainedFromIterator<T>> FromIterator<T> for Retained<U> {
111    #[inline]
112    fn from_iter<I: IntoIterator<Item = T>>(iter: I) -> Self {
113        U::retained_from_iter(iter)
114    }
115}
