9#[derive(Debug, PartialEq, Eq, Hash)]
10pub(crate) enum Inner {
11    MethodNotFound,
12    EncodingParseError(EncodingParseError),
13    MismatchedReturn(EncodingBox, Encoding),
14    MismatchedArgumentsCount(usize, usize),
15    MismatchedArgument(usize, EncodingBox, Encoding),
16}
17
18impl fmt::Display for Inner {
19    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
20        match self {
21            Self::MethodNotFound => write!(f, "method not found"),
22            Self::EncodingParseError(e) => write!(f, "{e}"),
23            Self::MismatchedReturn(expected, actual) => {
24                write!(
25                    f,
26                    "expected return to have type code '{expected}', but found '{actual}'",
27                )
28            }
29            Self::MismatchedArgumentsCount(expected, actual) => {
30                write!(f, "expected {expected} arguments, but {actual} were given",)
31            }
32            Self::MismatchedArgument(i, expected, actual) => {
33                write!(
34                    f,
35                    "expected argument at index {i} to have type code '{expected}', but found '{actual}'",
36                )
37            }
38        }
39    }
40}

75/// Relaxed version of `Encoding::equivalent_to_box` that allows interchanging
76/// more types.
77///
78/// NOTE: This is a top-level comparison, e.g. `*mut *mut c_void` or structs
79/// containing `*mut c_void` are not allowed differently than usual.
80fn relaxed_equivalent_to_box(encoding: &Encoding, expected: &EncodingBox) -> bool {
81    // Register-sized integers are interoperable regardless of sign. The Rust
82    // [docs on ABI compatibility][abi] says:
83    // > Noteworthy cases of types _not_ being ABI-compatible in general are:
84    // > - `bool` vs `u8`, `i32` vs `u32`, `char` vs `i32`: on some targets,
85    // >   the calling conventions for these types differ in terms of what
86    // >   they guarantee for the remaining bits in the register that are not
87    // >   used by the value.
88    // > - ...
89    // [abi]: https://doc.rust-lang.org/std/primitive.fn.html#abi-compatibility
90    //
91    // But if the type is the size of the target's register size, then there
92    // are no remaining bits that can differ, and hence these types should be
93    // ABI compatible.
94    let size_of_matching_integer_encodings = match (encoding, expected) {
95        (Encoding::Char | Encoding::UChar, EncodingBox::Char | EncodingBox::UChar) => {
96            size_of::<core::ffi::c_char>()
97        }
98        (Encoding::Short | Encoding::UShort, EncodingBox::Short | EncodingBox::UShort) => {
99            size_of::<core::ffi::c_short>()
100        }
101        (Encoding::Int | Encoding::UInt, EncodingBox::Int | EncodingBox::UInt) => {
102            size_of::<core::ffi::c_int>()
103        }
104        (Encoding::Long | Encoding::ULong, EncodingBox::Long | EncodingBox::ULong) => {
105            size_of::<core::ffi::c_long>()
106        }
107        (
108            Encoding::LongLong | Encoding::ULongLong,
109            EncodingBox::LongLong | EncodingBox::ULongLong,
110        ) => size_of::<core::ffi::c_longlong>(),
111        _ => 0,
112    };
113    // NOTE: We use `size_of<usize>()` here as a proxy for detecting the
114    // register size, though this is not strictly correct, we should perhaps
115    // use `size_of::<core::ffi::c_size_t>()` instead?
116    let register_size = size_of::<usize>();
117    if size_of_matching_integer_encodings == register_size && cfg!(target_vendor = "apple") {
118        // TODO: Swift is doing this, so it's definitely safe on Apple
119        // platforms. But is it also on others?
120        return true;
121    }
122
123    // Allow `*mut c_void` and `*const c_void` to be used in place of other
124    // pointers.
125    if cfg!(feature = "relax-void-encoding")
126        && matches!(encoding, Encoding::Pointer(&Encoding::Void))
127        && matches!(expected, EncodingBox::Pointer(_))
128    {
129        return true;
130    }
131
132    // Allow signed types where unsigned types are excepted.
133    if cfg!(feature = "relax-sign-encoding") {
134        let actual_signed = match encoding {
135            Encoding::UChar => &Encoding::Char,
136            Encoding::UShort => &Encoding::Short,
137            Encoding::UInt => &Encoding::Int,
138            Encoding::ULong => &Encoding::Long,
139            Encoding::ULongLong => &Encoding::LongLong,
140            enc => enc,
141        };
142        let expected_signed = match expected {
143            EncodingBox::UChar => &EncodingBox::Char,
144            EncodingBox::UShort => &EncodingBox::Short,
145            EncodingBox::UInt => &EncodingBox::Int,
146            EncodingBox::ULong => &EncodingBox::Long,
147            EncodingBox::ULongLong => &EncodingBox::LongLong,
148            enc => enc,
149        };
150        if actual_signed == expected_signed {
151            return true;
152        }
153    }
154
155    encoding.equivalent_to_box(expected)
156}

158pub(crate) fn verify_method_signature(
159    method: &Method,
160    args: &[Encoding],
161    ret: &Encoding,
162) -> Result<(), VerificationError> {
163    let mut iter = method.types();
164
165    // TODO: Verify stack layout
166    let (expected, _stack_layout) = iter.extract_return()?;
167    if !relaxed_equivalent_to_box(ret, &expected) {
168        return Err(Inner::MismatchedReturn(expected, ret.clone()).into());
169    }
170
171    iter.verify_receiver()?;
172    iter.verify_sel()?;
173
174    let actual_count = args.len();
175
176    for (i, actual) in args.iter().enumerate() {
177        if let Some(res) = iter.next() {
178            // TODO: Verify stack layout
179            let (expected, _stack_layout) = res?;
180            if !relaxed_equivalent_to_box(actual, &expected) {
181                return Err(Inner::MismatchedArgument(i, expected, actual.clone()).into());
182            }
183        } else {
184            return Err(Inner::MismatchedArgumentsCount(i, actual_count).into());
185        }
186    }
187
188    let remaining = iter.count();
189    if remaining != 0 {
190        return Err(Inner::MismatchedArgumentsCount(actual_count + remaining, actual_count).into());
191    }
192
193    let expected_count = method.name().number_of_arguments();
194    if expected_count != actual_count {
195        return Err(Inner::MismatchedArgumentsCount(expected_count, actual_count).into());
196    }
197
198    Ok(())
199}

227        let cls = test_utils::custom_class();
228
229        // Unimplemented selector (missing colon)
230        let err = cls.verify_sel::<(), ()>(sel!(setFoo)).unwrap_err();
231        assert_eq!(err.to_string(), "method not found");
232
233        // Incorrect return type
234        let err = cls.verify_sel::<(u32,), u64>(sel!(setFoo:)).unwrap_err();
235        assert_eq!(
236            err.to_string(),
237            "expected return to have type code 'v', but found 'Q'"
238        );
239
240        // Too many arguments
241        let err = cls.verify_sel::<(u32, i8), ()>(sel!(setFoo:)).unwrap_err();
242        assert_eq!(err.to_string(), "expected 1 arguments, but 2 were given");
243
244        // Too few arguments
245        let err = cls.verify_sel::<(), ()>(sel!(setFoo:)).unwrap_err();
246        assert_eq!(err.to_string(), "expected 1 arguments, but 0 were given");
247
248        // Incorrect argument type
249        let err = cls.verify_sel::<(Sel,), ()>(sel!(setFoo:)).unwrap_err();
250        assert_eq!(
251            err.to_string(),
252            "expected argument at index 0 to have type code 'I', but found ':'"
253        );
254
255        // <https://github.com/madsmtm/objc2/issues/566>
256        let res = cls.verify_sel::<(), ffi::NSUInteger>(sel!(getNSInteger));
257        let expected = if cfg!(any(
258            target_vendor = "apple",
259            feature = "relax-sign-encoding"
260        )) {
261            Ok(())
262        } else if cfg!(target_pointer_width = "64") {

266        };

275    }
