10/// Types that can be sent Objective-C messages.
11///
12/// Implementing this provides [`MessageReceiver`] implementations for common
13/// pointer types and references to the type, which allows using them as the
14/// receiver (first argument) in the [`msg_send!`][`crate::msg_send`] macro.
15///
16/// This trait also allows the object to be used in [`Retained`].
17///
18/// This is a subtrait of [`RefEncode`], meaning the type must also implement
19/// that, almost always with [`RefEncode::ENCODING_REF`] being
20/// [`Encoding::Object`].
21///
22/// This can be implemented for unsized (`!Sized`) types, but the intention is
23/// not to support dynamically sized types like slices, only `extern type`s
24/// (which is currently unstable).
25///
26/// [`MessageReceiver`]: crate::runtime::MessageReceiver
27/// [`Encoding::Object`]: crate::Encoding::Object
28///
29///
30/// # `Drop` interaction
31///
32/// If the inner type implements [`Drop`], that implementation will very
33/// likely not be called, since there is no way to ensure that the Objective-C
34/// runtime will do so. If you need to run some code when the object is
35/// destroyed, implement the `dealloc` method instead.
36///
37/// The [`declare_class!`] macro does this for you, but the [`extern_class!`]
38/// macro fundamentally cannot.
39///
40/// [`declare_class!`]: crate::declare_class
41/// [`extern_class!`]: crate::extern_class
42///
43///
44/// # Safety
45///
46/// The type must represent an Objective-C object, meaning it:
47/// - Must be valid to reinterpret as [`AnyObject`].
48/// - Must be able to be the receiver of an Objective-C message sent with
49///   [`objc_msgSend`] or similar.
50/// - Must respond to the standard memory management `retain`, `release` and
51///   `autorelease` messages.
52/// - Must support weak references. (In the future we should probably make a
53///   new trait for this, for example `NSTextView` only supports weak
54///   references on macOS 10.12 or above).
55///
56/// [`AnyObject`]: crate::runtime::AnyObject
57/// [`objc_msgSend`]: https://developer.apple.com/documentation/objectivec/1456712-objc_msgsend
58///
59///
60/// # Example
61///
62/// ```
63/// use objc2::runtime::NSObject;
64/// use objc2::{Encoding, Message, RefEncode};
65///
66/// #[repr(C)]
67/// struct MyObject {
68///     // This has the exact same layout as `NSObject`
69///     inner: NSObject
70/// }
71///
72/// unsafe impl RefEncode for MyObject {
73///     const ENCODING_REF: Encoding = Encoding::Object;
74/// }
75///
76/// unsafe impl Message for MyObject {}
77///
78/// // `*mut MyObject` and other pointer/reference types to the object can
79/// // now be used in `msg_send!`
80/// //
81/// // And `Retained<MyObject>` can now be constructed.
82/// ```
83pub unsafe trait Message: RefEncode {}

85/// Marks types that represent specific classes.
86///
87/// Sometimes it is enough to generically know that a type is messageable,
88/// e.g. [`Retained`] works with any type that implements the [`Message`]
89/// trait. But often, you have an object that you know represents a specific
90/// Objective-C class - this trait allows you to communicate that, as well as
91/// a few properties of the class to the rest of the type-system.
92///
93/// This is implemented automatically for your type by the
94/// [`declare_class!`][crate::declare_class] and
95/// [`extern_class!`][crate::extern_class] macros.
96///
97///
98/// # Safety
99///
100/// 1. The type must represent a specific class.
101/// 2. [`Self::Super`] must be a superclass of the class (or something that
102///    represents any object, like [`AnyObject`][crate::runtime::AnyObject]).
103/// 3. [`Self::Mutability`] must be specified correctly.
104///
105///    Note that very little Objective-C code follows Rust's usual ownership
106///    model. If you think your type's mutability should be [`Mutable`], think
107///    again, it _very_ rarely should!
108///
109///    If you're unsure of what to do, [`InteriorMutable`] + avoiding `&mut`
110///    is usually a good starting point.
111/// 4. [`Self::NAME`] must be correct.
112/// 5. The class returned by [`Self::class`] must be the class that this type
113///    represents.
114///
115/// [`Mutable`]: crate::mutability::Mutable
116/// [`InteriorMutable`]: crate::mutability::InteriorMutable
117///
118///
119/// # Examples
120///
121/// Use the trait to access the [`AnyClass`] of an object.
122///
123/// ```
124/// use objc2::{ClassType, msg_send_id};
125/// use objc2::rc::Retained;
126/// # use objc2::runtime::{NSObject as MyObject};
127///
128/// // Get the class of the object.
129/// let cls = <MyObject as ClassType>::class();
130/// // Or, since the trait is in scope.
131/// let cls = MyObject::class();
132///
133/// // We can now access properties of the class.
134/// assert_eq!(cls.name(), MyObject::NAME);
135///
136/// // And we can send messages to the class.
137/// //
138/// // SAFETY:
139/// // - The class is `MyObject`, which can safely be initialized with `new`.
140/// // - The return type is correctly specified.
141/// let obj: Retained<MyObject> = unsafe { msg_send_id![cls, new] };
142/// ```
143///
144/// Use the trait to allocate a new instance of an object.
145///
146/// ```
147/// use objc2::{ClassType, msg_send_id};
148/// use objc2::rc::Retained;
149/// # use objc2::runtime::{NSObject as MyObject};
150///
151/// let obj = MyObject::alloc();
152///
153/// // Now we can call initializers on this newly allocated object.
154/// //
155/// // SAFETY: `MyObject` can safely be initialized with `init`.
156/// let obj: Retained<MyObject> = unsafe { msg_send_id![obj, init] };
157/// ```
158///
159/// Use the [`extern_class!`][crate::extern_class] macro to implement this
160/// trait for a type.
161///
162/// ```
163/// use objc2::runtime::NSObject;
164/// use objc2::{extern_class, mutability, ClassType};
165///
166/// extern_class!(
167///     struct MyClass;
168///
169///     // SAFETY: The superclass and the mutability is correctly specified.
170///     unsafe impl ClassType for MyClass {
171///         type Super = NSObject;
172///         type Mutability = mutability::InteriorMutable;
173///         # // For testing purposes
174///         # const NAME: &'static str = "NSObject";
175///     }
176/// );
177///
178/// let cls = MyClass::class();
179/// let obj = MyClass::alloc();
180/// ```
181///
182/// Implement the trait manually for a class with a lifetime parameter.
183///
184/// ```
185#[doc = include_str!("../examples/class_with_lifetime.rs")]
186/// ```
187pub unsafe trait ClassType: Message {
188    /// The superclass of this class.
189    ///
190    /// If you have implemented [`Deref`] for your type, it is highly
191    /// recommended that this is equal to [`Deref::Target`].
192    ///
193    /// This may be [`AnyObject`] if the class is a root class.
194    ///
195    /// [`Deref`]: std::ops::Deref
196    /// [`Deref::Target`]: std::ops::Deref::Target
197    /// [`AnyObject`]: crate::runtime::AnyObject
198    type Super: Message;
199
200    /// Whether the type is mutable or immutable.
201    ///
202    /// See the [`mutability`][crate::mutability] module for further details
203    /// about class mutability.
204    type Mutability: Mutability;
205
206    /// The name of the Objective-C class that this type represents.
207    ///
208    /// `T::NAME` is the `const` version of `T::class().name()`.
209    const NAME: &'static str;
210
211    /// Get a reference to the Objective-C class that this type represents.
212    ///
213    /// May register the class with the runtime if it wasn't already.
214    ///
215    ///
216    /// # Panics
217    ///
218    /// This may panic if something went wrong with getting or declaring the
219    /// class, e.g. if the program is not properly linked to the framework
220    /// that defines the class.
221    fn class() -> &'static AnyClass;
222
223    /// Get an immutable reference to the superclass.
224    // Note: It'd be safe to provide a default impl using transmute here if
225    // we wanted to!
226    fn as_super(&self) -> &Self::Super;
227
228    /// Get a mutable reference to the superclass.
229    // Note: No `Self: IsMutable` bound required here, since there is no way
230    // to get `&mut self` in the first place.
231    //
232    // Or at least, if we have `&mut NSMutableString`, we're allowed to get
233    // `&mut NSString`, and from that it will also make sense to allow getting
234    // `&mut NSObject`.
235    fn as_super_mut(&mut self) -> &mut Self::Super;
236
237    /// Increment the reference count of the receiver.
238    ///
239    /// This extends the duration in which the receiver is alive by detaching
240    /// it from the lifetime information carried by the reference.
241    ///
242    /// This is similar to using [`Clone` on `Retained<Self>`][clone-id], with
243    /// the addition that it can be used on a plain reference. Note however
244    /// that this is not possible to use on certain types like `NSString`,
245    /// since if you only hold `&NSString`, that may have come from
246    /// `&mut NSMutableString`, in which case it would be unsound to erase the
247    /// lifetime information carried by the reference.
248    ///
249    /// In cases like that, you should rather use `NSCopying::copy` (since
250    /// that gives you a `NSString` whether the string was originally a
251    /// `NSString` or a `NSMutableString`).
252    ///
253    /// [clone-id]: crate::rc::Retained#impl-Clone-for-Retained<T>
254    //
255    // Note: We could have placed this on `mutability::IsRetainable`, but
256    // `ClassType` is more often already in scope, allowing easier access to
257    // `obj.retain()`.
258    #[inline]
259    #[doc(alias = "objc_retain")]
260    fn retain(&self) -> Retained<Self>
261    where
262        Self: IsRetainable,
263        Self: Sized, // Temporary
264    {
265        let ptr: *const Self = self;
266        let ptr: *mut Self = ptr as _;
267        // SAFETY:
268        // - The object is known to not be mutable (or have a mutable
269        //   subclass) due to the `IsRetainable` bound.
270        // - The pointer is valid since it came from `&self`.
271        // - The lifetime of the pointer itself is extended, but any lifetime
272        //   that the object may carry is still kept within the type itself.
273        let obj = unsafe { Retained::retain(ptr) };
274        // SAFETY: The pointer came from `&self`, which is always non-null
275        // (and objc_retain always returns the same value).
276        unsafe { obj.unwrap_unchecked() }
277    }
278
279    /// Allocate a new instance of the class.
280    ///
281    /// The return value can be used directly inside [`msg_send_id!`] to
282    /// initialize the object.
283    ///
284    /// For classes that are only usable on the main thread, you can use
285    /// `MainThreadMarker::alloc` instead.
286    ///
287    /// [`msg_send_id!`]: crate::msg_send_id
288    //
289    // Note: We could have placed this on `mutability::IsAllocableAnyThread`,
290    // but `ClassType` is more often already in scope, allowing easier access
291    // to `T::alloc()`.
292    #[inline]
293    fn alloc() -> Allocated<Self>
294    where
295        Self: IsAllocableAnyThread + Sized,
296    {
297        // SAFETY:
298        // - It is always safe to (attempt to) allocate an object.
299        // - The object is of the correct type, since we've used the class
300        //   from `Self::class`.
301        // - The object is safe to `dealloc` on the current thread (due to the
302        //   `IsAllocableAnyThread` bound which guarantees it is not
303        //   `MainThreadOnly`).
304        //
305        // See also `MainThreadMarker::alloc`.
306        unsafe { msg_send_id![Self::class(), alloc] }
307    }
308
309    // TODO: `fn alloc_on_main(mtm: MainThreadMarker)`
310    // TODO: `fn mtm(&self) -> MainThreadMarker where T::Mutability: MainThreadOnly`
311}

313/// Marks types whose implementation is defined in Rust.
314///
315/// This is used in [`declare_class!`], and allows access to the instance
316/// variables that a given type declares, see that macro for details.
317///
318/// [`declare_class!`]: crate::declare_class
319//
320// Note: We mark this trait as not `unsafe` for better documentation, since
321// implementing it inside `declare_class!` is not `unsafe`.
322//
323// Safety is ensured by `__UNSAFE_OFFSETS_CORRECT`.
324pub trait DeclaredClass: ClassType {
325    /// A type representing the instance variables that this class carries.
326    type Ivars: Sized;
327
328    // TODO: Add `ivars_ptr(this: NonNull<Self>) -> NonNull<Self::Ivars>`?
329
330    /// Get a reference to the instance variable data that this object
331    /// carries.
332    #[inline]
333    #[track_caller]
334    fn ivars(&self) -> &Self::Ivars
335    where
336        Self: Sized, // Required because of MSRV
337    {
338        let ptr: NonNull<Self> = NonNull::from(self);
339        // SAFETY: The pointer is valid and initialized.
340        let ivars = unsafe { get_initialized_ivar_ptr(ptr) };
341        // SAFETY: The lifetime of the instance variable is tied to the object.
342        unsafe { ivars.as_ref() }
343    }
344
345    /// Get a mutable reference to the instance variable data that this object
346    /// carries.
347    #[inline]
348    #[track_caller]
349    fn ivars_mut(&mut self) -> &mut Self::Ivars
350    where
351        Self: Sized, // Required because of MSRV
352    {
353        let ptr: NonNull<Self> = NonNull::from(self);
354        // SAFETY: The pointer is valid and initialized.
355        let mut ivars = unsafe { get_initialized_ivar_ptr(ptr) };
356        // SAFETY: The lifetime of the instance variable is tied to the object.
357        //
358        // Mutability is safe since the object itself is mutable. See
359        // `ClassType::as_super_mut` for why this is safe without
360        // `Self: IsMutable`.
361        unsafe { ivars.as_mut() }
362    }
363
364    #[doc(hidden)]
365    fn __ivars_offset() -> isize;
366
367    #[doc(hidden)]
368    fn __drop_flag_offset() -> isize;
369
370    /// # Safety
371    ///
372    /// The ivar offset and drop flag offsets must be implemented correctly.
373    #[doc(hidden)]
374    const __UNSAFE_OFFSETS_CORRECT: ();
375}

377/// Marks types that represent specific protocols.
378///
379/// This is the protocol equivalent of [`ClassType`].
380///
381/// This is implemented automatically by the [`extern_protocol!`] macro for
382/// `dyn T`, where `T` is the protocol.
383///
384/// [`ClassType`]: crate::ClassType
385/// [`extern_protocol!`]: crate::extern_protocol
386///
387///
388/// # Safety
389///
390/// This is meant to be a sealed trait, and should not be implemented outside
391/// of the [`extern_protocol!`] macro.
392///
393///
394/// # Examples
395///
396/// Use the trait to access the [`AnyProtocol`] of different objects.
397///
398/// ```
399/// use objc2::ProtocolType;
400/// use objc2::runtime::NSObjectProtocol;
401/// // Get a protocol object representing the `NSObject` protocol
402/// let protocol = <dyn NSObjectProtocol>::protocol().expect("NSObject to have a protocol");
403/// assert_eq!(<dyn NSObjectProtocol>::NAME, protocol.name());
404/// # // Ensure Foundation links on GNUStep
405/// # let _cls = objc2::class!(NSObject);
406/// ```
407///
408/// Use the [`extern_protocol!`] macro to implement this trait for a type.
409///
410/// ```no_run
411/// use objc2::{extern_protocol, ProtocolType};
412///
413/// extern_protocol!(
414///     unsafe trait MyProtocol {}
415///     unsafe impl ProtocolType for dyn MyProtocol {}
416/// );
417///
418/// let protocol = <dyn MyProtocol>::protocol();
419/// ```
420pub unsafe trait ProtocolType {
421    /// The name of the Objective-C protocol that this type represents.
422    const NAME: &'static str;
423
424    /// Get a reference to the Objective-C protocol object that this type
425    /// represents.
426    ///
427    /// May register the protocol with the runtime if it wasn't already.
428    ///
429    /// Note that some protocols [are not registered with the runtime][p-obj],
430    /// depending on various factors. In those cases, this function may return
431    /// `None`.
432    ///
433    /// [p-obj]: https://developer.apple.com/library/archive/documentation/Cocoa/Conceptual/ObjectiveC/Chapters/ocProtocols.html#//apple_ref/doc/uid/TP30001163-CH15-TPXREF149
434    ///
435    ///
436    /// # Panics
437    ///
438    /// This may panic if something went wrong with getting or declaring the
439    /// protocol, e.g. if the program is not properly linked to the framework
440    /// that defines the protocol.
441    fn protocol() -> Option<&'static AnyProtocol> {
442        AnyProtocol::get(Self::NAME)
443    }
444
445    #[doc(hidden)]
446    const __INNER: ();
