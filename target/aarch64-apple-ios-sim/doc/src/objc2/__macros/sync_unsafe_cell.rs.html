4/// `UnsafeCell`, but `Sync`.
5#[repr(transparent)]
6#[derive(Debug)]
7pub struct SyncUnsafeCell<T: ?Sized> {
8    value: UnsafeCell<T>,
9}
10
11// SAFETY: This type is used for making `static`s which contain `UnsafeCell`.
12// The user upholds that such usage is correct.
13unsafe impl<T: ?Sized + Sync> Sync for SyncUnsafeCell<T> {}
14
15impl<T> SyncUnsafeCell<T> {
16    #[inline]
17    pub const fn new(value: T) -> Self {
18        Self {
19            value: UnsafeCell::new(value),
20        }
21    }
22
23    #[inline]
24    pub fn into_inner(self) -> T {
25        self.value.into_inner()
26    }
27}
28
29impl<T: ?Sized> SyncUnsafeCell<T> {
30    #[inline]
