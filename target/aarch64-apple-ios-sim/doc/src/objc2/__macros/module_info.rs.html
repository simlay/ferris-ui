22/// Helper struct for emitting the module info that macOS 32-bit requires.
23///
24/// <https://github.com/llvm/llvm-project/blob/release/13.x/clang/lib/CodeGen/CGObjCMac.cpp#L5211-L5234>
25#[repr(C)]
26#[derive(Debug)]
27pub struct ModuleInfo {
28    version: usize,
29    size: usize,
30    name: *const u8,
31    symtab: *const (),
32}
33
34// SAFETY: ModuleInfo is immutable.
35unsafe impl Sync for ModuleInfo {}
36
37impl ModuleInfo {
38    /// This is hardcoded in clang as 7.
39    const VERSION: usize = 7;
40
41    pub const fn new(name: *const u8) -> Self {
42        Self {
43            version: Self::VERSION,
44            size: core::mem::size_of::<Self>(),
45            name,
46            // We don't expose any symbols
47            symtab: core::ptr::null(),
48        }
49    }
50}
