1/// Forward selector and arguments to `MsgSend::send_message[_error]`.
2///
3/// Note: We can't forward to `msg_send!` since that doesn't support selectors
4/// with space between.
5#[doc(hidden)]
6#[macro_export]
7macro_rules! __method_msg_send {
8    // Selector with no arguments
9    (
10        ($receiver:expr)
11        ($sel:ident)
12        ()
13
14        ()
15        ()
16        ($($method_family:tt)*)
17    ) => {
18        <$crate::__method_family!(($($method_family)*) ($sel)) as $crate::__macros::MsgSend<_, _>>::send_message(
19            $receiver,
20            $crate::sel!($sel),
21            (),
22        )
23    };
24
25    // Skip using `MainThreadMarker` in the message send.
26    //
27    // This is a purely textual match, and using e.g. `objc2::MainThreadMarker`
28    // would fail - but that would just be detected as giving a wrong number
29    // of arguments, so it's fine for now.
30    (
31        ($receiver:expr)
32        ($($sel_rest:tt)*)
33        ($arg:ident: MainThreadMarker $(, $($params_rest:tt)*)?)
34
35        ($($sel_parsed:tt)*)
36        ($($arg_parsed:tt)*)
37        ($($method_family:tt)*)
38    ) => ({
39        let _ = $arg;
40        $crate::__method_msg_send! {
41            ($receiver)
42            ($($sel_rest)*)
43            ($($($params_rest)*)?)
44
45            ($($sel_parsed)*)
46            ($($arg_parsed)*)
47            ($($method_family)*)
48        }
49    });
50
51    // Parse each argument-selector pair
52    (
53        ($receiver:expr)
54        ($($sel:ident)? : $($sel_rest:tt)*)
55        ($arg:ident : $_arg_ty:ty $(, $($params_rest:tt)*)?)
56
57        ($($sel_parsed:tt)*)
58        ($($arg_parsed:tt)*)
59        ($($method_family:tt)*)
60    ) => {
61        $crate::__method_msg_send! {
62            ($receiver)
63            ($($sel_rest)*)
64            ($($($params_rest)*)?)
65
66            ($($sel_parsed)* $($sel)? :)
67            ($($arg_parsed)* $arg,)
68            ($($method_family)*)
69        }
70    };
71    // Handle path separator token
72    (
73        ($receiver:expr)
74        ($($sel:ident)? :: $($sel_rest:tt)*)
75        ($arg1:ident : $_arg_ty1:ty, $arg2:ident : $_arg_ty2:ty $(, $($params_rest:tt)*)?)
76
77        ($($sel_parsed:tt)*)
78        ($($arg_parsed:tt)*)
79        ($($method_family:tt)*)
80    ) => {
81        $crate::__method_msg_send! {
82            ($receiver)
83            ($($sel_rest)*)
84            ($($($params_rest)*)?)
85
86            ($($sel_parsed)* $($sel)? : :)
87            ($($arg_parsed)* $arg1, $arg2,)
88            ($($method_family)*)
89        }
90    };
91
92    // Normal return
93    (
94        ($receiver:expr)
95        ()
96        ()
97
98        // Notice the "+" here; we must make sure we actually _did_ parse
99        // a selector, and haven't just gotten an empty `#[unsafe(method())]`.
100        ($($sel_parsed:tt)+)
101        ($($arg_parsed:tt)*)
102        ($($method_family:tt)*)
103    ) => {
104        <$crate::__method_family!(($($method_family)*) ($($sel_parsed)*)) as $crate::__macros::MsgSend<_, _>>::send_message(
105            $receiver,
106            $crate::sel!($($sel_parsed)*),
107            ($($arg_parsed)*),
108        )
109    };
110
111    // Error return
112    (
113        ($receiver:expr)
114        // `sel:_` without a corresponding argument
115        ($sel:ident : _)
116        ()
117
118        ($($sel_parsed:tt)*)
119        ($($arg_parsed:tt)*)
120        ($($method_family:tt)*)
121    ) => {
122        // Use error method
123        <$crate::__method_family!(($($method_family)*) ($($sel_parsed)* $sel :)) as $crate::__macros::MsgSendError<_, _>>::send_message_error(
124            $receiver,
125            $crate::sel!($($sel_parsed)* $sel :),
126            ($($arg_parsed)*),
127        )
128    };
129
130    // Variadic method
131    (
132        ($receiver:expr)
133        ($($sel:ident : _)?)
134        ($($arg:ident :)? ...)
135
136        ($($sel_parsed:tt)*)
137        ($($arg_parsed:tt)*)
138        ($($method_family:tt)*)
139    ) => ({
140        $crate::__macros::compile_error!(
141            "variadic methods are not yet supported"
142        )
143    });
144
145    // Mismatched selector/argument
146    (
147        ($receiver:expr)
148        ($($sel_rest:tt)*)
149        ($($params_rest:tt)*)
150
151        ($($sel_parsed:tt)*)
152        ($($arg_parsed:tt)*)
153        ($($method_family:tt)*)
154    ) => ({
155        $crate::__macros::compile_error!(
156            "number of arguments in function and selector did not match"
157        )
158    });
159}
