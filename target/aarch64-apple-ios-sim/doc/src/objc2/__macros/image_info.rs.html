1#[doc(hidden)]
2#[macro_export]
3#[cfg(target_vendor = "apple")]
4macro_rules! __statics_image_info {
5    ($hash:expr) => {
6        /// We always emit the image info tag, since we need it to:
7        /// - End up in the same codegen unit as the other statics below.
8        /// - End up in the final binary so it can be read by dyld.
9        ///
10        /// If it's not present in the codegen unit, then `ld64` won't set
11        /// `hasObjC` for that specific object file, and in turn it might
12        /// disable processing of the special Objective-C sections (currently
13        /// a category merging pass, in the future who knows what).
14        ///
15        /// Unfortunately however, this leads to duplicated tags - the linker
16        /// reports `__DATA/__objc_imageinfo has unexpectedly large size XXX`,
17        /// but things still seems to work.
18        #[cfg_attr(
19            not(all(target_os = "macos", target_arch = "x86")),
20            link_section = "__DATA,__objc_imageinfo,regular,no_dead_strip"
21        )]
22        #[cfg_attr(
23            all(target_os = "macos", target_arch = "x86"),
24            link_section = "__OBJC,__image_info,regular"
25        )]
26        #[export_name = $crate::__macros::concat!("\x01L_OBJC_IMAGE_INFO_", $hash)]
27        #[used] // Make sure this reaches the linker
28        static _IMAGE_INFO: $crate::__macros::ImageInfo = $crate::__macros::ImageInfo::system();
29    };
30}

32#[repr(C)]
33#[doc(hidden)]
34#[derive(Debug, Clone, Copy)]
35pub struct ImageInfo {
36    // These are not actually `unsigned int`, even though the docs say so
37    /// The version of the image info struct.
38    version: u32,
39    flags: u32,
40}
41
42#[allow(unused)]
43impl ImageInfo {
44    /// Unused
45    const FIX_AND_CONTINUE: u32 = 1 << 0;
46    const SUPPORTS_GARBAGE_COLLECTED: u32 = 1 << 1;
47    const REQUIRES_GARBAGE_COLLECTION: u32 = 1 << 2;
48    const OPTIMIZED_BY_DYLD: u32 = 1 << 3; // TODO
49    /// Unused
50    const CORRECTED_SYNTHESIZE: u32 = 1 << 4;
51    /// Whether we're compiling this to run on a simulator.
52    const IMAGE_IS_SIMULATED: u32 = 1 << 5;
53    /// Whether we are generating class properties.
54    const CLASS_PROPERTIES: u32 = 1 << 6;
55    const DYLD_PREOPTIMIZED: u32 = 1 << 7;
56
57    const SWIFT_ABI_VERSION_SHIFT: u32 = 8;
58    const SWIFT_ABI_VERSION_MASK: u32 = 0xff << Self::SWIFT_ABI_VERSION_SHIFT;
59    const SWIFT_MINOR_VERSION_SHIFT: u32 = 16;
60    const SWIFT_MINOR_VERSION_MASK: u32 = 0xff << Self::SWIFT_MINOR_VERSION_SHIFT;
61    const SWIFT_MAJOR_VERSION_SHIFT: u32 = 24;
62    const SWIFT_MAJOR_VERSION_MASK: u32 = 0xff << Self::SWIFT_MAJOR_VERSION_SHIFT;
63
64    /// Fetches the image info for the current runtime + target combination
65    #[inline]
66    pub const fn system() -> Self {
67        // We don't currently do anything relating to class properties, but
68        // let's mimic what Clang does!
69        let mut flags = Self::CLASS_PROPERTIES;
70
71        if cfg!(target_simulator) {
72            flags |= Self::IMAGE_IS_SIMULATED;
73        }
74
75        Self { version: 0, flags }
76    }
77}
