1//! # `@throw` and `@try/@catch` exceptions.
2//!
3//! By default, if the [`msg_send!`] macro causes an exception to be thrown,
4//! this will unwind into Rust, resulting in undefined behavior. However, this
5//! crate has an `"catch-all"` feature which, when enabled, wraps each
6//! [`msg_send!`] in a `@catch` and panics if an exception is caught,
7//! preventing Objective-C from unwinding into Rust.
8//!
9//! Most of the functionality in this module is only available when the
10//! `"exception"` feature is enabled.
11//!
12//! See the following links for more information:
13//! - [Exception Programming Topics for Cocoa](https://developer.apple.com/library/archive/documentation/Cocoa/Conceptual/Exceptions/Exceptions.html)
14//! - [The Objective-C Programming Language - Exception Handling](https://developer.apple.com/library/archive/documentation/Cocoa/Conceptual/ObjectiveC/Chapters/ocExceptionHandling.html)
15//! - [Exception Handling in LLVM](https://llvm.org/docs/ExceptionHandling.html)
16//!
17//! [`msg_send!`]: crate::msg_send

42/// An Objective-C exception.
43///
44/// While highly recommended that any exceptions you intend to throw are
45/// subclasses of `NSException`, this is not required by the runtime (similar
46/// to how Rust can panic with arbitary payloads using [`panic_any`]).
47///
48/// [`panic_any`]: std::panic::panic_any
49#[repr(transparent)]
50pub struct Exception(AnyObject);

74impl Exception {
75    fn is_nsexception(&self) -> Option<bool> {
76        if self.class().responds_to(sel!(isKindOfClass:)) {
77            // SAFETY: We only use `isKindOfClass:` on NSObject
78            let obj: *const Exception = self;
79            let obj = unsafe { obj.cast::<NSObject>().as_ref().unwrap() };
80            // Get class dynamically instead of with `class!` macro
81            Some(obj.isKindOfClass(AnyClass::get("NSException")?))
82        } else {
83            Some(false)
84        }
85    }
86}

105impl fmt::Debug for Exception {
106    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
107        write!(f, "exception ")?;
108
109        // Attempt to present a somewhat usable error message if the exception
110        // is an instance of NSException.
111        if let Some(true) = self.is_nsexception() {
112            autoreleasepool_leaking(|pool| {
113                // SAFETY: Just checked that object is an NSException
114                let (name, reason) = unsafe { (self.name(), self.reason()) };
115
116                // SAFETY: `name` and `reason` are guaranteed to be NSString.
117                let name = name
118                    .as_deref()
119                    .map(|name| unsafe { nsstring_to_str(name, pool) });
120                let reason = reason
121                    .as_deref()
122                    .map(|reason| unsafe { nsstring_to_str(reason, pool) });
123
124                let obj: &AnyObject = self.as_ref();
125                write!(f, "{obj:?} '{}'", name.unwrap_or_default())?;
126                if let Some(reason) = reason {
127                    write!(f, " reason:{reason}")?;
128                } else {
129                    write!(f, " reason:(NULL)")?;
130                }
131                Ok(())
132            })
133        } else {
134            // Fall back to `AnyObject` Debug
135            write!(f, "{:?}", self.0)
136        }
137    }
138}

140impl fmt::Display for Exception {
141    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
142        autoreleasepool_leaking(|pool| {
143            if let Some(true) = self.is_nsexception() {
144                // SAFETY: Just checked that object is an NSException
145                let reason = unsafe { self.reason() };
146
147                if let Some(reason) = &reason {
148                    // SAFETY: `reason` is guaranteed to be NSString.
149                    let reason = unsafe { nsstring_to_str(reason, pool) };
150                    return write!(f, "{reason}");
151                }
152            }
153
154            write!(f, "unknown exception")
155        })
156    }
157}

164/// Throws an Objective-C exception.
165///
166/// This is the Objective-C equivalent of Rust's [`panic!`].
167///
168///
169/// # Safety
170///
171/// This unwinds from Objective-C, and the exception must be caught using an
172/// Objective-C exception handler like [`catch`] (and specifically not
173/// [`catch_unwind`]).
174///
175/// This also invokes undefined behaviour until `C-unwind` is stabilized, see
176/// [RFC-2945] - you can try this out on nightly using the `unstable-c-unwind`
177/// feature flag.
178///
179/// [`catch_unwind`]: std::panic::catch_unwind
180/// [RFC-2945]: https://rust-lang.github.io/rfcs/2945-c-unwind-abi.html
181#[inline]
182#[cfg(feature = "exception")] // For consistency, not strictly required
183pub unsafe fn throw(exception: Retained<Exception>) -> ! {
184    // We consume the exception object since we can't make any guarantees
185    // about its mutability.
186    let ptr = exception.0.as_ptr() as *mut ffi::objc_object;
187    // SAFETY: The object is valid and non-null (nil exceptions are not valid
188    // in the old runtime).
189    unsafe { ffi::objc_exception_throw(ptr) }
190}
191
192#[cfg(feature = "exception")]
193unsafe fn try_no_ret<F: FnOnce()>(closure: F) -> Result<(), Option<Retained<Exception>>> {
194    #[cfg(not(feature = "unstable-c-unwind"))]
195    let f = {
196        extern "C" fn try_objc_execute_closure<F>(closure: &mut Option<F>)
197        where
198            F: FnOnce(),
199        {
200            // This is always passed Some, so it's safe to unwrap
201            let closure = closure.take().unwrap();
202            closure();
203        }
204
205        let f: extern "C" fn(&mut Option<F>) = try_objc_execute_closure;
206        let f: extern "C" fn(*mut c_void) = unsafe { mem::transmute(f) };
207        f
208    };
209
210    #[cfg(feature = "unstable-c-unwind")]
211    let f = {
212        extern "C-unwind" fn try_objc_execute_closure<F>(closure: &mut Option<F>)
213        where
214            F: FnOnce(),
215        {
216            // This is always passed Some, so it's safe to unwrap
217            let closure = closure.take().unwrap();
218            closure();
219        }
220
221        let f: extern "C-unwind" fn(&mut Option<F>) = try_objc_execute_closure;
222        let f: extern "C-unwind" fn(*mut c_void) = unsafe { mem::transmute(f) };
223        f
224    };
225
226    // Wrap the closure in an Option so it can be taken
227    let mut closure = Some(closure);
228    let context: *mut Option<F> = &mut closure;
229    let context = context.cast();
230
231    let mut exception = ptr::null_mut();
232    let success = unsafe { ffi::try_catch(f, context, &mut exception) };
233
234    if success == 0 {
235        Ok(())
236    } else {
237        // SAFETY:
238        // The exception is always a valid object or NULL.
239        //
240        // Since we do a retain inside `extern/exception.m`, the object has
241        // +1 retain count.
242        //
243        // Code throwing an exception know that they don't hold sole access to
244        // that object any more, so even if the type was originally mutable,
245        // it is okay to create a new `Retained` to it here.
246        Err(unsafe { Retained::from_raw(exception.cast()) })
247    }
248}

250/// Tries to execute the given closure and catches an Objective-C exception
251/// if one is thrown.
252///
253/// This is the Objective-C equivalent of Rust's [`catch_unwind`].
254/// Accordingly, if your Rust code is compiled with `panic=abort` this cannot
255/// catch the exception.
256///
257/// [`catch_unwind`]: std::panic::catch_unwind
258///
259///
260/// # Errors
261///
262/// Returns a `Result` that is either `Ok` if the closure succeeded without an
263/// exception being thrown, or an `Err` with the exception. The exception is
264/// automatically released.
265///
266/// The exception is `None` in the extremely exceptional case that the
267/// exception object is `nil`. This should basically never happen, but is
268/// technically possible on some systems with `@throw nil`, or in OOM
269/// situations.
270///
271///
272/// # Safety
273///
274/// The given closure must not panic (e.g. normal Rust unwinding into this
275/// causes undefined behaviour).
276///
277/// Additionally, this unwinds through the closure from Objective-C, which is
278/// undefined behaviour until `C-unwind` is stabilized, see [RFC-2945] - you
279/// can try this out on nightly using the `unstable-c-unwind` feature flag.
280///
281/// [RFC-2945]: https://rust-lang.github.io/rfcs/2945-c-unwind-abi.html
282#[cfg(feature = "exception")]
283pub unsafe fn catch<R>(
284    closure: impl FnOnce() -> R + UnwindSafe,
285) -> Result<R, Option<Retained<Exception>>> {
286    let mut value = None;
287    let value_ref = &mut value;
288    let closure = move || {
289        *value_ref = Some(closure());
290    };
291    let result = unsafe { try_no_ret(closure) };
292    // If the try succeeded, value was set so it's safe to unwrap
293    result.map(|()| value.unwrap_or_else(|| unreachable!()))
294}
