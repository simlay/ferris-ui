102/// Implement PartialEq, Eq and Hash using pointer semantics; there's not
103/// really a better way to do it for this type
104macro_rules! standard_pointer_impls {
105    ($name:ident) => {
106        impl PartialEq for $name {
107            #[inline]
108            fn eq(&self, other: &Self) -> bool {
109                self.as_ptr() == other.as_ptr()
110            }
111        }
112        impl Eq for $name {}
113        impl hash::Hash for $name {
114            #[inline]
115            fn hash<H: hash::Hasher>(&self, state: &mut H) {
116                self.as_ptr().hash(state)
117            }
118        }
119    };
120}

155/// A method selector.
156///
157/// The Rust equivalent of Objective-C's `SEL` type. You can create this
158/// statically using the [`sel!`] macro.
159///
160/// The main reason the Objective-C runtime uses a custom type for selectors,
161/// as opposed to a plain c-string, is to support efficient comparison - a
162/// a selector is effectively an [interned string], so this makes equiality
163/// comparisons very cheap.
164///
165/// This struct guarantees the null-pointer optimization, namely that
166/// `Option<Sel>` is the same size as `Sel`.
167///
168/// [`sel!`]: crate::sel
169/// [interned string]: https://en.wikipedia.org/wiki/String_interning
170#[repr(transparent)]
171#[derive(Copy, Clone)]
172#[doc(alias = "SEL")]
173#[doc(alias = "objc_selector")]
174pub struct Sel {
175    ptr: NonNull<ffi::objc_selector>,
176}
177
178// SAFETY: Sel is immutable (and can be retrieved from any thread using the
179// `sel!` macro).
180unsafe impl Sync for Sel {}
181unsafe impl Send for Sel {}
182impl UnwindSafe for Sel {}
183impl RefUnwindSafe for Sel {}

185impl Sel {
186    #[inline]
187    #[doc(hidden)]
188    pub const unsafe fn __internal_from_ptr(ptr: *const ffi::objc_selector) -> Self {
189        // Used in static selectors.
190        // SAFETY: Upheld by caller.
191        let ptr = unsafe { NonNull::new_unchecked(ptr as *mut ffi::objc_selector) };
192        Self { ptr }
193    }
194
195    #[inline]
196    pub(crate) unsafe fn from_ptr(ptr: *const ffi::objc_selector) -> Option<Self> {
197        // SAFETY: Caller verifies that the pointer is valid.
198        NonNull::new(ptr as *mut ffi::objc_selector).map(|ptr| Self { ptr })
199    }
200
201    /// Get a pointer to the raw selector.
202    ///
203    /// Useful when working with raw FFI methods.
204    #[inline]
205    pub const fn as_ptr(&self) -> *const ffi::objc_selector {
206        self.ptr.as_ptr()
207    }
208
209    // We explicitly don't do #[track_caller] here, since we expect the error
210    // to never actually happen.
211    pub(crate) unsafe fn register_unchecked(name: *const c_char) -> Self {
212        let ptr = unsafe { ffi::sel_registerName(name) };
213        // SAFETY: `sel_registerName` declares return type as `SEL _Nonnull`,
214        // at least when input is also `_Nonnull` (which it is in our case).
215        //
216        // Looking at the source code, it can fail and will return NULL if
217        // allocating space for the selector failed (which then subsequently
218        // invokes UB by calling `memcpy` with a NULL argument):
219        // <https://github.com/apple-oss-distributions/objc4/blob/objc4-841.13/runtime/objc-os.h#L1002-L1004>
220        //
221        // I suspect this will be really uncommon in practice, since the
222        // called selector is almost always going to be present in the binary
223        // already; but alas, we'll handle it!
224        unsafe { Self::from_ptr(ptr).expect("failed allocating selector") }
225    }

227    /// Registers a selector with the Objective-C runtime.
228    ///
229    /// This is the dynamic version of the [`sel!`] macro, prefer to use that
230    /// when your selector is static.
231    ///
232    /// [`sel!`]: crate::sel
233    ///
234    ///
235    /// # Panics
236    ///
237    /// Panics if `name` contains an internal NUL byte, or if the runtime
238    /// failed allocating space for the selector.
239    #[doc(alias = "sel_registerName")]
240    pub fn register(name: &str) -> Self {
241        let name = CString::new(name).unwrap();
242        // SAFETY: Input is a non-null, NUL-terminated C-string pointer.
243        unsafe { Self::register_unchecked(name.as_ptr()) }
244    }
245
246    /// Returns the string representation of the selector.
247    ///
248    /// # Panics
249    ///
250    /// Panics if the selector is not valid UTF-8 (however unlikely!)
251    #[doc(alias = "sel_getName")]
252    pub fn name(&self) -> &str {
253        // SAFETY: Input is non-null selector. Declares return type as
254        // `const char * _Nonnull`, source code agrees.
255        let ptr = unsafe { ffi::sel_getName(self.as_ptr()) };
256        // SAFETY: The string is a valid C-style NUL-terminated string, and
257        // likely has static lifetime since the selector has static lifetime
258        // (though we bind it to `&self` to be safe).
259        let name = unsafe { CStr::from_ptr(ptr) };
260        str::from_utf8(name.to_bytes()).unwrap()
261    }

263    pub(crate) fn number_of_arguments(self) -> usize {
264        self.name()
265            .as_bytes()
266            .iter()
267            .filter(|&&b| b == b':')
268            .count()
269    }

272impl PartialEq for Sel {
273    #[inline]
274    fn eq(&self, other: &Self) -> bool {
275        if cfg!(feature = "gnustep-1-7") {
276            // GNUStep implements "typed" selectors, which means their pointer
277            // values sometimes differ; so let's use the runtime-provided
278            // `sel_isEqual`.
279            unsafe { Bool::from_raw(ffi::sel_isEqual(self.as_ptr(), other.as_ptr())).as_bool() }
280        } else {
281            // `ffi::sel_isEqual` uses pointer comparison on Apple (the
282            // documentation explicitly notes this); so as an optimization,
283            // let's do that as well!
284            self.as_ptr() == other.as_ptr()
285        }
286    }
287}

392    /// Returns a pointer to the instance variable / ivar on the given object.
393    ///
394    /// This is similar to [`UnsafeCell::get`], see that for more information
395    /// on what is and isn't safe to do.
396    ///
397    /// Usually you will have defined the instance variable yourself with
398    /// [`ClassBuilder::add_ivar`], the type of the ivar `T` must match the
399    /// type used in that.
400    ///
401    /// Library implementors are strongly encouraged to expose a safe
402    /// interface to the ivar.
403    ///
404    /// [`UnsafeCell::get`]: core::cell::UnsafeCell::get
405    /// [`ClassBuilder::add_ivar`]: crate::declare::ClassBuilder::add_ivar
406    ///
407    ///
408    /// # Panics
409    ///
410    /// Panics when `debug_assertions` are enabled if the type encoding of the
411    /// ivar differs from the type encoding of `T`.
412    ///
413    ///
414    /// # Safety
415    ///
416    /// The object must have the given instance variable on it, and it must be
417    /// of type `T`. Any invariants that the object have assumed about the
418    /// value of the instance variable must not be violated.
419    ///
420    /// Note that an object can have multiple instance variables with the same
421    /// name; you must ensure that when the instance variable was retrieved,
422    /// was retrieved from the class that it was defined on. In particular,
423    /// getting a class dynamically using e.g. [`AnyObject::class`], and using
424    /// an instance variable from that here is _not_ sound in general.
425    ///
426    /// No thread syncronization is done on accesses to the variable, so you
427    /// must ensure that any access to the returned pointer do not cause data
428    /// races, and that Rust's mutability rules are not otherwise violated.
429    #[inline]
430    pub unsafe fn load_ptr<T: Encode>(&self, obj: &AnyObject) -> *mut T {
431        self.debug_assert_encoding(&T::ENCODING);
432
433        let ptr = NonNull::from(obj);
434        // SAFETY: That the ivar is valid is ensured by the caller
435        let ptr = unsafe { AnyObject::ivar_at_offset::<T>(ptr, self.offset()) };
436
437        // Safe as *mut T because `self` is `UnsafeCell`
438        ptr.as_ptr()
439    }

441    /// Returns a reference to the instance variable with the given name.
442    ///
443    /// See [`Ivar::load_ptr`] for more information.
444    ///
445    ///
446    /// # Panics
447    ///
448    /// Panics when `debug_assertions` are enabled if the type encoding of the
449    /// ivar differs from the type encoding of `T`.
450    ///
451    ///
452    /// # Safety
453    ///
454    /// The object must have the given instance variable on it, and it must be
455    /// of type `T`.
456    ///
457    /// No thread syncronization is done, so you must ensure that no other
458    /// thread is concurrently mutating the variable. This requirement can be
459    /// considered upheld if all mutation happens through [`Ivar::load_mut`]
460    /// (since that takes the object mutably).
461    #[inline]
462    pub unsafe fn load<'obj, T: Encode>(&self, obj: &'obj AnyObject) -> &'obj T {
463        // SAFETY: That the ivar is valid as `&T` is ensured by the caller,
464        // and the reference is properly bound to the object.
465        unsafe { self.load_ptr::<T>(obj).as_ref().unwrap_unchecked() }
466    }

468    /// Returns a mutable reference to the ivar with the given name.
469    ///
470    /// See [`Ivar::load_ptr`] for more information.
471    ///
472    ///
473    /// # Panics
474    ///
475    /// Panics when `debug_assertions` are enabled if the type encoding of the
476    /// ivar differs from the type encoding of `T`.
477    ///
478    ///
479    /// # Safety
480    ///
481    /// The object must have an instance variable with the given name, and it
482    /// must be of type `T`.
483    ///
484    /// This access happens through `&mut`, which means we know it to be the
485    /// only reference, hence you do not need to do any work to ensure that
486    /// data races do not happen.
487    #[inline]
488    pub unsafe fn load_mut<'obj, T: Encode>(&self, obj: &'obj mut AnyObject) -> &'obj mut T {
489        self.debug_assert_encoding(&T::ENCODING);
490
491        let ptr = NonNull::from(obj);
492        // SAFETY: That the ivar is valid is ensured by the caller
493        let mut ptr = unsafe { AnyObject::ivar_at_offset::<T>(ptr, self.offset()) };
494
495        // SAFETY: That the ivar is valid as `&mut T` is ensured by taking an
496        // `&mut` object
497        unsafe { ptr.as_mut() }
498    }

625    /// Set the implementation of this method.
626    ///
627    /// Note that any thread may at any point be changing method
628    /// implementations, so if you intend to call the previous method as
629    /// returned by e.g. [`Self::implementation`], beware that that may now be
630    /// stale.
631    ///
632    /// The previous implementation is returned from this function though, so
633    /// you can call that instead.
634    ///
635    /// See [Apple's documentation](https://developer.apple.com/documentation/objectivec/1418707-method_setimplementation?language=objc).
636    ///
637    ///
638    /// # Safety
639    ///
640    /// The given implementation function pointer must:
641    ///
642    /// 1. Have the signature expected by the Objective-C runtime and callers
643    ///    of this method.
644    ///
645    /// 2. Be at least as safe as the existing method, i.e. by overriding the
646    ///    previous method, it should not be possible for the program to cause

1261        let ptr: *const AnyClass = ptr.cast();
1262        // SAFETY: The class is not NULL because the object is not NULL.

1273    /// Offset an object pointer to get a pointer to an ivar.
1274    ///
1275    ///
1276    /// # Safety
1277    ///
1278    /// The offset must be valid for the given type.
1279    #[inline]
1280    pub(crate) unsafe fn ivar_at_offset<T>(ptr: NonNull<Self>, offset: isize) -> NonNull<T> {
1281        // `offset` is given in bytes, so we convert to `u8` and back to `T`
1282        let ptr: NonNull<u8> = ptr.cast();
1283        let ptr: *mut u8 = ptr.as_ptr();
1284        // SAFETY: The offset is valid
1285        let ptr: *mut u8 = unsafe { ptr.offset(offset) };
1286        // SAFETY: The offset operation is guaranteed to not end up computing
1287        // a NULL pointer.
1288        let ptr: NonNull<u8> = unsafe { NonNull::new_unchecked(ptr) };
1289        let ptr: NonNull<T> = ptr.cast();
1290        ptr
1291    }
