1//! Utility for parsing an Objective-C method type encoding.
2//!
3//! TODO: Move this to `objc2-encode` when more stable.
4use core::fmt;
5use core::num::ParseIntError;
6use core::str;
7use std::error::Error;
8
9use crate::encode::{Encoding, EncodingBox, ParseError};
10
11#[derive(Debug, PartialEq, Eq)]
12pub(crate) struct MethodEncodingIter<'a> {
13    s: &'a str,
14}
15
16impl<'a> MethodEncodingIter<'a> {
17    pub(crate) fn new(s: &'a str) -> Self {
18        Self { s }
19    }
20
21    pub(crate) fn extract_return(
22        &mut self,
23    ) -> Result<(EncodingBox, Option<isize>), EncodingParseError> {
24        // TODO: Verify stack layout
25        self.next().ok_or(EncodingParseError::MissingReturn)?
26    }
27
28    pub(crate) fn verify_receiver(&mut self) -> Result<(), EncodingParseError> {
29        // TODO: Verify stack layout
30        let (enc, _stack_layout) = self.next().ok_or(EncodingParseError::MissingReceiver)??;
31        if !Encoding::Object.equivalent_to_box(&enc) {
32            return Err(EncodingParseError::InvalidReceiver(enc));
33        }
34        Ok(())
35    }
36
37    pub(crate) fn verify_sel(&mut self) -> Result<(), EncodingParseError> {
38        let (enc, _stack_layout) = self.next().ok_or(EncodingParseError::MissingSel)??;
39        if !Encoding::Sel.equivalent_to_box(&enc) {
40            return Err(EncodingParseError::InvalidSel(enc));
41        }
42        Ok(())
43    }
44
45    fn extract_encoding(&mut self) -> Result<(EncodingBox, Option<isize>), EncodingParseError> {
46        // See also the following other approaches:
47        // objrs: https://gitlab.com/objrs/objrs/-/blob/b4f6598696b3fa622e6fddce7aff281770b0a8c2/src/test.rs
48        // libobjc2: https://github.com/gnustep/libobjc2/blob/v2.1/encoding2.c
49        // objc4: https://github.com/apple-oss-distributions/objc4/blob/objc4-841.13/runtime/objc-typeencoding.mm
50
51        let encoding = EncodingBox::from_start_of_str(&mut self.s)?;
52        let stack_layout = parse_stack_layout(&mut self.s)?;
53
54        Ok((encoding, stack_layout))
55    }
56}
57
58impl<'a> Iterator for MethodEncodingIter<'a> {
59    type Item = Result<(EncodingBox, Option<isize>), EncodingParseError>;
60
61    fn next(&mut self) -> Option<Self::Item> {
62        if self.s.is_empty() {
63            return None;
64        }
65        Some(self.extract_encoding())
66    }
67}
68
69// TODO: Is `isize` correct here?
70fn parse_stack_layout(s: &mut &str) -> Result<Option<isize>, ParseIntError> {
71    let rest = s.trim_start_matches(|c: char| c.is_ascii_digit() || c == '-' || c == '+');
72    let stack_layout = &s[..s.len() - rest.len()];
73    *s = rest;
74
75    if stack_layout.is_empty() {
76        return Ok(None);
77    }
78    stack_layout.parse().map(Some)
79}

81#[derive(Debug, PartialEq, Eq, Hash)]
82pub(crate) enum EncodingParseError {
83    ParseError(ParseError),
84    InvalidStackLayoutInteger,
85    MissingReturn,
86    MissingReceiver,
87    MissingSel,
88    InvalidReceiver(EncodingBox),
89    InvalidSel(EncodingBox),
90}
91
92impl From<ParseError> for EncodingParseError {
93    fn from(e: ParseError) -> Self {
94        Self::ParseError(e)
95    }
96}
97
98impl From<ParseIntError> for EncodingParseError {
99    fn from(_: ParseIntError) -> Self {
100        Self::InvalidStackLayoutInteger
101    }
102}
103
104impl fmt::Display for EncodingParseError {
105    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
106        if !matches!(self, Self::ParseError(_)) {
107            write!(f, "failed parsing encoding: ")?;
108        }
109
110        match self {
111            Self::ParseError(e) => write!(f, "{e}")?,
112            Self::InvalidStackLayoutInteger => write!(f, "invalid integer for stack layout")?,
113            Self::MissingReturn => write!(f, "return type must be present")?,
114            Self::MissingReceiver => write!(f, "receiver type must be present")?,
115            Self::MissingSel => write!(f, "selector type must be present")?,
116            Self::InvalidReceiver(enc) => {
117                write!(f, "receiver encoding must be '@', but it was '{enc}'")?;
118            }
119            Self::InvalidSel(enc) => {
120                write!(f, "selector encoding must be '@', but it was '{enc}'")?;
121            }
122        }
123        write!(f, ". This is likely a bug, please report it!")
124    }
125}

129#[cfg(test)]
130mod tests {
131    use super::*;
132    use alloc::boxed::Box;
133    use alloc::vec;
134    use alloc::vec::Vec;
135
136    fn assert_stack_layout(mut types: &str, expected: Option<isize>, rest: &str) {
137        let sl = parse_stack_layout(&mut types).unwrap();
138        assert_eq!(sl, expected);
139        assert_eq!(types, rest);
140    }
141
142    #[test]
143    fn stack_layout_extract() {
144        assert_stack_layout("", None, "");
145        assert_stack_layout("abc", None, "abc");
146        assert_stack_layout("abc12abc", None, "abc12abc");
147        assert_stack_layout("0", Some(0), "");
148        assert_stack_layout("1abc", Some(1), "abc");
149        assert_stack_layout("42def24", Some(42), "def24");
150        assert_stack_layout("1234567890xyz", Some(1234567890), "xyz");
151
152        assert_stack_layout("-1a", Some(-1), "a");
153        assert_stack_layout("-1a", Some(-1), "a");
154
155        // GNU runtime's register parameter hint??
156        assert_stack_layout("+1a", Some(1), "a");
157    }
158
159    fn assert_encoding_extract(s: &str, expected: &[(EncodingBox, Option<isize>)]) {
160        let actual: Vec<_> = MethodEncodingIter::new(s)
161            .collect::<Result<_, _>>()
162            .unwrap_or_else(|e| panic!("{}", e));
163        assert_eq!(&actual, expected);
164    }
165
166    #[test]
167    fn parse_bitfield() {
168        assert_encoding_extract(
169            "@48@0:8Ad16r^*24{bitfield=b64b1}32i48",
170            &[
171                (EncodingBox::Object, Some(48)),
172                (EncodingBox::Object, Some(0)),
173                (EncodingBox::Sel, Some(8)),
174                (EncodingBox::Atomic(Box::new(EncodingBox::Double)), Some(16)),
175                (
176                    EncodingBox::Pointer(Box::new(EncodingBox::String)),
177                    Some(24),
178                ),
179                (
180                    EncodingBox::Struct(
181                        "bitfield".into(),
182                        vec![
183                            EncodingBox::BitField(64, None),
184                            EncodingBox::BitField(1, None),
185                        ],
186                    ),
187                    Some(32),
188                ),
189                (EncodingBox::Int, Some(48)),
190            ],
191        );
192    }
