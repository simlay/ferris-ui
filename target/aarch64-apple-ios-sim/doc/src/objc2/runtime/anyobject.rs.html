14/// An Objective-C object.
15///
16/// This is slightly different from [`NSObject`] in that it may represent an
17/// instance of an _arbitrary_ Objective-C class (e.g. it does not have to be
18/// a subclass of `NSObject`, so it can represent other root classes like
19/// `NSProxy`).
20///
21/// `Retained<AnyObject>` is equivalent to Objective-C's `id _Nonnull`.
22///
23/// This is an opaque type that contains [`UnsafeCell`], and is similar to
24/// that in that one can safely access and perform interior mutability on this
25/// (both via [`msg_send!`] and through ivars), so long as Rust's mutability
26/// rules are upheld, and that data races are avoided.
27///
28/// Note: This is intentionally neither [`Sync`], [`Send`], [`UnwindSafe`],
29/// [`RefUnwindSafe`] nor [`Unpin`], since that is something that may change
30/// depending on the specific subclass. For example, `NSAutoreleasePool` is
31/// not `Send`, it has to be deallocated on the same thread that it was
32/// created. `NSLock` is not `Send` either.
33///
34/// [`NSObject`]: crate::runtime::NSObject
35/// [`UnsafeCell`]: core::cell::UnsafeCell
36/// [`msg_send!`]: crate::msg_send
37/// [`UnwindSafe`]: std::panic::UnwindSafe
38/// [`RefUnwindSafe`]: std::panic::RefUnwindSafe
39/// [`Unpin`]: std::marker::Unpin
40#[doc(alias = "id")]
41#[doc(alias = "objc_object")]
42#[repr(C)]
43pub struct AnyObject {
44    // `isa` field is deprecated, so we don't expose it here.
45    //
46    // Also, we need this to be a zero-sized, so that the compiler doesn't
47    // assume anything about the layout.
48    //
49    // Use `object_getClass` instead.
50    _priv: [u8; 0],
51    _p: ffi::OpaqueData,
52}

72    /// Dynamically find the class of this object.
73    ///
74    ///
75    /// # Panics
76    ///
77    /// May panic if the object is invalid (which may be the case for objects
78    /// returned from unavailable `init`/`new` methods).
79    ///
80    ///
81    /// # Example
82    ///
83    /// Check that an instance of `NSObject` has the precise class `NSObject`.
84    ///
85    /// ```
86    /// use objc2::ClassType;
87    /// use objc2::runtime::NSObject;
88    ///
89    /// let obj = NSObject::new();
90    /// assert_eq!(obj.class(), NSObject::class());
91    /// ```
92    #[inline]
93    #[doc(alias = "object_getClass")]
94    pub fn class(&self) -> &'static AnyClass {
95        let ptr = unsafe { ffi::object_getClass(self) };
96        // SAFETY: The pointer is valid, and it is safe as `'static` since
97        // classes are static (can also be verified with the fact that they
98        // can be retrieved via `AnyClass::get(self.class().name())`).
99        let cls = unsafe { ptr.as_ref() };
100
101        // The class _should_ not be NULL, because the docs only say that if
102        // the object is NULL, the class also is; and in practice, certain
103        // invalid objects can contain a NULL isa pointer.
104        cls.unwrap_or_else(|| panic!("invalid object {:?} (had NULL class)", self as *const Self))
105    }

107    /// Change the class of the object at runtime.
108    ///
109    /// Returns the object's previous class.
110    ///
111    ///
112    /// # Safety
113    ///
114    /// The new class must:
115    ///
116    /// 1. Be a subclass of the object's current class.
117    ///
118    /// 2. The subclass must not add any instance variables - importantly, the
119    ///    instance size of old and the new classes must be the same.
120    ///
121    /// 3. Any overridden methods on the new class must be fully compatible
122    ///    with the old ones.
123    ///
124    /// Note that in the general case, where arbitrary parts of the program
125    /// may be trying to modify the class of the object concurrently, these
126    /// requirements are not actually possible to uphold.
127    ///
128    /// Since usage of this function is expected to be extremely rare, and
129    /// even more so trying to do it concurrently, it is recommended that you
130    /// verify that the returned class is what you would expect, and if not,
131    /// panic.
132    #[inline]
133    #[doc(alias = "object_setClass")]
134    pub unsafe fn set_class<'s>(this: &Self, cls: &AnyClass) -> &'s AnyClass {
135        let this: *const Self = this;
136        let this = this as *mut Self;
137        let ptr = unsafe { ffi::object_setClass(this, cls) };
138        // SAFETY: The class is not NULL because the object is not NULL.
139        let old_cls = unsafe { ptr.as_ref().unwrap_unchecked() };
140        // TODO: Check the superclass requirement too?
141        debug_assert_eq!(
142            old_cls.instance_size(),
143            cls.instance_size(),
144            "old and new class sizes were not equal; this is UB!"
145        );
146        old_cls
147    }

149    /// Offset an object pointer to get a pointer to an ivar.
150    ///
151    ///
152    /// # Safety
153    ///
154    /// The offset must be valid for the given type.
155    #[inline]
156    pub(crate) unsafe fn ivar_at_offset<T>(ptr: NonNull<Self>, offset: isize) -> NonNull<T> {
157        // `offset` is given in bytes, so we convert to `u8` and back to `T`
158        let ptr: NonNull<u8> = ptr.cast();
159        let ptr: *mut u8 = ptr.as_ptr();
160        // SAFETY: The offset is valid
161        let ptr: *mut u8 = unsafe { ptr.offset(offset) };
162        // SAFETY: The offset operation is guaranteed to not end up computing
163        // a NULL pointer.
164        let ptr: NonNull<u8> = unsafe { NonNull::new_unchecked(ptr) };
165        let ptr: NonNull<T> = ptr.cast();
166        ptr
167    }

169    pub(crate) fn lookup_instance_variable_dynamically(&self, name: &str) -> &'static Ivar {
170        let name = CString::new(name).unwrap();
171        let cls = self.class();
172        cls.instance_variable(&name)
173            .unwrap_or_else(|| panic!("ivar {name:?} not found on class {cls}"))
174    }

176    /// Use [`Ivar::load`] instead.
177    ///
178    ///
179    /// # Safety
180    ///
181    /// The object must have an instance variable with the given name, and it
182    /// must be of type `T`.
183    ///
184    /// See [`Ivar::load_ptr`] for details surrounding this.
185    #[deprecated = "this is difficult to use correctly, use `Ivar::load` instead."]

187        let ivar = self.lookup_instance_variable_dynamically(name);
188        // SAFETY: Upheld by caller

208    pub(crate) fn is_kind_of_class(&self, cls: &AnyClass) -> Bool {
209        // SAFETY: The signature is correct.
210        //
211        // Note that `isKindOfClass:` is not available on every object, but it
212        // is still safe to _use_, since the runtime will simply crash if the
213        // selector isn't implemented. This is of course not _ideal_, but it
214        // works for all of Apple's Objective-C classes, and it's what Swift
215        // does.
216        //
217        // In theory, someone could have made a root object, and overwritten
218        // `isKindOfClass:` to do something bogus - but that would conflict
219        // with normal Objective-C code as well, so we will consider such a
220        // thing unsound by construction.
221        unsafe { msg_send![self, isKindOfClass: cls] }
222    }

224    /// Attempt to downcast the object to a class of type `T`.
225    ///
226    /// This is the reference-variant. Use [`Retained::downcast`] if you want
227    /// to convert a retained object to another type.
228    ///
229    /// [`Retained::downcast`]: crate::rc::Retained::downcast
230    ///
231    ///
232    /// # Mutable classes
233    ///
234    /// Some classes have immutable and mutable variants, such as `NSString`
235    /// and `NSMutableString`.
236    ///
237    /// When some Objective-C API signature says it gives you an immutable
238    /// class, it generally expects you to not mutate that, even though it may
239    /// technically be mutable "under the hood".
240    ///
241    /// So using this method to convert a `NSString` to a `NSMutableString`,
242    /// while not unsound, is generally frowned upon unless you created the
243    /// string yourself, or the API explicitly documents the string to be
244    /// mutable.
245    ///
246    /// See Apple's [documentation on mutability][apple-mut] and [on
247    /// `isKindOfClass:`][iskindof-doc] for more details.
248    ///
249    /// [iskindof-doc]: https://developer.apple.com/documentation/objectivec/1418956-nsobject/1418511-iskindofclass?language=objc
250    /// [apple-mut]: https://developer.apple.com/library/archive/documentation/General/Conceptual/CocoaEncyclopedia/ObjectMutability/ObjectMutability.html
251    ///
252    ///
253    /// # Generic classes
254    ///
255    /// Objective-C generics are called "lightweight generics", and that's
256    /// because they aren't exposed in the runtime. This makes it impossible
257    /// to safely downcast to generic collections, so this is disallowed by
258    /// this method.
259    ///
260    /// You can, however, safely downcast to generic collections where all the
261    /// type-parameters are [`AnyObject`].
262    ///
263    ///
264    /// # Panics
265    ///
266    /// This works internally by calling `isKindOfClass:`. That means that the
267    /// object must have the instance method of that name, and an exception
268    /// will be thrown (if CoreFoundation is linked) or the process will abort
269    /// if that is not the case. In the vast majority of cases, you don't need
270    /// to worry about this, since both root objects [`NSObject`] and
271    /// `NSProxy` implement this method.
272    ///
273    /// [`NSObject`]: crate::runtime::NSObject
274    ///
275    ///
276    /// # Examples
277    ///
278    /// Cast an `NSString` back and forth from `NSObject`.
279    ///
280    /// ```
281    /// use objc2::rc::Retained;
282    /// # use objc2::runtime::NSObject;
283    /// # objc2::extern_class!(
284    /// #     #[unsafe(super(NSObject))]
285    /// #     struct NSString;
286    /// # );
287    /// # impl NSString {
288    /// #     objc2::extern_methods!(
289    /// #         #[unsafe(method(new))]
290    /// #         fn new() -> objc2::rc::Retained<Self>;
291    /// #     );
292    /// # }
293    /// # #[cfg(available_in_foundation)]
294    /// use objc2_foundation::{NSObject, NSString};
295    ///
296    /// let obj: Retained<NSObject> = NSString::new().into_super();
297    /// let string = obj.downcast_ref::<NSString>().unwrap();
298    /// // Or with `downcast`, if we do not need the object afterwards
299    /// let string = obj.downcast::<NSString>().unwrap();
300    /// ```
301    ///
302    /// Try (and fail) to cast an `NSObject` to an `NSString`.
303    ///
304    /// ```
305    /// # use objc2::runtime::NSObject;
306    /// # objc2::extern_class!(
307    /// #     #[unsafe(super(NSObject))]
308    /// #     struct NSString;
309    /// # );
310    /// # #[cfg(available_in_foundation)]
311    /// use objc2_foundation::{NSObject, NSString};
312    ///
313    /// let obj = NSObject::new();
314    /// assert!(obj.downcast_ref::<NSString>().is_none());
315    /// ```
316    ///
317    /// Try to cast to an array of strings.
318    ///
319    /// ```ignore
320    /// use objc2_foundation::{NSArray, NSObject, NSString};
321    ///
322    /// let arr = NSArray::from_retained_slice(&[NSObject::new()]);
323    /// // This is invalid and doesn't type check.
324    /// let arr = arr.downcast_ref::<NSArray<NSString>>();
325    /// ```
326    ///
327    /// This fails to compile, since it would require enumerating over the
328    /// array to ensure that each element is of the desired type, which is a
329    /// performance pitfall.
330    ///
331    /// Downcast when processing each element instead.
332    ///
333    /// ```ignore
334    /// use objc2_foundation::{NSArray, NSObject, NSString};
335    ///
336    /// let arr = NSArray::from_retained_slice(&[NSObject::new()]);
337    ///
338    /// for elem in arr {
339    ///     if let Some(data) = elem.downcast_ref::<NSString>() {
340    ///         // handle `data`
341    ///     }
342    /// }
343    /// ```
344    #[inline]
345    pub fn downcast_ref<T: DowncastTarget>(&self) -> Option<&T> {
346        if self.is_kind_of_class(T::class()).as_bool() {
347            // SAFETY: Just checked that the object is a class of type `T`.
348            //
349            // Generic `T` like `NSArray<NSString>` are ruled out by
350            // `T: DowncastTarget`.
351            Some(unsafe { &*(self as *const Self).cast::<T>() })
352        } else {
353            None
354        }
355    }
