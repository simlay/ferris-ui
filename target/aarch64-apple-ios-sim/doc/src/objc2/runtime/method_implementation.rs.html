13/// Types that can be used as the implementation of an Objective-C method.
14///
15/// This is a sealed trait that is implemented for a lot of `extern "C"`
16/// function pointer types.
17//
18// Note: `Sized` is intentionally added to make the trait dyn-incompatible.
19pub trait MethodImplementation: private::Sealed + Sized {
20    /// The callee type of the method.
21    type Callee: ?Sized + RefEncode;
22
23    /// The argument types of the method.
24    type Arguments: EncodeArguments;
25
26    /// The return type of the method.
27    type Return: EncodeReturn;
28
29    #[doc(hidden)]
30    fn __imp(self) -> Imp;
31}
32
33macro_rules! method_impl_inner {
34    ($(($unsafe:ident))? $abi:literal; $($t:ident),*) => {
35        impl<T, R, $($t),*> private::Sealed for $($unsafe)? extern $abi fn(T, Sel $(, $t)*) -> R
36        where
37            T: ?Sized + MessageReceiver,
38            R: EncodeReturn,
39            $($t: EncodeArgument,)*
40        {}
41
42        impl<T, R, $($t),*> MethodImplementation for $($unsafe)? extern $abi fn(T, Sel $(, $t)*) -> R
43        where
44            T: ?Sized + MessageReceiver,
45            R: EncodeReturn,
46            $($t: EncodeArgument,)*
47        {
48            type Callee = T::__Inner;
49            type Arguments = ($($t,)*);
50            type Return = R;
51
52            fn __imp(self) -> Imp {
53                // SAFETY: Transmuting to an `unsafe` function pointer
54                unsafe { mem::transmute(self) }
55            }
56        }
57
58        impl<T, $($t),*> private::Sealed for $($unsafe)? extern $abi fn(Allocated<T>, Sel $(, $t)*) -> RetainedReturnValue
59        where
60            T: ?Sized + Message,
61            $($t: EncodeArgument,)*
62        {}
63
64        #[doc(hidden)]
65        impl<T, $($t),*> MethodImplementation for $($unsafe)? extern $abi fn(Allocated<T>, Sel $(, $t)*) -> RetainedReturnValue
66        where
67            T: ?Sized + Message,
68            $($t: EncodeArgument,)*
69        {
70            type Callee = T;
71            type Arguments = ($($t,)*);
72            type Return = RetainedReturnValue;
73
74            fn __imp(self) -> Imp {
75                // SAFETY: `Allocated<T>` is the same as `NonNull<T>`, except
76                // with the assumption of a +1 calling convention.
77                //
78                // The calling convention is ensured to be upheld by having
79                // `RetainedReturnValue` in the type, since that type is private
80                // and hence only internal macros like `#[method_id]` will be
81                // able to produce it (and that, in turn, only allows it if
82                // the selector is `init` as checked by `MessageReceiveRetained`).
83                unsafe { mem::transmute(self) }
84            }
85        }
86    };
87}
88
89macro_rules! method_impl {
90    ($($t:ident),*) => {
91        method_impl_inner!((unsafe) "C"; $($t),*);
92        method_impl_inner!("C"; $($t),*);
93        method_impl_inner!((unsafe) "C-unwind"; $($t),*);
94        method_impl_inner!("C-unwind"; $($t),*);
95    };
96}
