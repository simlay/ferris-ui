9/// A type used to identify and manage memory zones.
10///
11/// Zones are ignored on all newer platforms, you should very rarely need to
12/// use this, but may be useful if you need to implement `copyWithZone:` or
13/// `allocWithZone:`.
14///
15/// See [Apple's documentation](https://developer.apple.com/documentation/foundation/nszone?language=objc).
16#[repr(C)]
17pub struct NSZone {
18    // Use `objc_object` to mark the types as !Send, !Sync and UnsafeCell.
19    //
20    // This works since `objc_object` is a ZST
21    _inner: ffi::objc_object,
22}
23
24impl fmt::Debug for NSZone {
25    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
26        write!(f, "<NSZone {self:p}>")
27    }
28}
29
30// Note: We don't know anything about the internals of `NSZone`, so best not
31// to make it `Send` and `Sync` for now.
32
33impl UnwindSafe for NSZone {}
34impl RefUnwindSafe for NSZone {}
35
36unsafe impl RefEncode for NSZone {
37    #[cfg(not(feature = "gnustep-1-7"))]
38    const ENCODING_REF: Encoding = Encoding::Pointer(&Encoding::Struct("_NSZone", &[]));
39    #[cfg(feature = "gnustep-1-7")]
40    const ENCODING_REF: Encoding = Encoding::Pointer(&Encoding::Struct(
41        "_NSZone",
42        &[
43            // Functions
44            Encoding::Pointer(&Encoding::Unknown),
45            Encoding::Pointer(&Encoding::Unknown),
46            Encoding::Pointer(&Encoding::Unknown),
47            Encoding::Pointer(&Encoding::Unknown),
48            Encoding::Pointer(&Encoding::Unknown),
49            Encoding::Pointer(&Encoding::Unknown),
50            // Stats
51            Encoding::Pointer(&Encoding::Unknown),
52            // Zone granularity
53            usize::ENCODING,
54            // Name of zone
55            Encoding::Object,
56            // Next zone - note that the contents of this doesn't matter,
57            // since this is nested far enough that the encoding string ends
58            // up ignoring it.
59            Encoding::Pointer(&Encoding::Struct("_NSZone", &[])),
60        ],
61    ));
62}
