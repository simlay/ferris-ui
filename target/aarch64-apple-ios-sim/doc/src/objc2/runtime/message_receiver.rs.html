8/// Wrap the given closure in `exception::catch` if the `catch-all` feature is
9/// enabled.
10///
11/// This is a macro to help with monomorphization when the feature is
12/// disabled, as well as improving the final stack trace (`#[track_caller]`
13/// doesn't really work on closures).
14#[cfg(not(feature = "catch-all"))]
15macro_rules! conditional_try {
16    (|| $expr:expr) => {
17        $expr
18    };
19}
20
21#[cfg(feature = "catch-all")]
22macro_rules! conditional_try {
23    (|| $expr:expr) => {{
24        let f = core::panic::AssertUnwindSafe(|| $expr);
25        match crate::exception::catch(f) {
26            Ok(r) => r,
27            Err(exception) => {
28                if let Some(exception) = exception {
29                    panic!("uncaught {exception:?}")
30                } else {
31                    panic!("uncaught exception nil")
32                }
33            }
34        }
35    }};
36}

70    #[cfg(target_arch = "arm")]
71    /// Double-word sized fundamental data types don't use stret, but any
72    /// composite type larger than 4 bytes does.
73    ///
74    /// <https://web.archive.org/web/20191016000656/http://infocenter.arm.com/help/topic/com.arm.doc.ihi0042f/IHI0042F_aapcs.pdf>
75    /// <https://developer.arm.com/documentation/ihi0042/latest>
76    /// <https://github.com/llvm/llvm-project/blob/llvmorg-17.0.6/clang/lib/CodeGen/Targets/ARM.cpp#L531>
77    unsafe impl<T: EncodeReturn> MsgSendFn for T {
78        const MSG_SEND: Imp = {
79            if let Encoding::LongLong | Encoding::ULongLong | Encoding::Double = T::ENCODING_RETURN
80            {
81                ffi::objc_msgSend
82            } else if mem::size_of::<T>() <= 4 {
83                ffi::objc_msgSend
84            } else {
85                ffi::objc_msgSend_stret
86            }
87        };
88        const MSG_SEND_SUPER: Imp = {
89            if let Encoding::LongLong | Encoding::ULongLong | Encoding::Double = T::ENCODING_RETURN
90            {
91                ffi::objc_msgSendSuper
92            } else if mem::size_of::<T>() <= 4 {
93                ffi::objc_msgSendSuper
94            } else {
95                ffi::objc_msgSendSuper_stret
96            }
97        };
98    }

100    #[cfg(target_arch = "x86")]
101    /// Structures 1 or 2 bytes in size are placed in EAX.
102    /// Structures 4 or 8 bytes in size are placed in: EAX and EDX.
103    /// Structures of other sizes are placed at the address supplied by the caller.
104    ///
105    /// <https://developer.apple.com/library/mac/documentation/DeveloperTools/Conceptual/LowLevelABI/130-IA-32_Function_Calling_Conventions/IA32.html>
106    /// <https://github.com/llvm/llvm-project/blob/llvmorg-17.0.6/clang/lib/CodeGen/Targets/X86.cpp#L472>
107    unsafe impl<T: EncodeReturn> MsgSendFn for T {
108        const MSG_SEND: Imp = {
109            // See https://github.com/apple-oss-distributions/objc4/blob/objc4-818.2/runtime/message.h#L156-L172
110            if let Encoding::Float | Encoding::Double | Encoding::LongDouble = T::ENCODING_RETURN {
111                ffi::objc_msgSend_fpret
112            } else if let 0 | 1 | 2 | 4 | 8 = mem::size_of::<T>() {
113                ffi::objc_msgSend
114            } else {
115                ffi::objc_msgSend_stret
116            }
117        };
118        const MSG_SEND_SUPER: Imp = {
119            if let 0 | 1 | 2 | 4 | 8 = mem::size_of::<T>() {
120                ffi::objc_msgSendSuper
121            } else {
122                ffi::objc_msgSendSuper_stret
123            }
124        };
125    }

127    #[cfg(target_arch = "x86_64")]
128    /// If the size of an object is larger than two eightbytes, it has class
129    /// MEMORY. If the type has class MEMORY, then the caller provides space for
130    /// the return value and passes the address of this storage.
131    ///
132    /// <https://www.uclibc.org/docs/psABI-x86_64.pdf>
133    /// <https://github.com/llvm/llvm-project/blob/llvmorg-17.0.6/clang/lib/CodeGen/Targets/X86.cpp#L2532>
134    unsafe impl<T: EncodeReturn> MsgSendFn for T {
135        const MSG_SEND: Imp = {
136            // See https://github.com/apple-oss-distributions/objc4/blob/objc4-818.2/runtime/message.h#L156-L172
137            if let Encoding::LongDouble = T::ENCODING_RETURN {
138                ffi::objc_msgSend_fpret
139            } else if let Encoding::LongDoubleComplex = T::ENCODING_RETURN {
140                ffi::objc_msgSend_fp2ret
141            } else if mem::size_of::<T>() <= 16 {
142                ffi::objc_msgSend
143            } else {
144                ffi::objc_msgSend_stret
145            }
146        };
147        const MSG_SEND_SUPER: Imp = {
148            if mem::size_of::<T>() <= 16 {
149                ffi::objc_msgSendSuper
150            } else {
151                ffi::objc_msgSendSuper_stret
152            }
153        };
154    }

156    #[inline]
157    #[track_caller]
158    pub(crate) unsafe fn send<A: EncodeArguments, R: EncodeReturn>(
159        receiver: *mut AnyObject,
160        sel: Sel,
161        args: A,
162    ) -> R {
163        let msg_send_fn = R::MSG_SEND;
164        // Note: Modern Objective-C compilers have a workaround to ensure that
165        // messages to `nil` with a struct return produces `mem::zeroed()`,
166        // see:
167        // <https://www.sealiesoftware.com/blog/archive/2012/2/29/objc_explain_return_value_of_message_to_nil.html>
168        //
169        // We _could_ technically do something similar, but since we're
170        // disallowing messages to `nil` with `debug_assertions` enabled
171        // anyhow, and since Rust has a much stronger type-system that
172        // disallows NULL/nil in most cases, we won't bother supporting it.
173        unsafe { A::__invoke(msg_send_fn, receiver, sel, args) }
174    }
175
176    #[inline]
177    #[track_caller]
178    pub(crate) unsafe fn send_super<A: EncodeArguments, R: EncodeReturn>(
179        receiver: *mut AnyObject,
180        superclass: &AnyClass,
181        sel: Sel,
182        args: A,
183    ) -> R {
184        let superclass: *const AnyClass = superclass;
185        let mut sup = ffi::objc_super {
186            receiver: receiver.cast(),
187            super_class: superclass.cast(),
188        };
189        let receiver: *mut ffi::objc_super = &mut sup;
190        let receiver = receiver.cast();
191
192        let msg_send_fn = R::MSG_SEND_SUPER;
193        unsafe { A::__invoke(msg_send_fn, receiver, sel, args) }
194    }
195}

205    #[inline]
206    fn unwrap_msg_send_fn(msg_send_fn: Option<Imp>) -> Imp {
207        match msg_send_fn {
208            Some(msg_send_fn) => msg_send_fn,
209            None => {
210                // SAFETY: This will never be NULL, even if the selector is not
211                // found a callable function pointer will still be returned!
212                //
213                // `clang` doesn't insert a NULL check here either.
214                unsafe { core::hint::unreachable_unchecked() }
215            }
216        }
217    }
218
219    #[track_caller]
220    pub(crate) unsafe fn send<A: EncodeArguments, R: EncodeReturn>(
221        receiver: *mut AnyObject,
222        sel: Sel,
223        args: A,
224    ) -> R {
225        // If `receiver` is NULL, objc_msg_lookup will return a standard
226        // C-method taking two arguments, the receiver and the selector.
227        //
228        // Transmuting and calling such a function with multiple parameters is
229        // safe as long as the return value is a primitive (and e.g. not a big
230        // struct or array).
231        //
232        // However, when the return value is a floating point value, the float
233        // will end up as some undefined value, usually NaN, which is
234        // incompatible with Apple's platforms. As such, we insert this extra
235        // NULL check here.
236        if receiver.is_null() {
237            // SAFETY: Caller guarantees that messages to NULL-receivers only
238            // return pointers or primitive values, and a mem::zeroed pointer
239            // / primitive is just a NULL-pointer or a zeroed primitive.
240            return unsafe { mem::zeroed() };
241        }
242
243        let msg_send_fn = unsafe { ffi::objc_msg_lookup(receiver.cast(), sel.as_ptr()) };
244        let msg_send_fn = unwrap_msg_send_fn(msg_send_fn);
245        unsafe { A::__invoke(msg_send_fn, receiver, sel, args) }
246    }
247
248    #[track_caller]
249    pub(crate) unsafe fn send_super<A: EncodeArguments, R: EncodeReturn>(
250        receiver: *mut AnyObject,
251        superclass: &AnyClass,
252        sel: Sel,
253        args: A,
254    ) -> R {
255        if receiver.is_null() {
256            // SAFETY: Same as in `send`.
257            return unsafe { mem::zeroed() };
258        }
259
260        let superclass: *const AnyClass = superclass;
261        let sup = ffi::objc_super {
262            receiver: receiver.cast(),
263            super_class: superclass.cast(),
264        };
265        let msg_send_fn = unsafe { ffi::objc_msg_lookup_super(&sup, sel.as_ptr()) };
266        let msg_send_fn = unwrap_msg_send_fn(msg_send_fn);
267        unsafe { A::__invoke(msg_send_fn, receiver, sel, args) }
268    }
269}

296/// Help with monomorphizing in framework crates
297#[cfg(debug_assertions)]
298#[track_caller]
299fn msg_send_check(
300    obj: Option<&AnyObject>,
301    sel: Sel,
302    args: &[crate::encode::Encoding],
303    ret: &crate::encode::Encoding,
304) {
305    let cls = if let Some(obj) = obj {
306        obj.class()
307    } else {
308        panic_null(sel)
309    };
310
311    msg_send_check_class(cls, sel, args, ret);
312}
313
314#[cfg(debug_assertions)]
315#[track_caller]
316fn msg_send_check_class(
317    cls: &AnyClass,
318    sel: Sel,
319    args: &[crate::encode::Encoding],
320    ret: &crate::encode::Encoding,
321) {
322    use crate::verify::{verify_method_signature, Inner, VerificationError};
323
324    let err = if let Some(method) = cls.instance_method(sel) {
325        if let Err(err) = verify_method_signature(method, args, ret) {
326            err
327        } else {
328            return;
329        }
330    } else {
331        VerificationError::from(Inner::MethodNotFound)
332    };
333
334    panic_verify(cls, sel, &err);
335}

337#[cfg(debug_assertions)]
338#[track_caller]
339fn panic_null(sel: Sel) -> ! {
340    panic!("messsaging {sel} to nil")
341}

343#[cfg(debug_assertions)]
344#[track_caller]
345fn panic_verify(cls: &AnyClass, sel: Sel, err: &crate::runtime::VerificationError) -> ! {
346    panic!(
347        "invalid message send to {}[{cls} {sel}]: {err}",
348        if cls.is_metaclass() { "+" } else { "-" },
349    )

356/// Types that can directly be used as the receiver of Objective-C messages.
357///
358/// Examples include objects pointers, class pointers, and block pointers.
359///
360///
361/// # Safety
362///
363/// This is a sealed trait, and should not need to be implemented. Open an
364/// issue if you know a use-case where this restrition should be lifted!
365pub unsafe trait MessageReceiver: private::Sealed + Sized {
366    #[doc(hidden)]
367    type __Inner: ?Sized + RefEncode;
368
369    #[doc(hidden)]
370    fn __as_raw_receiver(self) -> *mut AnyObject;
371
372    /// Sends a message to the receiver with the given selector and arguments.
373    ///
374    /// This should be used instead of the [`performSelector:`] family of
375    /// methods, as this is both more performant and flexible than that.
376    ///
377    /// The correct version of `objc_msgSend` will be chosen based on the
378    /// return type. For more information, see [the Messaging section in
379    /// Apple's Objective-C Runtime Programming Guide][guide-messaging].
380    ///
381    /// If the selector is known at compile-time, it is recommended to use the
382    /// [`msg_send!`] macro rather than this method.
383    ///
384    /// [`performSelector:`]: https://developer.apple.com/documentation/objectivec/1418956-nsobject/1418867-performselector?language=objc
385    /// [guide-messaging]: https://developer.apple.com/library/archive/documentation/Cocoa/Conceptual/ObjCRuntimeGuide/Articles/ocrtHowMessagingWorks.html
386    ///
387    ///
388    /// # Safety
389    ///
390    /// This shares the same safety requirements as [`msg_send!`].
391    ///
392    /// The added invariant is that the selector must take the same number of
393    /// arguments as is given.
394    ///
395    /// [`msg_send!`]: crate::msg_send
396    ///
397    ///
398    /// # Example
399    ///
400    /// Call the `copy` method, but using a dynamic selector instead.
401    ///
402    /// ```no_run
403    /// use objc2::rc::Retained;
404    /// use objc2::runtime::MessageReceiver;
405    /// use objc2::sel;
406    /// # use objc2::runtime::NSObject as MyObject;
407    ///
408    /// let obj = MyObject::new();
409    /// // SAFETY: The `copy` method takes no arguments, and returns an object
410    /// let copy: *mut MyObject = unsafe { obj.send_message(sel!(copy), ()) };
411    /// // SAFETY: The `copy` method returns an object with +1 retain count
412    /// let copy = unsafe { Retained::from_raw(copy) }.unwrap();
413    /// ```
414    #[inline]
415    #[track_caller]
416    #[doc(alias = "performSelector")]
417    #[doc(alias = "performSelector:")]
418    #[doc(alias = "performSelector:withObject:")]
419    #[doc(alias = "performSelector:withObject:withObject:")]
420    unsafe fn send_message<A: EncodeArguments, R: EncodeReturn>(self, sel: Sel, args: A) -> R {
421        let receiver = self.__as_raw_receiver();
422        #[cfg(debug_assertions)]
423        {
424            // SAFETY: Caller ensures only valid or NULL pointers.
425            let obj = unsafe { receiver.as_ref() };
426            msg_send_check(obj, sel, A::ENCODINGS, &R::ENCODING_RETURN);
427        }
428
429        // SAFETY: Upheld by caller
430        //
431        // The @catch is safe since message sending primitives are guaranteed
432        // to do Objective-C compatible unwinding.
433        unsafe { conditional_try!(|| msg_send_primitive::send(receiver, sel, args)) }
434    }
435
436    /// Sends a message to a specific superclass with the given selector and
437    /// arguments.
438    ///
439    /// The correct version of `objc_msgSend_super` will be chosen based on the
440    /// return type. For more information, see the section on "Sending
441    /// Messages" in Apple's [documentation][runtime].
442    ///
443    /// If the selector is known at compile-time, it is recommended to use the
444    /// [`msg_send!(super(...), ...)`] macro rather than this method.
445    ///
446    /// [runtime]: https://developer.apple.com/documentation/objectivec/objective-c_runtime?language=objc
447    ///
448    ///
449    /// # Safety
450    ///
451    /// This shares the same safety requirements as
452    /// [`msg_send!(super(...), ...)`].
453    ///
454    /// The added invariant is that the selector must take the same number of
455    /// arguments as is given.
456    ///
457    /// [`msg_send!(super(...), ...)`]: crate::msg_send
458    #[inline]
459    #[track_caller]
460    unsafe fn send_super_message<A: EncodeArguments, R: EncodeReturn>(
461        self,
462        superclass: &AnyClass,
463        sel: Sel,
464        args: A,
465    ) -> R {
466        let receiver = self.__as_raw_receiver();
467        #[cfg(debug_assertions)]
468        {
469            if receiver.is_null() {
470                panic_null(sel);
471            }
472            msg_send_check_class(superclass, sel, A::ENCODINGS, &R::ENCODING_RETURN);
473        }
474
475        // SAFETY: Same as in `send_message`
476        unsafe {
477            conditional_try!(|| msg_send_primitive::send_super(receiver, superclass, sel, args))
478        }
479    }
480}
