17/// The number of UTF-8 code units in the given string.
18///
19/// # Safety
20///
21/// The object must be an instance of `NSString`.
22//
23// Note: While this is not public, it is still a breaking change to modify,
24// since `objc2-foundation` relies on it.
25#[inline]
26pub unsafe fn nsstring_len(obj: &NSObject) -> NSUInteger {
27    unsafe { msg_send![obj, lengthOfBytesUsingEncoding: UTF8_ENCODING] }
28}

30/// Extract a [`str`](`prim@str`) representation out of the given NSString.
31///
32/// Uses [`UTF8String`] under the hood.
33///
34/// [`UTF8String`]: https://developer.apple.com/documentation/foundation/nsstring/1411189-utf8string?language=objc
35///
36///
37/// # Safety
38///
39/// - The object must be an instance of `NSString`.
40/// - The returned string must not be moved outside the autorelease pool into
41///   which it (may) have been released.
42///
43/// Furthermore, the object must not, as is always the case for strings, be
44/// mutated in parallel.
45//
46// Note: While this is not public, it is still a breaking change to modify,
47// since `objc2-foundation` relies on it.
48pub unsafe fn nsstring_to_str<'r, 's: 'r, 'p: 'r>(
49    obj: &'s NSObject,
50    pool: AutoreleasePool<'p>,
51) -> &'r str {
52    // This is necessary until `auto` types stabilizes.
53    pool.__verify_is_inner();
54
55    // The documentation on `UTF8String` is quite sparse, but with educated
56    // guesses, testing, reading the code for `CFString` and a bit of
57    // reverse-engineering, we've determined that `NSString` stores a pointer
58    // to the string data, sometimes with an UTF-8 encoding (usual for ascii
59    // data), sometimes in other encodings (often UTF-16).
60    //
61    // `UTF8String` then checks the internal encoding:
62    // - If the data is UTF-8 encoded, and (since macOS 10.6) if the string is
63    //   immutable, it returns the internal pointer using
64    //   `CFStringGetCStringPtr`.
65    // - Otherwise, if the data is in another encoding or is mutable, it
66    //   creates a new allocation, writes the UTF-8 representation of the
67    //   string into it, autoreleases the allocation, and returns a pointer to
68    //   it (similar to `CFStringGetCString`).
69    //
70    // If the string is a tagged pointer, or a custom subclass, then another
71    // code-path is taken that always creates a new allocation and copies the
72    // string into that using (effectively) `length` and `characterAtIndex:`.
73    //
74    // As a result, the lifetime of the returned pointer is either the same as
75    // the passed-in `NSString` OR the lifetime of the current / innermost
76    // `@autoreleasepool`.
77    //
78    // Furthermore, we can allow creating a `&str` from `&obj`, even if the
79    // string is originally a `NSMutableString` which may be mutated later on,
80    // since in that case the lifetime will be tied to the pool and not the
81    // string.
82    let bytes: *const c_char = unsafe { msg_send![obj, UTF8String] };
83    let bytes: *const u8 = bytes.cast();
84
85    // SAFETY: Caller ensures that the object is an instance of `NSString`.
86    let len = unsafe { nsstring_len(obj) };
87
88    // SAFETY:
89    // The held AutoreleasePool is the innermost, and the reference is
90    // constrained both by the pool and the NSString.
91    //
92    // `len` is the length of the string in the UTF-8 encoding.
93    //
94    // `bytes` is a null-terminated C string (with length = len + 1), so
95    // it is never a NULL pointer.
96    let bytes: &'r [u8] = unsafe { slice::from_raw_parts(bytes, len) };
97
98    // SAFETY: The bytes are valid UTF-8.
99    #[cfg(not(debug_assertions))]
100    unsafe {
101        str::from_utf8_unchecked(bytes)
102    }
103
104    #[cfg(debug_assertions)]
105    {
106        str::from_utf8(bytes).expect("invalid UTF-8 in NSString")
107    }
108}
