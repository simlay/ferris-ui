13pub const UTF8_ENCODING: usize = 4;

17/// The number of UTF-8 code units in the given string.
18///
19/// # Safety
20///
21/// The object must be an instance of `NSString`.
22//
23// Note: While this is not public, it is still a breaking change to modify,
24// since `objc2-foundation` relies on it.
25pub unsafe fn nsstring_len(obj: &NSObject) -> NSUInteger {
26    unsafe { msg_send![obj, lengthOfBytesUsingEncoding: UTF8_ENCODING] }
27}

29/// Extract a [`str`](`prim@str`) representation out of the given NSString.
30///
31/// # Safety
32///
33/// The object must be an instance of `NSString`.
34//
35// Note: While this is not public, it is still a breaking change to modify,
36// since `objc2-foundation` relies on it.
37pub unsafe fn nsstring_to_str<'r, 's: 'r, 'p: 'r>(
38    obj: &'s NSObject,
39    pool: AutoreleasePool<'p>,
40) -> &'r str {
41    // This is necessary until `auto` types stabilizes.
42    pool.__verify_is_inner();
43
44    // The documentation on `UTF8String` is a bit sparse, but with
45    // educated guesses and testing I've determined that NSString stores
46    // a pointer to the string data, sometimes with an UTF-8 encoding,
47    // (usual for ascii data), sometimes in other encodings (UTF-16?).
48    //
49    // `UTF8String` then checks the internal encoding:
50    // - If the data is UTF-8 encoded, it returns the internal pointer.
51    // - If the data is in another encoding, it creates a new allocation,
52    //   writes the UTF-8 representation of the string into it,
53    //   autoreleases the allocation and returns a pointer to it.
54    //
55    // So the lifetime of the returned pointer is either the same as the
56    // NSString OR the lifetime of the innermost @autoreleasepool.
57    //
58    // https://developer.apple.com/documentation/foundation/nsstring/1411189-utf8string?language=objc
59    let bytes: *const c_char = unsafe { msg_send![obj, UTF8String] };
60    let bytes: *const u8 = bytes.cast();
61
62    // SAFETY: Caller ensures that the object is an instance of `NSString`.
63    let len = unsafe { nsstring_len(obj) };
64
65    // SAFETY:
66    // The held AutoreleasePool is the innermost, and the reference is
67    // constrained both by the pool and the NSString.
68    //
69    // `len` is the length of the string in the UTF-8 encoding.
70    //
71    // `bytes` is a null-terminated C string (with length = len + 1), so
72    // it is never a NULL pointer.
73    let bytes: &'r [u8] = unsafe { slice::from_raw_parts(bytes, len) };
74
75    // TODO: Always UTF-8, so should we use `from_utf8_unchecked`?
76    str::from_utf8(bytes).unwrap()
77}
