1//! A minimal alternative to crates like `malloc_buf`, `mbox` and `malloced`.
2use core::ffi::c_char;
3use core::ffi::CStr;
4use core::fmt;
5use core::marker::PhantomData;
6use core::ops::Deref;
7use core::ptr::{self, NonNull};
8
9use crate::ffi;
10
11#[repr(transparent)]
12pub(crate) struct MallocSlice<T> {
13    ptr: NonNull<[T]>,
14    // Necessary for dropck
15    _p: PhantomData<[T]>,
16}
17
18impl<T> MallocSlice<T> {
19    pub(crate) unsafe fn from_array(mut ptr: *mut T, len: usize) -> Self {
20        // If the length is 0, the pointer is usually NULL, and as such we
21        // need to conjure some other pointer (slices are always non-null).
22        if len == 0 {
23            ptr = NonNull::dangling().as_ptr();
24        }
25
26        let ptr = ptr::slice_from_raw_parts_mut(ptr, len);
27        let ptr = NonNull::new(ptr).expect("tried to construct MallocSlice from a NULL pointer");
28        Self {
29            ptr,
30            _p: PhantomData,
31        }
32    }
33
34    fn len(&self) -> usize {
35        // TODO: Use `self.ptr.len()` once in MSRV
36        (**self).len()
37    }
38}
39
40impl<T> Drop for MallocSlice<T> {
41    #[allow(clippy::len_zero)]
42    fn drop(&mut self) {
43        // If the length is 0, then the pointer is dangling from `from_array`
44        // (since the length is immutable), and we can skip calling `free`.
45        if self.len() != 0 {
46            // SAFETY: We take ownership over the slice elements in
47            // `from_array`.
48            unsafe { ptr::drop_in_place(self.ptr.as_ptr()) };
49            // SAFETY: We take ownership over the pointer in `from_array`,
50            // and the pointer is valid if the length is non-zero.
51            unsafe { ffi::free(self.ptr.cast().as_ptr()) };
52        }
53    }
54}
55
56impl<T> Deref for MallocSlice<T> {
57    type Target = [T];
58
59    #[inline]
60    fn deref(&self) -> &[T] {
61        // SAFETY:
62        // - That the pointer is aligned, dereferenceable and initialized is
63        //   ensured by the caller of `from_array` (which usually get it from
64        //   some external API that will do this for you).
65        // - The lifetime is bound to the `MallocSlice`, which in turn ensures
66        //   the pointer is valid until it is dropped.
67        unsafe { self.ptr.as_ref() }
68    }
69}
70
71impl<T: fmt::Debug> fmt::Debug for MallocSlice<T> {
72    #[inline]
73    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
74        fmt::Debug::fmt(&**self, f)
75    }
76}
77
78impl<T> AsRef<[T]> for MallocSlice<T> {
79    #[inline]
80    fn as_ref(&self) -> &[T] {
81        self
82    }
83}

85#[repr(transparent)]
86pub(crate) struct MallocCStr {
87    ptr: NonNull<CStr>,
88}
89
90impl MallocCStr {
91    pub(crate) unsafe fn from_c_str(ptr: *mut c_char) -> Self {
92        if ptr.is_null() {
93            panic!("tried to construct MallocStr from a NULL pointer");
94        }
95        // SAFETY: We just checked that the pointer is not NULL.
96        //
97        // Further validity of the pointer is ensured by the caller.
98        let cstr = unsafe { CStr::from_ptr(ptr) };
99        // Note that we construct this `NonNull` from an immutable reference
100        // (there is not yet a `CStr::from_mut_ptr`).
101        //
102        // This means that we're (probably) no longer allowed to mutate the
103        // value, if that is desired for `MallocStr` in the future, then we'll
104        // have to implement this method a bit differently.
105        let ptr = NonNull::from(cstr);
106        Self { ptr }
107    }
108}
109
110impl Drop for MallocCStr {
111    #[inline]
112    fn drop(&mut self) {
113        // SAFETY: We take ownership in `from_c_str`.
114        unsafe { ffi::free(self.ptr.cast().as_ptr()) };
115    }
116}
117
118impl Deref for MallocCStr {
119    type Target = CStr;
120
121    #[inline]
122    fn deref(&self) -> &CStr {
123        // SAFETY: Same as `MallocSlice::deref`
124        unsafe { self.ptr.as_ref() }
125    }
126}
127
128impl fmt::Debug for MallocCStr {
129    #[inline]
130    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
131        fmt::Debug::fmt(&**self, f)
132    }
133}
134
135impl AsRef<CStr> for MallocCStr {
136    #[inline]
137    fn as_ref(&self) -> &CStr {
138        self
139    }
