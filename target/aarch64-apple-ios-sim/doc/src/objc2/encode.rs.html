1//! # Support for type-encodings.
2//!
3//! This module contains traits for annotating types that has an Objective-C
4//! type-encoding: Specifically [`Encode`] for structs/numeric types and
5//! [`RefEncode`] for references.
6//!
7//! Additionally, this exports the [`Encoding`] and [`EncodingBox`] types from
8//! [`objc2-encode`][objc2_encode], see that crate for a few more details on
9//! what Objective-C type-encodings are.
10//!
11//!
12//! ## Examples
13//!
14//! Implementing [`Encode`] and [`RefEncode`] for a custom type:
15//!
16//! ```
17//! use objc2::encode::{Encode, Encoding, RefEncode};
18//!
19//! #[repr(C)]
20//! struct MyStruct {
21//!     a: f32, // float
22//!     b: i16, // int16_t
23//! }
24//!
25//! unsafe impl Encode for MyStruct {
26//!     const ENCODING: Encoding = Encoding::Struct(
27//!         "MyStruct", // Must use the same name as defined in C header files
28//!         &[
29//!             f32::ENCODING, // Same as Encoding::Float
30//!             i16::ENCODING, // Same as Encoding::Short
31//!         ],
32//!     );
33//! }
34//!
35//! // @encode(MyStruct) -> "{MyStruct=fs}"
36//! assert!(MyStruct::ENCODING.equivalent_to_str("{MyStruct=fs}"));
37//!
38//! unsafe impl RefEncode for MyStruct {
39//!     const ENCODING_REF: Encoding = Encoding::Pointer(&Self::ENCODING);
40//! }
41//!
42//! // @encode(MyStruct*) -> "^{MyStruct=fs}"
43//! assert!(MyStruct::ENCODING_REF.equivalent_to_str("^{MyStruct=fs}"));
44//! ```
45//!
46//! Implementing [`Encode`] for a few core-graphics types.
47//!
48//! Note that these are available in `objc2-foundation`, so the implementation
49//! here is mostly for demonstration.
50//!
51//! ```
52#![doc = include_str!("../examples/encode_core_graphics.rs")]
53//! ```
54//!
55//! Implementing [`Encode`] and [`RefEncode`] for a transparent newtype.
56//!
57//! ```
58#![doc = include_str!("../examples/encode_nsuinteger.rs")]
59//! ```
60//!
61//! Implementing [`RefEncode`] for an object, in this case `NSString`.
62//!
63//! ```
64#![doc = include_str!("../examples/encode_nsstring.rs")]
65//! ```
66//!
67//! Implementing [`RefEncode`] for a type where you don't necessarily know
68//! about the exact internals / the internals are not representable in Rust.
69//!
70//! ```
71#![doc = include_str!("../examples/encode_opaque_type.rs")]
72//! ```

89/// Types that have an Objective-C type-encoding.
90///
91/// Usually you will want to implement [`RefEncode`] as well.
92///
93/// If your type is an opaque type you should not need to implement this;
94/// there you will only need [`RefEncode`].
95///
96///
97/// # Safety
98///
99/// The type must be FFI-safe, meaning a C-compatible `repr` (`repr(C)`,
100/// `repr(u8)`, `repr(transparent)` where the inner types are C-compatible,
101/// and so on). See the [nomicon on other `repr`s][reprs].
102///
103/// Objective-C will make assumptions about the type (like its size, alignment
104/// and ABI) from its encoding, so the implementer must verify that the
105/// encoding is accurate.
106///
107/// Concretely, [`Self::ENCODING`] must match the result of running `@encode`
108/// in Objective-C with the type in question.
109///
110/// You should also beware of having [`Drop`] types implement this, since when
111/// passed to Objective-C via. `objc2::msg_send!` their destructor will not be
112/// called!
113///
114///
115/// # Examples
116///
117/// Implementing for a struct:
118///
119/// ```
120/// # use objc2::encode::{Encode, Encoding, RefEncode};
121/// # use core::ffi::c_void;
122/// #
123/// #[repr(C)]
124/// struct MyType {
125///     a: i32,
126///     b: f64,
127///     c: *const c_void,
128/// }
129///
130/// unsafe impl Encode for MyType {
131///     const ENCODING: Encoding = Encoding::Struct(
132///         // The name of the type that Objective-C sees.
133///         "MyType",
134///         &[
135///             // Delegate to field's implementations.
136///             // The order is the same as in the definition.
137///             i32::ENCODING,
138///             f64::ENCODING,
139///             <*const c_void>::ENCODING,
140///         ],
141///     );
142/// }
143///
144/// // Note: You would also implement `RefEncode` for this type.
145/// ```
146///
147/// [reprs]: https://doc.rust-lang.org/nomicon/other-reprs.html
148pub unsafe trait Encode {
149    /// The Objective-C type-encoding for this type.
150    const ENCODING: Encoding;
151}

153/// Types whoose references has an Objective-C type-encoding.
154///
155/// Implementing this for `T` provides [`Encode`] implementations for:
156/// - `*const T`
157/// - `*mut T`
158/// - `&T`
159/// - `&mut T`
160/// - `NonNull<T>`
161/// - `Option<&T>`
162/// - `Option<&mut T>`
163/// - `Option<NonNull<T>>`
164///
165///
166/// # Reasoning behind this trait's existence
167///
168/// External crates cannot implement [`Encode`] for pointers or [`Option`]s
169/// containing references, so instead, they can implement this trait.
170/// Additionally it would be very cumbersome if every type had to implement
171/// [`Encode`] for all possible pointer types.
172///
173/// Finally, having this trait allows for much cleaner generic code that need
174/// to represent types that can be encoded as pointers.
175///
176///
177/// # Safety
178///
179/// References to the object must be FFI-safe.
180///
181/// See the nomicon entry on [representing opaque structs][opaque] for
182/// information on how to represent objects that you don't know the layout of
183/// (or use `extern type` ([RFC-1861]) if you're using nightly).
184///
185/// Objective-C will make assumptions about the type (like its size, alignment
186/// and ABI) from its encoding, so the implementer must verify that the
187/// encoding is accurate.
188///
189/// Concretely, [`Self::ENCODING_REF`] must match the result of running
190/// `@encode` in Objective-C with a pointer to the type in question.
191///
192/// [opaque]: https://doc.rust-lang.org/nomicon/ffi.html#representing-opaque-structs
193/// [RFC-1861]: https://rust-lang.github.io/rfcs/1861-extern-types.html
194pub unsafe trait RefEncode {
195    /// The Objective-C type-encoding for a reference of this type.
196    ///
197    /// Should be one of [`Encoding::Object`], [`Encoding::Block`],
198    /// [`Encoding::Class`], [`Encoding::Pointer`], [`Encoding::Sel`] or
199    /// [`Encoding::Unknown`].
200    ///
201    ///
202    /// # Examples
203    ///
204    /// This is usually implemented either as an object pointer:
205    /// ```
206    /// # use objc2::encode::{Encoding, RefEncode};
207    /// # #[repr(C)]
208    /// # struct MyObject {
209    /// #     _priv: [u8; 0],
210    /// # }
211    /// # unsafe impl RefEncode for MyObject {
212    /// const ENCODING_REF: Encoding = Encoding::Object;
213    /// # }
214    /// ```
215    ///
216    /// Or as a pointer to the type, delegating the rest to the [`Encode`]
217    /// implementation:
218    /// ```
219    /// # use objc2::encode::{Encode, Encoding, RefEncode};
220    /// # #[repr(transparent)]
221    /// # struct MyType(i32);
222    /// # unsafe impl Encode for MyType {
223    /// #     const ENCODING: Encoding = i32::ENCODING;
224    /// # }
225    /// # unsafe impl RefEncode for MyType {
226    /// const ENCODING_REF: Encoding = Encoding::Pointer(&Self::ENCODING);
227    /// # }
228    /// ```
229    const ENCODING_REF: Encoding;
230}

232/// A helper trait for types that follow the "null pointer optimization", and
233/// are encodable inside an [`Option`].
234///
235/// See [the `Option` documentation][option-repr] for details on which types
236/// this holds for, and [the nomicon][nomicon-npo] for more details on the
237/// null pointer optimization.
238///
239/// This trait used to work around the orphan rule, which would normally
240/// prevent you from implementing [`Encode`]/[`RefEncode`] for
241/// `Option<CustomType>`.
242///
243/// [option-repr]: https://doc.rust-lang.org/1.75.0/std/option/index.html#representation
244/// [nomicon-npo]: https://doc.rust-lang.org/nightly/nomicon/ffi.html#the-nullable-pointer-optimization
245///
246///
247/// # Safety
248///
249/// You must ensure that the implemented type `T` has the same layout and
250/// function call ABI as `Option<T>`.
251///
252///
253/// # Examples
254///
255/// ```
256/// use objc2::encode::{Encode, Encoding, OptionEncode};
257/// use core::ptr::NonNull;
258/// use core::ffi::c_void;
259///
260/// #[repr(transparent)]
261/// struct MyBlockType(NonNull<c_void>);
262///
263/// // SAFETY: `MyBlockType` is meant to represent a pointer to a block
264/// unsafe impl Encode for MyBlockType {
265///     const ENCODING: Encoding = Encoding::Block;
266/// }
267///
268/// // SAFETY: `MyBlockType` is `repr(transparent)` over `NonNull`, which
269/// // means that `Option<MyBlockType>` has the same layout.
270/// unsafe impl OptionEncode for MyBlockType {}
271///
272/// assert_eq!(<Option<MyBlockType>>::ENCODING, MyBlockType::ENCODING);
273/// ```
274pub unsafe trait OptionEncode {}
275
276// SAFETY: Implementor of `OptionEncode` guarantees this impl is sound
277unsafe impl<T: Encode + OptionEncode> Encode for Option<T> {
278    const ENCODING: Encoding = {
279        if mem::size_of::<T>() != mem::size_of::<Option<T>>() {
280            panic!("invalid OptionEncode + Encode implementation");
281        }
282        T::ENCODING
283    };
284}
285
286// SAFETY: Implementor of `OptionEncode` guarantees this impl is sound
287unsafe impl<T: RefEncode + OptionEncode> RefEncode for Option<T> {
288    const ENCODING_REF: Encoding = {
289        if mem::size_of::<T>() != mem::size_of::<Option<T>>() {
290            panic!("invalid OptionEncode + RefEncode implementation");
291        }
292        T::ENCODING_REF
293    };
294}

300/// Types that are safe as the return value from Objective-C.
301///
302/// This is a sealed trait, and should not need to be implemented manually.
303///
304///
305/// # Safety
306///
307/// Similar to [`Encode`], except the value is only guaranteed to be valid as
308/// a return value, both from functions/methods you're calling, and from
309/// declared functions/methods.
310///
311/// It does not have to be valid as e.g. an instance variable, or as an
312/// argument to a function.
313pub unsafe trait EncodeReturn: return_private::Sealed {
314    /// The Objective-C type-encoding for this type.
315    const ENCODING_RETURN: Encoding;
316}
317
318impl return_private::Sealed for () {}
319// SAFETY: `()` is the same as C's `void` type, which is a valid return type
320unsafe impl EncodeReturn for () {
321    const ENCODING_RETURN: Encoding = Encoding::Void;
322}
323
324impl<T: Encode> return_private::Sealed for T {}
325// SAFETY: All `Encode` types are also valid as return types
326unsafe impl<T: Encode> EncodeReturn for T {
327    const ENCODING_RETURN: Encoding = T::ENCODING;
328}

334/// Types that are safe as arguments to Objective-C methods.
335///
336/// This is a sealed trait, and should not need to be implemented manually.
337///
338///
339/// # Safety
340///
341/// Similar to [`Encode`], except the value is only guaranteed to be valid as
342/// an argument or a parameter, both from functions/methods you're calling and
343/// from declared functions/methods.
344///
345/// It does not have to be valid as e.g. an instance variable, or as an
346/// argument to a function.
347//
348// Note: This is mostly implemented for consistency; there are (not that I've
349// found at least) no places where this is not just `Encode`.
350//
351// You might be tempted to think that `bool` could work in this, but that
352// would be a mistake (even ignoring that its size is different on certain
353// targets) because it cannot be safely used in declared methods.
354pub unsafe trait EncodeArgument: argument_private::Sealed {
355    /// The Objective-C type-encoding for this type.
356    const ENCODING_ARGUMENT: Encoding;
357}
358
359impl<T: Encode> argument_private::Sealed for T {}
360// SAFETY: All `Encode` types are also valid as argument types
361unsafe impl<T: Encode> EncodeArgument for T {
362    const ENCODING_ARGUMENT: Encoding = T::ENCODING;
363}

369/// Types that represent an ordered group of function arguments, where each
370/// argument has an Objective-C type-encoding, or can be converted from one.
371///
372/// This is implemented for tuples of up to 16 arguments, where each argument
373/// implements [`EncodeArgument`]. It is a sealed trait, and should not need
374/// to be implemented manually - it is primarily used to make generic code a
375/// bit easier to read and understand.
376///
377/// Note that tuples themselves don't implement [`Encode`] directly, because
378/// they're not FFI-safe!
379pub trait EncodeArguments: args_private::Sealed {
380    /// The encodings for the arguments.
381    const ENCODINGS: &'static [Encoding];
382
383    /// Invoke a message sending function with the given object, selector,
384    /// and arguments.
385    ///
386    /// Implementation-wise, this is a bit ugly, but simply just easiest to
387    /// have the method on this trait, since inside `MessageReceiver` we only
388    /// want to publicly require `EncodeArguments`, and not another private
389    /// trait.
390    #[doc(hidden)]
391    unsafe fn __invoke<R: EncodeReturn>(
392        msg_send_fn: Imp,
393        receiver: *mut AnyObject,
394        sel: Sel,
395        args: Self,
396    ) -> R;
397}
398
399macro_rules! encode_args_impl {
400    ($($a:ident: $T: ident),*) => {
401        impl<$($T: EncodeArgument),*> args_private::Sealed for ($($T,)*) {}
402
403        impl<$($T: EncodeArgument),*> EncodeArguments for ($($T,)*) {
404            const ENCODINGS: &'static [Encoding] = &[
405                $($T::ENCODING_ARGUMENT),*
406            ];
407
408            #[inline]
409            unsafe fn __invoke<R: EncodeReturn>(msg_send_fn: Imp, receiver: *mut AnyObject, sel: Sel, ($($a,)*): Self) -> R {
410                // Message sending works by passing the receiver as the first
411                // argument, the selector as the second argument, and the rest
412                // of the arguments after that.
413                //
414                // The imp must be cast to the appropriate function pointer
415                // type before being called; contrary to how the headers and
416                // documentation describe them, the msgSend functions are not
417                // parametric on all platforms, instead they "trampoline" to
418                // the actual method implementations.
419                //
420                // SAFETY: We're transmuting an `unsafe` function pointer to
421                // another `unsafe` function pointer.
422                #[cfg(not(feature = "unstable-c-unwind"))]
423                let msg_send_fn: unsafe extern "C" fn(*mut AnyObject, Sel $(, $T)*) -> R = unsafe {
424                    mem::transmute(msg_send_fn)
425                };
426                #[cfg(feature = "unstable-c-unwind")]
427                let msg_send_fn: unsafe extern "C-unwind" fn(*mut AnyObject, Sel $(, $T)*) -> R = unsafe {
428                    mem::transmute(msg_send_fn)
429                };
430
431                // SAFETY: Caller upholds that the imp is safe to call with
432                // the given receiver, selector and arguments.
433                //
434                // TODO: On x86_64 it would be more efficient to use a GOT
435                // entry here (e.g. adding `nonlazybind` in LLVM).
436                // Same can be said of e.g. `objc_retain` and `objc_release`.
437                unsafe { msg_send_fn(receiver, sel $(, $a)*) }
438            }
439        }
440    };
441}

772/// Helper for implementing `Encode`/`RefEncode` for pointers to types that
773/// implement `RefEncode`.
774///
775/// Using `?Sized` is safe here because we delegate to other implementations
776/// (which will verify that the implementation is safe for the unsized type).
777macro_rules! encode_pointer_impls {
778    (unsafe impl<T: RefEncode> $x:ident for Pointer<T> {
779        const $c:ident = $e:expr;
780    }) => (
781        unsafe impl<T: RefEncode + ?Sized> $x for *const T {
782            const $c: Encoding = $e;
783        }
784
785        unsafe impl<T: RefEncode + ?Sized> $x for *mut T {
786            const $c: Encoding = $e;
787        }
788
789        unsafe impl<'a, T: RefEncode + ?Sized> $x for &'a T {
790            const $c: Encoding = $e;
791        }
792
793        unsafe impl<'a, T: RefEncode + ?Sized> $x for &'a mut T {
794            const $c: Encoding = $e;
795        }
796
797        unsafe impl<T: RefEncode + ?Sized> $x for NonNull<T> {
798            const $c: Encoding = $e;
799        }
800    );
801}

803// Implement `Encode` for types that are `RefEncode`.
804//
805// This allows users to implement `Encode` for custom types that have a
806// specific encoding as a pointer, instead of having to implement it for each
807// pointer-like type in turn.
808encode_pointer_impls!(
809    unsafe impl<T: RefEncode> Encode for Pointer<T> {
810        const ENCODING = T::ENCODING_REF;
811    }
812);

814// Implement `RefEncode` for pointers to types that are `RefEncode`.
815//
816// This implements `Encode` for pointers to pointers (to pointers, and so on),
817// which would otherwise be very cumbersome to do manually.
818encode_pointer_impls!(
819    unsafe impl<T: RefEncode> RefEncode for Pointer<T> {
820        const ENCODING_REF = Encoding::Pointer(&T::ENCODING_REF);
821    }

825unsafe impl<'a, T: RefEncode + ?Sized> OptionEncode for &'a T {}
826unsafe impl<'a, T: RefEncode + ?Sized> OptionEncode for &'a mut T {}
827unsafe impl<T: RefEncode + ?Sized> OptionEncode for NonNull<T> {}

829/// Helper for implementing [`Encode`]/[`RefEncode`] for function pointers
830/// whoose arguments implement [`Encode`].
831///
832/// Ideally we'd implement it for all function pointers, but due to coherence
833/// issues, see <https://github.com/rust-lang/rust/issues/56105>, function
834/// pointers that are higher-ranked over lifetimes don't get implemented.
835///
836/// We could fix it by adding those impls and allowing `coherence_leak_check`,
837/// but it would have to be done for _all_ references, `Option<&T>` and such
838/// as well. So trying to do it quickly requires generating a polynomial
839/// amount of implementations, which IMO is overkill for such a small issue.
840///
841/// Using `?Sized` is probably not safe here because C functions can only take
842/// and return items with a known size.
843macro_rules! encode_fn_pointer_impl {
844    (@ $FnTy: ty, $($Arg: ident),*) => {
845        unsafe impl<Ret: EncodeReturn, $($Arg: EncodeArgument),*> Encode for $FnTy {
846            const ENCODING: Encoding = Encoding::Pointer(&Encoding::Unknown);
847        }
848        unsafe impl<Ret: EncodeReturn, $($Arg: EncodeArgument),*> RefEncode for $FnTy {
849            const ENCODING_REF: Encoding = Encoding::Pointer(&Self::ENCODING);
850        }
851        // SAFETY: Function pointers have a NULL niche
852        unsafe impl<Ret: EncodeReturn, $($Arg: EncodeArgument),*> OptionEncode for $FnTy {}
853    };
854    (# $abi:literal; $($Arg: ident),+) => {
855        // Normal functions
856        encode_fn_pointer_impl!(@ extern $abi fn($($Arg),+) -> Ret, $($Arg),+ );
857        encode_fn_pointer_impl!(@ unsafe extern $abi fn($($Arg),+) -> Ret, $($Arg),+ );
858        // Variadic functions
859        encode_fn_pointer_impl!(@ extern $abi fn($($Arg),+ , ...) -> Ret, $($Arg),+ );
860        encode_fn_pointer_impl!(@ unsafe extern $abi fn($($Arg),+ , ...) -> Ret, $($Arg),+ );
861    };
862    (# $abi:literal; ) => {
863        // No variadic functions with 0 parameters
864        encode_fn_pointer_impl!(@ extern $abi fn() -> Ret, );
865        encode_fn_pointer_impl!(@ unsafe extern $abi fn() -> Ret, );
866    };
867    ($($Arg: ident),*) => {
868        encode_fn_pointer_impl!(# "C"; $($Arg),*);
869        #[cfg(feature = "unstable-c-unwind")]
870        encode_fn_pointer_impl!(# "C-unwind"; $($Arg),*);
871    };
872}
