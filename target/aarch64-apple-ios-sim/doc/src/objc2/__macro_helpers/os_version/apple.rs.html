14/// The deployment target for the current OS.
15pub(crate) const DEPLOYMENT_TARGET: OSVersion = {
16    // Intentionally use `#[cfg]` guards instead of `cfg!` here, to avoid
17    // recompiling when unrelated environment variables change.
18    #[cfg(target_os = "macos")]
19    let var = option_env!("MACOSX_DEPLOYMENT_TARGET");
20    #[cfg(target_os = "ios")] // Also used on Mac Catalyst.
21    let var = option_env!("IPHONEOS_DEPLOYMENT_TARGET");
22    #[cfg(target_os = "tvos")]
23    let var = option_env!("TVOS_DEPLOYMENT_TARGET");
24    #[cfg(target_os = "watchos")]
25    let var = option_env!("WATCHOS_DEPLOYMENT_TARGET");
26    #[cfg(target_os = "visionos")]
27    let var = option_env!("XROS_DEPLOYMENT_TARGET");
28
29    if let Some(var) = var {
30        OSVersion::from_str(var)
31    } else {
32        // Default operating system version.
33        // See <https://github.com/rust-lang/rust/blob/1e5719bdc40bb553089ce83525f07dfe0b2e71e9/compiler/rustc_target/src/spec/base/apple/mod.rs#L207-L215>
34        //
35        // Note that we cannot do as they suggest, and use
36        // `rustc --print=deployment-target`, as this has to work at `const`
37        // time.
38        #[allow(clippy::if_same_then_else)]
39        let os_min = if cfg!(target_os = "macos") {
40            (10, 12, 0)
41        } else if cfg!(target_os = "ios") {
42            (10, 0, 0)
43        } else if cfg!(target_os = "tvos") {
44            (10, 0, 0)
45        } else if cfg!(target_os = "watchos") {
46            (5, 0, 0)
47        } else if cfg!(target_os = "visionos") {
48            (1, 0, 0)
49        } else {
50            panic!("unknown Apple OS")
51        };
52
53        // On certain targets it makes sense to raise the minimum OS version.
54        //
55        // See <https://github.com/rust-lang/rust/blob/1e5719bdc40bb553089ce83525f07dfe0b2e71e9/compiler/rustc_target/src/spec/base/apple/mod.rs#L217-L231>
56        //
57        // Note that we cannot do all the same checks as `rustc` does, because
58        // we have no way of knowing if the architecture is `arm64e` without
59        // reading the target triple itself (and we want to get rid of build
60        // scripts).
61        #[allow(clippy::if_same_then_else)]
62        let min = if cfg!(all(target_os = "macos", target_arch = "aarch64")) {
63            (11, 0, 0)
64        } else if cfg!(all(
65            target_os = "ios",
66            target_arch = "aarch64",
67            target_abi_macabi
68        )) {
69            (14, 0, 0)
70        } else if cfg!(all(
71            target_os = "ios",
72            target_arch = "aarch64",
73            target_simulator
74        )) {
75            (14, 0, 0)
76        } else if cfg!(all(target_os = "tvos", target_arch = "aarch64")) {
77            (14, 0, 0)
78        } else if cfg!(all(target_os = "watchos", target_arch = "aarch64")) {
79            (7, 0, 0)
80        } else {
81            os_min
82        };
83
84        OSVersion {
85            major: min.0,
86            minor: min.1,
87            patch: min.2,
88        }
89    }
90};

92/// Look up the current version at runtime.
93///
94/// Note that this doesn't work with "zippered" `dylib`s yet, though
95/// that's probably fine, `rustc` doesn't support those either:
96/// <https://github.com/rust-lang/rust/issues/131216>
97#[inline]
98pub(crate) fn current_version() -> OSVersion {
99    // Cache the lookup for performance.
100    //
101    // We assume that 0.0.0 is never gonna be a valid version,
102    // and use that as our sentinel value.
103    static CURRENT_VERSION: AtomicU32 = AtomicU32::new(0);
104
105    // We use relaxed atomics, it doesn't matter if two threads end up racing
106    // to read or write the version.
107    let version = CURRENT_VERSION.load(Ordering::Relaxed);
108    OSVersion::from_u32(if version == 0 {
109        // TODO: Consider using `std::panic::abort_unwind` here for code-size?
110        let version = lookup_version().get();
111        CURRENT_VERSION.store(version, Ordering::Relaxed);
112        version
113    } else {
114        version
115    })
116}

118#[cold]
119fn lookup_version() -> NonZeroU32 {
120    // Since macOS 10.15, libSystem has provided the undocumented
121    // `_availability_version_check` via `libxpc` for doing this version
122    // lookup, though it's usage may be a bit dangerous, see:
123    // - https://reviews.llvm.org/D150397
124    // - https://github.com/llvm/llvm-project/issues/64227
125    //
126    // So instead, we use the safer approach of reading from `sysctl`, and
127    // if that fails, we fall back to the property list (this is what
128    // `_availability_version_check` does internally).
129    let version = version_from_sysctl().unwrap_or_else(version_from_plist);
130    // Use `NonZeroU32` to try to make it clearer to the optimizer that this
131    // will never return 0.
132    NonZeroU32::new(version.to_u32()).expect("version cannot be 0.0.0")
133}

135/// Read the version from `kern.osproductversion` or `kern.iossupportversion`.
136fn version_from_sysctl() -> Option<OSVersion> {
137    // This won't work in the simulator, `kern.osproductversion` will return
138    // the host macOS version.
139    if cfg!(target_simulator) {
140        return None;
141    }
142
143    // SAFETY: Same signature as in `libc`
144    extern "C" {
145        fn sysctlbyname(
146            name: *const c_char,
147            oldp: *mut c_void,
148            oldlenp: *mut usize,
149            newp: *mut c_void,
150            newlen: usize,
151        ) -> c_uint;
152    }
153
154    let name = if cfg!(target_abi_macabi) {
155        b"kern.iossupportversion\0".as_ptr().cast()
156    } else {
157        // Introduced in macOS 10.13.4.
158        b"kern.osproductversion\0".as_ptr().cast()
159    };
160
161    let mut buf: [u8; 32] = [0; 32];
162    let mut size = buf.len();
163    let ret = unsafe { sysctlbyname(name, buf.as_mut_ptr().cast(), &mut size, ptr::null_mut(), 0) };
164    if ret != 0 {
165        // `sysctlbyname` is not available.
166        return None;
167    }
168
169    Some(OSVersion::from_bytes(&buf[..(size - 1)]))
170}

172/// Look up the current OS version from the `ProductVersion` or
173/// `iOSSupportVersion` in `/System/Library/CoreServices/SystemVersion.plist`.
174/// This file was introduced in macOS 10.3.0.
175///
176/// This is also what is done in `compiler-rt`:
177/// <https://github.com/llvm/llvm-project/blob/llvmorg-19.1.1/compiler-rt/lib/builtins/os_version_check.c>
178///
179/// NOTE: I don't _think_ we need to do a similar thing as what Zig does to
180/// handle the fake 10.16 versions returned when the SDK version of the binary
181/// is less than 11.0:
182/// <https://github.com/ziglang/zig/blob/0.13.0/lib/std/zig/system/darwin/macos.zig>
183///
184/// My reasoning is that we _want_ to follow Apple's behaviour here, and
185/// return 10.16 when compiled with an older SDK; the user should upgrade
186/// their tooling.
187///
188/// NOTE: `rustc` currently doesn't set the right SDK version when linking
189/// with ld64, so this will usually have the wrong behaviour on x86_64. But
190/// that's a `rustc` bug, and is tracked in:
191/// <https://github.com/rust-lang/rust/issues/129432>
192///
193///
194/// # Panics
195///
196/// Panics if reading or parsing the PList fails (or if the system was out of
197/// memory).
198///
199/// We deliberately choose to panic, as having this lookup silently return
200/// an empty OS version would be impossible for a user to debug.
201fn version_from_plist() -> OSVersion {
202    // Use Foundation's mechanisms for reading the PList.
203    autoreleasepool(|pool| {
204        let path: Retained<NSObject> = if cfg!(target_simulator) {
205            let root = std::env::var_os("IPHONE_SIMULATOR_ROOT")
206                    .expect("environment variable `IPHONE_SIMULATOR_ROOT` must be set when executing under simulator");
207            let path = PathBuf::from(root).join("System/Library/CoreServices/SystemVersion.plist");
208            let path = path.as_os_str().as_bytes();
209
210            // SAFETY: Allocating a string is valid on all threads.
211            let alloc: Allocated<NSObject> = unsafe { Allocated::alloc(class!(NSString)) };
212            // SAFETY: The bytes are valid, and the length is correct.
213            unsafe {
214                let bytes_ptr: *const c_void = path.as_ptr().cast();
215                msg_send![
216                    alloc,
217                    initWithBytes: bytes_ptr,
218                    length: path.len(),
219                    // OsStr is a superset of UTF-8 on unix platforms
220                    encoding: UTF8_ENCODING,
221                ]
222            }
223        } else {
224            let path: *const c_char = b"/System/Library/CoreServices/SystemVersion.plist\0"
225                .as_ptr()
226                .cast();
227            // SAFETY: The path is NULL terminated.
228            unsafe { msg_send![class!(NSString), stringWithUTF8String: path] }
229        };
230
231        // SAFETY: dictionaryWithContentsOfFile: is safe to call.
232        let data: Option<Retained<NSObject>> =
233            unsafe { msg_send![class!(NSDictionary), dictionaryWithContentsOfFile: &*path] };
234
235        let data = data.expect(
236            "`/System/Library/CoreServices/SystemVersion.plist` must be readable, and contain a valid PList",
237        );
238
239        // Read `ProductVersion`, except when running on Mac Catalyst, then we
240        // read `iOSSupportVersion` instead.
241        let lookup_key: *const c_char = if cfg!(target_abi_macabi) {
242            b"iOSSupportVersion\0".as_ptr().cast()
243        } else {
244            b"ProductVersion\0".as_ptr().cast()
245        };
246        // SAFETY: The lookup key is NULL terminated.
247        let lookup_key: Retained<NSObject> =
248            unsafe { msg_send![class!(NSString), stringWithUTF8String: lookup_key] };
249
250        let version: Retained<NSObject> = unsafe { msg_send![&data, objectForKey: &*lookup_key] };
251
252        assert!(
253            version.isKindOfClass(class!(NSString)),
254            "`ProductVersion` key in `/System/Library/CoreServices/SystemVersion.plist` must be a string"
255        );
256
257        // SAFETY: The given object is an NSString, and the returned string
258        // slice is not used outside of the current pool.
259        let version = unsafe { nsstring_to_str(&version, pool) };
260
261        OSVersion::from_str(version)
262    })
263}
