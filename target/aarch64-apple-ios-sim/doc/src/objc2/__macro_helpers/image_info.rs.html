4pub struct ImageInfo {
5    // These are not actually `unsigned int`, even though the docs say so
6    /// The version of the image info struct.
7    version: u32,
8    flags: u32,
9}
10
11#[allow(unused)]
12impl ImageInfo {
13    /// Unused
14    const FIX_AND_CONTINUE: u32 = 1 << 0;
15    const SUPPORTS_GARBAGE_COLLECTED: u32 = 1 << 1;
16    const REQUIRES_GARBAGE_COLLECTION: u32 = 1 << 2;
17    const OPTIMIZED_BY_DYLD: u32 = 1 << 3; // TODO
18    /// Unused
19    const CORRECTED_SYNTHESIZE: u32 = 1 << 4;
20    /// Whether we're compiling this to run on a simulator.
21    const IMAGE_IS_SIMULATED: u32 = 1 << 5;
22    /// Whether we are generating class properties.
23    const CLASS_PROPERTIES: u32 = 1 << 6;
24    const DYLD_PREOPTIMIZED: u32 = 1 << 7;
25
26    const SWIFT_ABI_VERSION_SHIFT: u32 = 8;
27    const SWIFT_ABI_VERSION_MASK: u32 = 0xff << Self::SWIFT_ABI_VERSION_SHIFT;
28    const SWIFT_MINOR_VERSION_SHIFT: u32 = 16;
29    const SWIFT_MINOR_VERSION_MASK: u32 = 0xff << Self::SWIFT_MINOR_VERSION_SHIFT;
30    const SWIFT_MAJOR_VERSION_SHIFT: u32 = 24;
31    const SWIFT_MAJOR_VERSION_MASK: u32 = 0xff << Self::SWIFT_MAJOR_VERSION_SHIFT;
32
33    /// Fetches the image info for the current runtime + target combination
34    #[inline]
35    pub const fn system() -> Self {
36        // We don't currently do anything relating to class properties, but
37        // let's mimic what Clang does!
38        let mut flags = Self::CLASS_PROPERTIES;
39
40        if cfg!(target_simulator) {
41            flags |= Self::IMAGE_IS_SIMULATED;
42        }
43
44        Self { version: 0, flags }
45    }
