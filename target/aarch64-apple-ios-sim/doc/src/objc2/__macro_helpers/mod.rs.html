70    #[cfg(any(feature = "unstable-static-sel", feature = "unstable-static-class"))]
71    use crate::__hash_idents;
72
73    #[test]
74    #[cfg(any(feature = "unstable-static-sel", feature = "unstable-static-class"))]
75    fn hash_idents_different() {
76        assert_ne!(__hash_idents!(abc), __hash_idents!(def));
77    }
78
79    #[test]
80    #[cfg(any(feature = "unstable-static-sel", feature = "unstable-static-class"))]
81    fn hash_idents_same_no_equal() {
82        assert_ne!(__hash_idents!(abc), __hash_idents!(abc));
83        assert_ne!(__hash_idents!(abc def ghi), __hash_idents!(abc def ghi));
84    }
85
86    #[test]
87    #[cfg(any(feature = "unstable-static-sel", feature = "unstable-static-class"))]
88    fn hash_idents_exact_same_ident() {
89        macro_rules! x {
90            ($x:ident) => {
91                (__hash_idents!($x), __hash_idents!($x))
92            };
93        }
94        let (ident1, ident2) = x!(abc);
95        // This is a limitation of `__hash_idents`, ideally we'd like these
96        // to be different!
97        assert_eq!(ident1, ident2);
98    }
99
100    #[test]

103        ignore = "Only relevant on macOS 32-bit"
104    )]
