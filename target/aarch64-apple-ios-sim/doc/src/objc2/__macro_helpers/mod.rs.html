45/// Helper struct for emitting the module info that macOS 32-bit requires.
46///
47/// <https://github.com/llvm/llvm-project/blob/release/13.x/clang/lib/CodeGen/CGObjCMac.cpp#L5211-L5234>
48#[repr(C)]
49#[derive(Debug)]
50pub struct ModuleInfo {
51    version: usize,
52    size: usize,
53    name: *const u8,
54    symtab: *const (),
55}
56
57// SAFETY: ModuleInfo is immutable.
58unsafe impl Sync for ModuleInfo {}
59
60impl ModuleInfo {
61    /// This is hardcoded in clang as 7.
62    const VERSION: usize = 7;
63
64    pub const fn new(name: *const u8) -> Self {
65        Self {
66            version: Self::VERSION,
67            size: core::mem::size_of::<Self>(),
68            name,
69            // We don't expose any symbols
70            symtab: core::ptr::null(),
71        }
72    }
73}

82    #[test]
83    #[cfg(feature = "objc2-proc-macros")]
84    fn hash_idents_different() {
85        assert_ne!(__hash_idents!(abc), __hash_idents!(def));
86    }
87
88    #[test]
89    #[cfg(feature = "objc2-proc-macros")]
90    fn hash_idents_same_no_equal() {
91        assert_ne!(__hash_idents!(abc), __hash_idents!(abc));
92        assert_ne!(__hash_idents!(abc def ghi), __hash_idents!(abc def ghi));
93    }
94
95    #[test]
96    #[cfg(feature = "objc2-proc-macros")]
97    fn hash_idents_exact_same_ident() {
98        macro_rules! x {
99            ($x:ident) => {
100                (__hash_idents!($x), __hash_idents!($x))
101            };
102        }
103        let (ident1, ident2) = x!(abc);
104        // This is a limitation of `__hash_idents`, ideally we'd like these
105        // to be different!
106        assert_eq!(ident1, ident2);
107    }
108
109    #[test]
110    #[cfg_attr(
111        not(all(target_vendor = "apple", target_os = "macos", target_arch = "x86")),
112        ignore = "Only relevant on macOS 32-bit"
113    )]
114    fn ensure_size_of_module_info() {
115        assert_eq!(core::mem::size_of::<ModuleInfo>(), 16);
116    }
