1/// Helper struct for emitting the module info that macOS 32-bit requires.
2///
3/// <https://github.com/llvm/llvm-project/blob/release/13.x/clang/lib/CodeGen/CGObjCMac.cpp#L5211-L5234>
4#[repr(C)]
5#[derive(Debug)]
6pub struct ModuleInfo {
7    version: usize,
8    size: usize,
9    name: *const u8,
10    symtab: *const (),
11}
12
13// SAFETY: ModuleInfo is immutable.
14unsafe impl Sync for ModuleInfo {}
15
16impl ModuleInfo {
17    /// This is hardcoded in clang as 7.
18    const VERSION: usize = 7;
19
20    pub const fn new(name: *const u8) -> Self {
21        Self {
22            version: Self::VERSION,
23            size: core::mem::size_of::<Self>(),
24            name,
25            // We don't expose any symbols
26            symtab: core::ptr::null(),
27        }
28    }
29}
