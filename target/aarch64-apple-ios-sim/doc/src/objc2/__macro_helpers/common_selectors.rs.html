1//! Common selectors.
2//!
3//! These are put here to deduplicate the cached selector, and when using
4//! `unstable-static-sel`, the statics.
5//!
6//! Note that our assembly tests of `unstable-static-sel-inlined` output a GOT
7//! entry for such accesses, but that is just a limitation of our tests - the
8//! actual assembly is as one would expect.
9use crate::runtime::Sel;
10use crate::{__sel_data, __sel_inner};
11
12#[inline]
13pub fn alloc_sel() -> Sel {
14    __sel_inner!(__sel_data!(alloc), "alloc")
15}
16
17#[inline]
18pub fn init_sel() -> Sel {
19    __sel_inner!(__sel_data!(init), "init")
20}
21
22#[inline]
23pub fn new_sel() -> Sel {
24    __sel_inner!(__sel_data!(new), "new")
25}
26
27#[inline]
28pub fn dealloc_sel() -> Sel {
29    __sel_inner!(__sel_data!(dealloc), "dealloc")
30}

32/// An undocumented selector called by the Objective-C runtime when
33/// initalizing instance variables.
34#[inline]
35#[allow(dead_code)] // May be useful in the future
36fn cxx_construct_sel() -> Sel {
37    __sel_inner!(".cxx_construct\0", ".cxx_construct")
38}

40/// Objective-C runtimes call `.cxx_destruct` as part of the final `dealloc`
41/// call inside `NSObject` (and has done so since macOS 10.4).
42///
43/// While [GCC does document it somewhat][gcc-docs], this is still severely
44/// undocumented in clang - but since the selector is emitted into the final
45/// binary, it is fine to rely on it being used.
46///
47/// Unfortunately though, this only works if the class has been declared
48/// statically, since in that case a flag is set to inform the runtime that it
49/// needs to run destructors. So unfortunately we can't use this on
50/// dynamically declared classes.
51///
52///
53/// # ABI
54///
55/// The ABI of `.cxx_destruct` in Apple's runtime is actually that it does NOT
56/// take a selector, unlike every other Objective-C method, see:
57/// <https://github.com/apple-oss-distributions/objc4/blob/objc4-906/runtime/objc-class.mm#L457>
58///
59/// So the signature is `extern "C" fn(*mut AnyObject)`.
60///
61/// This is likely because it's not a real Objective-C method that can be
62/// called from userspace / objc_msgSend, and it's more efficient to not pass
63/// the selector.
64///
65/// Note that even if Apple decides to suddenly add the selector one day,
66/// ignoring it will still be sound, since the function uses the C calling
67/// convention, where such an ignored parameter would be allowed on all
68/// relevant architectures.
69///
70/// TODO: Unsure whether "C-unwind" is allowed?
71///
72/// [gcc-docs]: https://gcc.gnu.org/onlinedocs/gcc/Objective-C-and-Objective-C_002b_002b-Dialect-Options.html#index-fobjc-call-cxx-cdtors
73#[inline]
74#[allow(dead_code)] // May be useful in the future
75fn cxx_destruct_sel() -> Sel {
76    __sel_inner!(".cxx_destruct\0", ".cxx_destruct")
77}

79#[cfg(test)]
80mod tests {
81    use core::sync::atomic::{AtomicBool, Ordering};
82
83    use crate::rc::Retained;
84    use crate::runtime::ClassBuilder;
85    use crate::runtime::NSObject;
86    use crate::{msg_send_id, ClassType};
87
88    use super::*;
89
90    /// Test the unfortunate fact that we can't use .cxx_destruct on dynamic classes.
91    #[test]
92    fn test_destruct_dynamic() {
93        static HAS_RUN: AtomicBool = AtomicBool::new(false);
94
95        let mut builder = ClassBuilder::new("TestCxxDestruct", NSObject::class()).unwrap();
96
97        unsafe extern "C" fn destruct(_: *mut NSObject, _: Sel) {
98            HAS_RUN.store(true, Ordering::Relaxed);
99        }
100
101        // Note: The ABI is not upheld here, but its fine for this test
102        unsafe { builder.add_method(cxx_destruct_sel(), destruct as unsafe extern "C" fn(_, _)) };
103
104        let cls = builder.register();
105
106        let obj: Retained<NSObject> = unsafe { msg_send_id![cls, new] };
107        drop(obj);
108        let has_run_destruct = HAS_RUN.load(Ordering::Relaxed);
109
110        // This does work on GNUStep, but unfortunately not in Apple's objc4
111        if cfg!(feature = "gnustep-1-7") {
112            assert!(has_run_destruct);
113        } else {
114            assert!(!has_run_destruct);
115        }
116    }
