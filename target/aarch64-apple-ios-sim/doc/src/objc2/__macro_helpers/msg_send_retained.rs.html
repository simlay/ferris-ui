13//
14// MsgSend
15//
16
17pub trait MsgSend<Receiver, Return> {
18    #[track_caller]
19    unsafe fn send_message<A: ConvertArguments>(receiver: Receiver, sel: Sel, args: A) -> Return;
20}
21
22impl<Receiver, Return, MethodFamily> MsgSend<Receiver, Return> for MethodFamily
23where
24    MethodFamily: RetainSemantics<Receiver, Return, KindSendMessage>,
25{
26    #[inline]
27    unsafe fn send_message<A: ConvertArguments>(receiver: Receiver, sel: Sel, args: A) -> Return {
28        let ptr = Self::prepare_message_send(receiver).__as_raw_receiver();
29
30        // SAFETY: The writeback helper is not leaked (it is dropped at the
31        // end of this scope).
32        let (args, _helper) = unsafe { A::__into_arguments(args) };
33
34        // SAFETY: Upheld by caller.
35        let ret = unsafe { MessageReceiver::send_message(ptr, sel, args) };
36
37        // SAFETY: The pointers are valid (or, in the case of the receiver
38        // pointer, at least valid when the message send is not `init`).
39        unsafe { Self::convert_message_return(ret, ptr, sel) }
40    }
41}

44// MsgSendSuper
45//
46
47pub trait MsgSendSuper<Receiver, Return> {
48    type Inner: ?Sized + RefEncode;
49
50    #[track_caller]
51    unsafe fn send_super_message<A: ConvertArguments>(
52        receiver: Receiver,
53        superclass: &AnyClass,
54        sel: Sel,
55        args: A,
56    ) -> Return;
57
58    #[inline]
59    #[track_caller]
60    unsafe fn send_super_message_static<A: ConvertArguments>(
61        receiver: Receiver,
62        sel: Sel,
63        args: A,
64    ) -> Return
65    where
66        Self::Inner: ClassType,
67        <Self::Inner as ClassType>::Super: ClassType,
68    {
69        unsafe {
70            Self::send_super_message(
71                receiver,
72                <Self::Inner as ClassType>::Super::class(),
73                sel,
74                args,
75            )
76        }
77    }
78}

109// MsgSendError

111
112pub trait MsgSendError<Receiver, Return> {
113    /// Add an extra error argument to the argument list, call `send_message`
114    /// with that, and return an error if one occurred.
115    #[track_caller]
116    unsafe fn send_message_error<A, E>(
117        receiver: Receiver,
118        sel: Sel,
119        args: A,
120    ) -> Result<Return, Retained<E>>
121    where
122        *mut *mut E: Encode,
123        A: TupleExtender<*mut *mut E>,
124        <A as TupleExtender<*mut *mut E>>::PlusOneArgument: ConvertArguments,
125        E: ClassType;
126}
127
128// `Option<Retained<T>>` return.
129impl<Receiver, Return, MethodFamily> MsgSendError<Receiver, Retained<Return>> for MethodFamily
130where
131    MethodFamily: MsgSend<Receiver, Option<Retained<Return>>>,
132{
133    #[inline]
134    unsafe fn send_message_error<A, E>(
135        receiver: Receiver,
136        sel: Sel,
137        args: A,
138    ) -> Result<Retained<Return>, Retained<E>>
139    where
140        *mut *mut E: Encode,
141        A: TupleExtender<*mut *mut E>,
142        <A as TupleExtender<*mut *mut E>>::PlusOneArgument: ConvertArguments,
143        E: ClassType,
144    {
145        let mut err: *mut E = ptr::null_mut();
146        let args = args.add_argument(&mut err);
147        let ret = unsafe { Self::send_message(receiver, sel, args) };
148        // As per the Cocoa documentation:
149        // > Success or failure is indicated by the return value of the
150        // > method. Although Cocoa methods that indirectly return error
151        // > objects in the Cocoa error domain are guaranteed to return such
152        // > objects if the method indicates failure by directly returning
153        // > `nil` or `NO`, you should always check that the return value is
154        // > `nil` or `NO` before attempting to do anything with the `NSError`
155        // > object.
156        if let Some(ret) = ret {
157            // In this case, the error is likely not created. If it is, it is
158            // autoreleased anyhow, so it would be a waste to retain and
159            // release it here.
160            Ok(ret)
161        } else {
162            // In this case, the error has very likely been created, but has
163            // been autoreleased (as is common for "out parameters", see
164            // `src/__macro_helpers/writeback.rs`). Hence we need to retain it
165            // if we want it to live across autorelease pools.
166            //
167            // SAFETY: The message send is guaranteed to populate the error
168            // object, or leave it as NULL. The error is shared, and all
169            // holders of the error know this, so is safe to retain.
170            Err(unsafe { encountered_error(err) })
171        }
172    }
173}

175// Bool return.
176impl<Receiver, MethodFamily> MsgSendError<Receiver, ()> for MethodFamily
177where
178    MethodFamily: MsgSend<Receiver, bool>,
179{
180    #[inline]
181    unsafe fn send_message_error<A, E>(
182        receiver: Receiver,
183        sel: Sel,
184        args: A,
185    ) -> Result<(), Retained<E>>
186    where
187        *mut *mut E: Encode,
188        A: TupleExtender<*mut *mut E>,
189        <A as TupleExtender<*mut *mut E>>::PlusOneArgument: ConvertArguments,
190        E: ClassType,
191    {
192        let mut err: *mut E = ptr::null_mut();
193        let args = args.add_argument(&mut err);
194        let ret = unsafe { Self::send_message(receiver, sel, args) };
195        if ret {
196            Ok(())
197        } else {
198            Err(unsafe { encountered_error(err) })
199        }
200    }
201}

789    /// This is imperfect, but will do for now.
790    const fn autorelease_skipped(self_declared: bool) -> bool {
791        if cfg!(feature = "gnustep-1-7") {
792            // GNUStep does the optimization a different way, so it isn't
793            // optimization-dependent.
794            true
795        } else if cfg!(all(target_arch = "arm", panic = "unwind")) {
796            // 32-bit ARM unwinding sometimes interferes with the optimization
797            false
798        } else if self_declared {
799            // FIXME: Autorelease_return is not currently tail-called, so the
800            // optimization doesn't work on define_class! functions.
801            false
802        } else if cfg!(feature = "catch-all") {
803            // FIXME: `catch-all` is inserted before we get a chance to retain.
804            false
805        } else if cfg!(debug_assertions) {
806            // `debug_assertions` ~proxy for if optimizations are off.
807            false
808        } else {
809            true
810        }
811    }

813    macro_rules! test_error_retained {
814        ($expected:expr, $if_autorelease_not_skipped:expr, $sel:ident, $($obj:tt)*) => {
815            // Succeeds
816            let res = autoreleasepool(|_pool| {
817                let res: Result<Retained<RcTestObject>, Retained<NSObject>> = unsafe {
818                    msg_send![$($obj)*, $sel: false, error: _]
819                };
820                let res = res.expect("not ok");
821                $expected.alloc += 1;
822                $expected.init += 1;
823                $expected.autorelease += $if_autorelease_not_skipped;
824                $expected.retain += $if_autorelease_not_skipped;
825                $expected.assert_current();
826                res
827            });
828            $expected.release += $if_autorelease_not_skipped;
829            $expected.assert_current();
830
831            drop(res);
832            $expected.release += 1;
833            $expected.drop += 1;
834            $expected.assert_current();
835
836            // Errors
837            let res = autoreleasepool(|_pool| {
838                let res: Result<Retained<RcTestObject>, Retained<NSObject>> = unsafe {
839                    msg_send![$($obj)*, $sel: true, error: _]
840                };
841                $expected.alloc += 1;
842                $expected.init += 1;
843                $expected.autorelease += 1;
844                $expected.retain += 1;
845                $expected.assert_current();
846                res.expect_err("not err")
847            });
848            $expected.release += 1;
849            $expected.assert_current();
850
851            drop(res);
852            $expected.release += 1;
853            $expected.drop += 1;
854            $expected.assert_current();
855        }
856    }

920        let obj = ManuallyDrop::new(NSObject::new());

922            let obj: *mut NSObject = msg_send![&*obj, autorelease];
923            // All code between the `msg_send!` and the `retain_autoreleased`

926        }

929    #[test]
930    fn test_retain_autoreleased() {
931        autoreleasepool(|_| {
932            // Run once to allow DYLD to resolve the symbol stubs.
933            // Required for making `retain_autoreleased` work on x86_64.
934            let _data = create_obj();
935
936            // When compiled in release mode / with optimizations enabled,
937            // subsequent usage of `retain_autoreleased` will succeed in
938            // retaining the autoreleased value!
939            let expected = if autorelease_skipped(false) { 1 } else { 2 };
940
941            let data = create_obj();
942            assert_eq!(data.retainCount(), expected);
943
944            let data = create_obj();
945            assert_eq!(data.retainCount(), expected);
946
947            // Here we manually clean up the autorelease, so it will always be 1.
948            let data = autoreleasepool(|_| create_obj());
949            assert_eq!(data.retainCount(), 1);
950        });
951    }
