1//! Support for passing "out"-parameters to `msg_send!` and family.
2//!
3//! See clang's documentation:
4//! <https://clang.llvm.org/docs/AutomaticReferenceCounting.html#passing-to-an-out-parameter-by-writeback>
5//!
6//! Note: We differ from that in that we do not create a temporary, whose
7//! address we then work on; instead, we directly reuse the pointer that the
8//! user provides (since, if it's a mutable pointer, we know that it's not
9//! shared elsewhere in the program, and hence it is safe to modify directly).
10//!
11//! Another important consideration is unwinding; I haven't researched how
12//! Clang handles that, but the correct thing is to do the writeback
13//! retain/release dance regardless of whether the function unwinded or not.
14//! We ensure this by doing it in `Drop`.
15use core::mem::ManuallyDrop;
16use core::ptr::NonNull;
17
18use super::ConvertArgument;
19use crate::rc::Retained;
20use crate::Message;

22// Note the `'static` bound here - this may not be necessary, but I'm unsure
23// of the exact requirements, so we better keep it for now.
24impl<T: Message + 'static> ConvertArgument for &mut Retained<T> {
25    // We use `*mut T` as the inner value instead of `NonNull<T>`, since we
26    // want to do debug checking that the value hasn't unexpectedly been
27    // overwritten to contain NULL (which is clear UB, but the user might have
28    // made a mistake).
29    type __Inner = NonNull<*mut T>;
30
31    type __WritebackOnDrop = WritebackOnDrop<T>;
32
33    #[inline]
34    fn __from_defined_param(_inner: Self::__Inner) -> Self {
35        todo!("`&mut Retained<_>` is not supported in `define_class!` yet")
36    }
37
38    #[inline]
39    unsafe fn __into_argument(self) -> (Self::__Inner, Self::__WritebackOnDrop) {
40        let ptr: NonNull<Retained<T>> = NonNull::from(self);
41        // `Retained` is `#[repr(transparent)]` over `NonNull`.
42        let ptr: NonNull<NonNull<T>> = ptr.cast();
43
44        // SAFETY: The value came from `&mut _`, and we only read a pointer.
45        let old: NonNull<T> = unsafe { *ptr.as_ptr() };
46
47        // `NonNull<T>` has the same layout as `*mut T`.
48        let ptr: NonNull<*mut T> = ptr.cast();
49
50        (ptr, WritebackOnDrop { ptr, old })
51    }
52}

54#[derive(Debug)]
55pub struct WritebackOnDrop<T: Message> {
56    /// A copy of the argument, so that we can retain it after the message
57    /// send.
58    ///
59    /// Ideally, we'd work with e.g. `&mut *mut T`, but we can't do that
60    /// inside the generic context of `MessageArguments::__invoke`, while
61    /// working within Rust's aliasing rules.
62    ptr: NonNull<*mut T>,
63    /// The old value, stored so that we can release if after the message
64    /// send.
65    old: NonNull<T>,
66}
67
68impl<T: Message> Drop for WritebackOnDrop<T> {
69    #[inline]
70    fn drop(&mut self) {
71        // In terms of provenance, we roughly want to do the following:
72        // ```
73        // fn do(value: &mut Retained<T>) {
74        //     let old = value.clone();
75        //     msg_send![... value ...];
76        //     let _ = value.clone();
77        //     drop(old);
78        // }
79        // ```
80        //
81        // Which is definitely valid under stacked borrows! See also this
82        // playground link for testing something equivalent in Miri:
83        // <https://play.rust-lang.org/?version=stable&mode=debug&edition=2021&gist=5ad8fcff1f870819081aa534ec754b86>
84        //
85        //
86        //
87        // In Objective-C terms, we want to retain the new value and release
88        // the old, and importantly, in that order (such that we don't dealloc
89        // the value if it didn't change). So something like this:
90        // ```
91        // fn do(value: &mut Retained) {
92        //     let old = *value;
93        //     msg_send![... value ...];
94        //     objc_retain(*value);
95        //     objc_release(old);
96        // }
97        // ```
98
99        // SAFETY: Caller ensures that the pointer is either left as-is, or is
100        // safe to retain at this point.
101        let new: Option<Retained<T>> = unsafe { Retained::retain(*self.ptr.as_ptr()) };
102        // We ignore the result of `retain`, since it always returns the same
103        // value as was given (and it would be unnecessary work to write that
104        // value back into `ptr` again).
105        let _new = ManuallyDrop::new(new);
106        #[cfg(debug_assertions)]
107        if _new.is_none() {
108            panic!("found that NULL was written to `&mut Retained<_>`, which is UB! You should handle this with `&mut Option<Retained<_>>` instead");
109        }
110
111        // SAFETY: The old pointer was valid when it was constructed.
112        //
113        // If the message send modified the argument, they would have left a
114        // +1 retain count on the old pointer; so either we have +1 from that,
115        // or the message send didn't modify the pointer and we instead have
116        // +1 retain count from the `retain` above.
117        let _: Retained<T> = unsafe { Retained::new_nonnull(self.old) };
118    }
119}

121impl<T: Message + 'static> ConvertArgument for &mut Option<Retained<T>> {
122    type __Inner = NonNull<*mut T>;
123
124    type __WritebackOnDrop = WritebackOnDropNullable<T>;
125
126    #[inline]
127    fn __from_defined_param(_inner: Self::__Inner) -> Self {
128        todo!("`&mut Option<Retained<_>>` is not supported in `define_class!` yet")
129    }
130
131    #[inline]
132    unsafe fn __into_argument(self) -> (Self::__Inner, Self::__WritebackOnDrop) {
133        let ptr: NonNull<Option<Retained<T>>> = NonNull::from(self);
134        // `Option<Retained<T>>` has the same memory layout as `*mut T`.
135        let ptr: NonNull<*mut T> = ptr.cast();
136        // SAFETY: Same as for `&mut Retained`
137        let old: *mut T = unsafe { *ptr.as_ptr() };
138
139        (ptr, WritebackOnDropNullable { ptr, old })
140    }
141}

143/// Mostly the same as `WritebackOnDrop`, except that the old value is
144/// nullable.
145#[derive(Debug)]
146pub struct WritebackOnDropNullable<T: Message> {
147    ptr: NonNull<*mut T>,
148    old: *mut T,
149}
150
151impl<T: Message> Drop for WritebackOnDropNullable<T> {
152    #[inline]
153    fn drop(&mut self) {
154        // SAFETY: Same as for `&mut Retained`
155        let new: Option<Retained<T>> = unsafe { Retained::retain(*self.ptr.as_ptr()) };
156        let _ = ManuallyDrop::new(new);
157
158        // SAFETY: Same as for `&mut Retained`
159        //
160        // Note: We explicitly keep the `if old == nil { objc_release(old) }`
161        // check, since we expect that the user would often do:
162        //
163        // ```
164        // let mut value = None
165        // do(&mut value);
166        // ```
167        //
168        // And in that case, we can elide the `objc_release`!
169        let _: Option<Retained<T>> = unsafe { Retained::from_raw(self.old) };
170    }
171}
172
173// Note: For `Option<&mut ...>` we explicitly want to do the `if Some` checks
174// before anything else, since whether `None` or `Some` was passed is often
175// known at compile-time, and for the `None` case it would be detrimental to
176// have extra `retain/release` calls here.
177
178impl<T: Message + 'static> ConvertArgument for Option<&mut Retained<T>> {
179    type __Inner = Option<NonNull<*mut T>>;
180
181    type __WritebackOnDrop = Option<WritebackOnDrop<T>>;
182
183    #[inline]
184    fn __from_defined_param(_inner: Self::__Inner) -> Self {
185        todo!("`Option<&mut Retained<_>>` is not supported in `define_class!` yet")
186    }
187
188    #[inline]
189    unsafe fn __into_argument(self) -> (Self::__Inner, Self::__WritebackOnDrop) {
190        if let Some(this) = self {
191            // SAFETY: Upheld by caller.
192            let (ptr, helper) = unsafe { this.__into_argument() };
193            (Some(ptr), Some(helper))
194        } else {
195            (None, None)
196        }
197    }
198}

231    fn test_bool_error() {
232        let mut expected = ThreadTestData::current();
233
234        fn bool_error(should_error: bool, error: Option<&mut Option<Retained<RcTestObject>>>) {
235            let cls = RcTestObject::class();
236            let did_succeed: bool =
237                unsafe { msg_send![cls, boolAndShouldError: should_error, error: error] };
238            assert_ne!(should_error, did_succeed);
239        }
240
241        bool_error(false, None);
242        bool_error(true, None);
243        expected.assert_current();
244
245        fn helper(
246            expected: &mut ThreadTestData,
247            should_error: bool,
248            mut error: Option<Retained<RcTestObject>>,
249        ) {
250            autoreleasepool(|_| {
251                bool_error(should_error, Some(&mut error));
252                if should_error {
253                    expected.alloc += 1;
254                    expected.init += 1;
255                    expected.autorelease += 1;
256                }
257                expected.assert_current();
258            });
259
260            if should_error {
261                expected.release += 1;
262            }
263            expected.assert_current();
264
265            if error.is_some() {
266                expected.release += 1;
267                expected.drop += 1;
268            }
269            drop(error);
270            expected.assert_current();
271        }
272
273        helper(&mut expected, false, None);
274
275        expected.retain += 1;
276        helper(&mut expected, true, None);
277
278        expected.alloc += 1;
279        expected.init += 1;
280        expected.retain += 1;
281        expected.release += 1;
282        helper(&mut expected, false, Some(RcTestObject::new()));
283
284        expected.alloc += 1;
285        expected.init += 1;
286        expected.retain += 1;
287        expected.release += 1;
288        expected.drop += 1;
289        helper(&mut expected, true, Some(RcTestObject::new()));
290    }

421    #[test]
422    #[cfg_attr(
423        any(feature = "catch-all", panic = "abort"),
424        ignore = "panics intentionally"
425    )]
426    fn method_panics() {
427        let cases = [
428            (false, None),
429            (true, None),
430            // Pre-existing parameter passed in.
431            (false, Some(RcTestObject::new())),
432            (true, Some(RcTestObject::new())),
433        ];
434
435        let mut expected = ThreadTestData::current();
436
437        for (panic_after, mut param) in cases {
438            let initially_set = param.is_some();
439
440            autoreleasepool(|_| {
441                let unwindsafe = AssertUnwindSafe(&mut param);
442                let res = catch_unwind(|| {
443                    let param = unwindsafe;
444                    will_panic(Some(param.0), panic_after);
445                });
446                assert!(res.is_err());
447
448                if panic_after {
449                    expected.alloc += 1;
450                    expected.init += 1;
451                    expected.autorelease += 1;
452                }
453                if panic_after || initially_set {
454                    expected.retain += 1;
455                }
456                if initially_set {
457                    expected.release += 1;
458                    if panic_after {
459                        expected.drop += 1;
460                    }
461                }
462                expected.assert_current();
463            });
464
465            if panic_after {
466                expected.release += 1;
467            }
468            expected.assert_current();
469
470            drop(param);
471            if panic_after || initially_set {
472                expected.release += 1;
473                expected.drop += 1;
474            }
475            expected.assert_current();
476        }
477    }
