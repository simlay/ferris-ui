12pub struct App {
13    window: Option<Window>,
14    root_ui_view: Option<Retained<UIView>>,
15    root_view_fn: Box<dyn Fn(EventLoopProxy<GUIEvent>) -> Box<dyn View>>,
16    root_view: Option<Box<dyn View>>,

18}

20impl App {
21    pub fn new(
22        proxy: EventLoopProxy<GUIEvent>,
23        root_view_fn: Box<dyn Fn(EventLoopProxy<GUIEvent>) -> Box<dyn View>>,
24    ) -> Self {
25        Self {
26            window: None,
27            root_ui_view: None,
28            root_view_fn,
29            root_view: None,
30            proxy,
31        }
32    }
33}
34impl ApplicationHandler<GUIEvent> for App {
35    fn user_event(&mut self, _event_loop: &ActiveEventLoop, event: GUIEvent) {
36        //debug!("NEW EVENT: {event:?}");
37        let view = self.root_ui_view.clone().unwrap();
38        let image = take_screenshot(view.bounds().size);
39        if let Some(image) = image {
40            save_image(image);
41        }
42    }
43    fn resumed(&mut self, event_loop: &ActiveEventLoop) {
44        let window = event_loop
45            .create_window(Window::default_attributes())
46            .unwrap();
47
48        if let Ok(handle) = window.window_handle()
49            && let RawWindowHandle::UiKit(handle) = handle.as_raw()
50        {
51            let ui_view = handle.ui_view.as_ptr();
52            let ui_view: Retained<UIView> = unsafe { Retained::retain(ui_view.cast()) }.unwrap();
53            let root_frame = ui_view.frame();
54            let root_view = (self.root_view_fn)(self.proxy.clone());
55            let root_ui_view = root_view.raw_view();
56            root_ui_view.setFrame(root_frame);
57            ui_view.addSubview(root_ui_view.as_ref());
58
59            let bg_color = objc2_ui_kit::UIColor::systemBackgroundColor();
60            objc2_ui_kit::UIColor::labelColor();
61            root_ui_view.setBackgroundColor(Some(&bg_color));
62            self.root_ui_view = Some(ui_view);
63            self.root_view = Some(root_view);
64        }
65
66        self.window = Some(window);
67    }
68
69    fn window_event(&mut self, event_loop: &ActiveEventLoop, _id: WindowId, event: WindowEvent) {
70        match event {
71            WindowEvent::CloseRequested => {
72                debug!("The close button was pressed; stopping");
73                event_loop.exit();
74            }
75            WindowEvent::RedrawRequested => {
76                // Redraw the application.
77                //
78                // It's preferable for applications that do not render continuously to render in
79                // this event rather than in AboutToWait, since rendering in here allows
80                // the program to gracefully handle redraws requested by the OS.
81
82                // Draw.
83
84                // Queue a RedrawRequested event.
85                //
86                // You only need to call this if you've determined that you need to redraw in
87                // applications which do not always need to. Applications that redraw continuously
88                // can render here instead.
89                self.window.as_ref().unwrap().request_redraw();
90            }
91            _e => {
92                //log::trace!("{e:#?}");
93            }
94        }
95    }
