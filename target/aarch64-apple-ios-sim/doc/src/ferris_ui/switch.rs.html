1use objc2::rc::Retained;
2use objc2::{DeclaredClass, MainThreadOnly, define_class, msg_send};
3use objc2_foundation::{MainThreadMarker, NSObject};
4use objc2_ui_kit::{UIControlEvents, UIResponder, UISwitch, UIView};
5use std::cell::RefCell;
6
7use crate::{GUIEvent, View};
8use log::debug;
9use winit::event_loop::EventLoopProxy;
10
11pub struct SwitchState {
12    proxy: EventLoopProxy<GUIEvent>,
13    event_fn: RefCell<Option<Box<dyn Fn(&Switch)>>>,
14}
15
16define_class!(
17    #[unsafe(super(UISwitch, UIView, UIResponder, NSObject))]
18    #[thread_kind = MainThreadOnly]
19    #[name = "FerrisUISwitch"]
20    #[ivars = SwitchState]
21    pub struct Switch;
22
23    impl Switch {
24        #[unsafe(method(toggle))]
25        fn toggle(&self) {
26            let is_on = self.is_on();
27            if let Some(event_fn) = &*self.ivars().event_fn.borrow() {
28                event_fn(self);
29            }
30
31            let _ = self.ivars().proxy.send_event(GUIEvent::SwitchToggle(is_on));
32            debug!("SWITCH TOGGLED!: {is_on}");
33        }
34    }
35);
36
37impl Switch {
38    pub fn new(mtm: MainThreadMarker, proxy: EventLoopProxy<GUIEvent>) -> Retained<Self> {
39        let this = mtm.alloc().set_ivars(SwitchState {
40            proxy,
41            event_fn: RefCell::new(None),
42        });
43        let this: Retained<Self> = unsafe { msg_send![super(this), init] };
44
45        unsafe {
46            this.addTarget_action_forControlEvents(
47                Some(&this),
48                objc2::sel!(toggle),
49                UIControlEvents::ValueChanged,
50            );
51        }
52
53        this
54    }
55    pub fn is_on(&self) -> bool {
56        unsafe { self.isOn() }
57    }
58}
59
60impl View for Switch {
61    fn ui_view(&self) -> Box<&UIView> {
62        Box::new(self.as_ref())
63    }
64    #[cfg(feature = "nightly")]
65    fn with_event_fn(self: Retained<Self>, event_fn: Box<dyn Fn(&Self)>) -> Retained<Self> {
66        *self.ivars().event_fn.borrow_mut() = Some(event_fn);
67        self
68    }
