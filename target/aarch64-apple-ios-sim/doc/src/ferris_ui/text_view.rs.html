9use std::cell::RefCell;

12pub struct TextViewState {
13    delegate: RefCell<Retained<TextFieldDelegate>>,
14    proxy: EventLoopProxy<GUIEvent>,
15    event_fn: RefCell<Option<Box<dyn Fn(&TextView)>>>,
16    place_holder_text: RefCell<Option<String>>,
17}
18
19define_class!(
20    #[unsafe(super(UITextView, UIView, UIResponder, NSObject))]
21    #[thread_kind = MainThreadOnly]
22    #[name = "FerrisUITextView"]
23    #[ivars = TextViewState]
24    pub struct TextView;
25
26    impl TextView { }
27);
28
29define_class!(
30    #[unsafe(super(NSObject))]
31    #[thread_kind = MainThreadOnly]
32    #[name = "FerrisTextViewDelegate"]
33    pub struct TextFieldDelegate;
34
35    unsafe impl NSObjectProtocol for TextFieldDelegate {}
36    unsafe impl UIScrollViewDelegate for TextFieldDelegate {}
37    unsafe impl UITextViewDelegate for TextFieldDelegate {
38        #[unsafe(method(textViewDidChangeSelection:))]
39        fn did_change_selection(&self, _sender: &TextView) {}
40        #[unsafe(method(textViewDidBeginEditing:))]
41        fn did_begin_editing(&self, sender: &TextView) {
42            sender.began_editing();
43        }
44
45        #[unsafe(method(textViewDidEndEditing:))]
46        fn did_end_editing(&self, sender: &TextView) {
47            sender.ended_editing();
48        }
49
50        #[unsafe(method(textViewDidChange:))]
51        fn did_change(&self, sender: &TextView) {
52            sender.text_changed();
53        }
54    }
55);

57impl TextView {
58    pub fn new(mtm: MainThreadMarker, proxy: EventLoopProxy<GUIEvent>) -> Retained<Self> {
59        let delegate: Retained<TextFieldDelegate> =
60            unsafe { objc2::msg_send![TextFieldDelegate::alloc(mtm), init] };
61        let this = Self::alloc(mtm).set_ivars(TextViewState {
62            delegate: RefCell::new(delegate),
63            proxy,
64            event_fn: RefCell::new(None),
65            place_holder_text: RefCell::new(None),
66        });
67        let this: Retained<TextView> = unsafe { msg_send![super(this), init] };
68        {
69            let delegate = this.ivars().delegate.borrow();
70            unsafe {
71                this.setDelegate(Some(ProtocolObject::from_ref(&*delegate.clone())));
72            }
73        }
74
75        this
76    }
77
78    pub fn get_text(&self) -> String {
79        self.text().to_string()
80    }
81
82    fn began_editing(&self) {
83        self.setText(None);
84        self.setTextColor(Some(&UIColor::blackColor()));
85    }
86
87    fn ended_editing(&self) {
88        let place_holder_text = self
89            .ivars()
90            .place_holder_text
91            .borrow()
92            .clone()
93            .unwrap_or_default();
94        if !place_holder_text.is_empty() {
95            self.setText(Some(&NSString::from_str(place_holder_text.as_str())));
96            self.setTextColor(Some(&UIColor::grayColor()));
97        }
98    }
99
100    fn text_changed(&self) {
101        let text = self.get_text();
102
103        let _ = self.ivars().proxy.send_event(GUIEvent::Text(text.clone()));
104
105        if let Some(event_fn) = &*self.ivars().event_fn.borrow() {
106            event_fn(self);
107        }
108    }
109
110    pub fn with_place_holder_text(self: Retained<Self>, place_holder: String) -> Retained<Self> {
111        *self.ivars().place_holder_text.borrow_mut() = Some(place_holder);
112        self
113    }
114}
115
116impl View for TextView {
117    fn raw_view(&self) -> Box<&UIView> {
118        self.ended_editing();
119        Box::new(self.as_ref())
120    }
121    #[cfg(feature = "nightly")]
122    fn with_event_fn(self: Retained<Self>, event_fn: Box<dyn Fn(&Self)>) -> Retained<Self> {
123        *self.ivars().event_fn.borrow_mut() = Some(event_fn);
124        self
125    }
