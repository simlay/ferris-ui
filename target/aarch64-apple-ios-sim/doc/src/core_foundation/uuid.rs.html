10//! Core Foundation UUID objects.
11
12#[cfg(feature = "with-uuid")]
13extern crate uuid;
14
15use core_foundation_sys::base::kCFAllocatorDefault;
16pub use core_foundation_sys::uuid::*;
17
18use crate::base::TCFType;
19
20#[cfg(feature = "with-uuid")]
21use self::uuid::Uuid;
22
23declare_TCFType! {
24    /// A UUID.
25    CFUUID, CFUUIDRef
26}
27impl_TCFType!(CFUUID, CFUUIDRef, CFUUIDGetTypeID);
28impl_CFTypeDescription!(CFUUID);
29
30impl CFUUID {
31    #[inline]
32    pub fn new() -> CFUUID {
33        unsafe {
34            let uuid_ref = CFUUIDCreate(kCFAllocatorDefault);
35            TCFType::wrap_under_create_rule(uuid_ref)
36        }
37    }
38}
39
40impl Default for CFUUID {
41    fn default() -> Self {
42        Self::new()
43    }
44}
45
46#[cfg(feature = "with-uuid")]
47impl From<CFUUID> for Uuid {
48    fn from(val: CFUUID) -> Self {
49        let b = unsafe { CFUUIDGetUUIDBytes(val.0) };
50        let bytes = [
51            b.byte0, b.byte1, b.byte2, b.byte3, b.byte4, b.byte5, b.byte6, b.byte7, b.byte8,
52            b.byte9, b.byte10, b.byte11, b.byte12, b.byte13, b.byte14, b.byte15,
53        ];
54        Uuid::from_bytes(&bytes).unwrap()
55    }
56}
57
58#[cfg(feature = "with-uuid")]
59impl From<Uuid> for CFUUID {
60    fn from(uuid: Uuid) -> CFUUID {
61        let b = uuid.as_bytes();
62        let bytes = CFUUIDBytes {
63            byte0: b[0],
64            byte1: b[1],
65            byte2: b[2],
66            byte3: b[3],
67            byte4: b[4],
68            byte5: b[5],
69            byte6: b[6],
70            byte7: b[7],
71            byte8: b[8],
72            byte9: b[9],
73            byte10: b[10],
74            byte11: b[11],
75            byte12: b[12],
76            byte13: b[13],
77            byte14: b[14],
78            byte15: b[15],
79        };
80        unsafe {
81            let uuid_ref = CFUUIDCreateFromUUIDBytes(kCFAllocatorDefault, bytes);
82            TCFType::wrap_under_create_rule(uuid_ref)
83        }
84    }
85}

87#[cfg(test)]
88#[cfg(feature = "with-uuid")]
89mod test {
90    use super::CFUUID;
91    use uuid::Uuid;
92
93    #[test]
94    fn uuid_conversion() {
95        let cf_uuid = CFUUID::new();
96        let uuid: Uuid = cf_uuid.clone().into();
97        let converted = CFUUID::from(uuid);
98        assert_eq!(cf_uuid, converted);
99    }
