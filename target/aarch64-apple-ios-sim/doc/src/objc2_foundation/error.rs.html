10/// Creation methods.
11impl NSError {
12    /// Construct a new [`NSError`] with the given code in the given domain.
13    #[cfg(feature = "NSDictionary")]
14    #[cfg(feature = "NSString")]
15    pub fn new(
16        code: objc2::ffi::NSInteger,
17        domain: &crate::NSErrorDomain,
18    ) -> objc2::rc::Retained<Self> {
19        use objc2::ClassType;
20        // SAFETY: `domain` and `user_info` are copied to the error object, so
21        // even if the `&NSString` came from a `&mut NSMutableString`, we're
22        // still good!
23        unsafe { Self::initWithDomain_code_userInfo(Self::alloc(), domain, code, None) }
24    }
25}

41impl fmt::Debug for NSError {
42    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
43        f.debug_struct("NSError")
44            .field("domain", &self.domain())
45            .field("code", &self.code())
46            .field("user_info", &self.userInfo())
47            .finish()
48    }
49}
