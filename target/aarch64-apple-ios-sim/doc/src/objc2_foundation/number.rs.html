1//! Note that due to limitations in Objective-C type encodings, it is not
2//! possible to distinguish between an `NSNumber` created from [`bool`],
3//! and one created from an [`i8`]/[`u8`]. You should use the getter
4//! methods that fit your use-case instead!
5//!
6//! This does not implement [`Eq`] nor [`Ord`], since it may contain a
7//! floating point value. Beware that the implementation of [`PartialEq`]
8//! and [`PartialOrd`] does not properly handle NaNs either. Compare
9//! [`NSNumber::encoding`] with [`Encoding::Float`] or
10//! [`Encoding::Double`], and use [`NSNumber::as_f32`] or
11//! [`NSNumber::as_f64`] to get the desired floating point value directly.
12//!
13//! TODO: Once we have better CoreFoundation support, use that to create
14//! efficient static numbers. See:
15//! <https://github.com/nvzqz/fruity/blob/811d7787495eaaee6bc39d372004e5d96ef9f49b/src/foundation/ns_number.rs#L328-L401>
16//!
17//! (Same goes for `NSNull`).

32macro_rules! def_new_fn {
33    {$(
34        $(#[$($m:meta)*])*
35        ($fn_name:ident($fn_inp:ty); $method_name:ident),
36    )*} => {$(
37        $(#[$($m)*])*
38        pub fn $fn_name(val: $fn_inp) -> Retained<Self> {
39            Self::$method_name(val as _)
40        }
41    )*}
42}

47        (new_bool(bool); numberWithBool),
48        (new_i8(i8); numberWithChar),
49        (new_u8(u8); numberWithUnsignedChar),
50        (new_i16(i16); numberWithShort),
51        (new_u16(u16); numberWithUnsignedShort),
52        (new_i32(i32); numberWithInt),
53        (new_u32(u32); numberWithUnsignedInt),
54        (new_i64(i64); numberWithLongLong),
55        (new_u64(u64); numberWithUnsignedLongLong),
56        (new_isize(isize); numberWithInteger),
57        (new_usize(usize); numberWithUnsignedInteger),
58        (new_f32(f32); numberWithFloat),
59        (new_f64(f64); numberWithDouble),
60    }
61
62    #[inline]
63    #[cfg(feature = "NSGeometry")]
64    pub fn new_cgfloat(val: crate::Foundation::CGFloat) -> Retained<Self> {
65        #[cfg(target_pointer_width = "64")]
66        {
67            Self::new_f64(val)
68        }
69        #[cfg(not(target_pointer_width = "64"))]
70        {
71            Self::new_f32(val)
72        }
73    }
74}

76macro_rules! def_get_fn {
77    {$(
78        $(#[$($m:meta)*])*
79        ($fn_name:ident -> $fn_ret:ty; $method_name:ident),
80    )*} => {$(
81        $(#[$($m)*])*
82        pub fn $fn_name(&self) -> $fn_ret {
83            self.$method_name() as _
84        }
85    )*}
86}

119    /// The Objective-C encoding of this `NSNumber`.
120    ///
121    /// This is guaranteed to return one of:
122    /// - [`Encoding::Char`]
123    /// - [`Encoding::UChar`]
124    /// - [`Encoding::Short`]
125    /// - [`Encoding::UShort`]
126    /// - [`Encoding::Int`]
127    /// - [`Encoding::UInt`]
128    /// - [`Encoding::Long`]
129    /// - [`Encoding::ULong`]
130    /// - [`Encoding::LongLong`]
131    /// - [`Encoding::ULongLong`]
132    /// - [`Encoding::Float`]
133    /// - [`Encoding::Double`]
134    ///
135    ///
136    /// # Examples
137    ///
138    /// Convert an `NSNumber` to/from an enumeration describing the different
139    /// number properties.
140    ///
141    /// ```
142    /// use objc2_foundation::NSNumber;
143    /// use objc2::encode::Encoding;
144    /// use objc2::rc::Retained;
145    ///
146    /// // Note: `bool` would convert to either `Signed` or `Unsigned`,
147    /// // depending on platform
148    /// #[derive(Copy, Clone)]
149    /// pub enum Number {
150    ///     Signed(i64),
151    ///     Unsigned(u64),
152    ///     Floating(f64),
153    /// }
154    ///
155    /// impl Number {
156    ///     fn into_nsnumber(self) -> Retained<NSNumber> {
157    ///         match self {
158    ///             Self::Signed(val) => NSNumber::new_i64(val),
159    ///             Self::Unsigned(val) => NSNumber::new_u64(val),
160    ///             Self::Floating(val) => NSNumber::new_f64(val),
161    ///         }
162    ///     }
163    /// }
164    ///
165    /// impl From<&NSNumber> for Number {
166    ///     fn from(n: &NSNumber) -> Self {
167    ///         match n.encoding() {
168    ///             Encoding::Char
169    ///             | Encoding::Short
170    ///             | Encoding::Int
171    ///             | Encoding::Long
172    ///             | Encoding::LongLong => Self::Signed(n.as_i64()),
173    ///             Encoding::UChar
174    ///             | Encoding::UShort
175    ///             | Encoding::UInt
176    ///             | Encoding::ULong
177    ///             | Encoding::ULongLong => Self::Unsigned(n.as_u64()),
178    ///             Encoding::Float
179    ///             | Encoding::Double => Self::Floating(n.as_f64()),
180    ///             _ => unreachable!(),
181    ///         }
182    ///     }
183    /// }
184    /// ```
185    pub fn encoding(&self) -> Encoding {
186        // Use NSValue::encoding
187        let enc = (**self)
188            .encoding()
189            .expect("NSNumber must have an encoding!");
190
191        // Guaranteed under "Subclassing Notes"
192        // <https://developer.apple.com/documentation/foundation/nsnumber?language=objc#1776615>
193        match enc {
194            "c" => Encoding::Char,
195            "C" => Encoding::UChar,
196            "s" => Encoding::Short,
197            "S" => Encoding::UShort,
198            "i" => Encoding::Int,
199            "I" => Encoding::UInt,
200            "l" => Encoding::Long,
201            "L" => Encoding::ULong,
202            "q" => Encoding::LongLong,
203            "Q" => Encoding::ULongLong,
204            "f" => Encoding::Float,
205            "d" => Encoding::Double,
206            _ => unreachable!("invalid encoding for NSNumber"),
207        }
208    }
209}

218/// Beware: This uses the Objective-C method "isEqualToNumber:", which has
219/// different floating point NaN semantics than Rust!
220impl PartialEq for NSNumber {
221    #[doc(alias = "isEqualToNumber:")]
222    fn eq(&self, other: &Self) -> bool {
223        // Use isEqualToNumber: instaed of isEqual: since it is faster
224        self.isEqualToNumber(other)
225    }
226}

228/// Beware: This uses the Objective-C method "isEqualToNumber:", which has
229/// different floating point NaN semantics than Rust!
230//
231// This is valid since the following pass (i.e. Objective-C says that two NaNs
232// are equal):
233// ```
234// let nan = NSNumber::from_f32(f32::NAN);
235// assert_eq!(nan, nan);
236// ```
237impl Eq for NSNumber {}

239/// Beware: This uses the Objective-C method "compare:", which has different
240/// floating point NaN semantics than Rust!
241#[cfg(feature = "NSObjCRuntime")]
242impl PartialOrd for NSNumber {
243    #[doc(alias = "compare:")]
244    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
245        Some(self.cmp(other))
246    }
247}

249/// Beware: This uses the Objective-C method "compare:", which has different
250/// floating point NaN semantics than Rust!
251#[cfg(feature = "NSObjCRuntime")]
252impl Ord for NSNumber {
253    #[doc(alias = "compare:")]
254    fn cmp(&self, other: &Self) -> Ordering {
255        // Use Objective-C semantics for comparison
256        self.compare(other).into()
257    }
258}
