13/// Swift and Objective-C both have a stack buffer size of 16, so we do that
14/// as well.
15///
16/// TODO: Consider lowering this a bit, since the most common type of
17/// enumeration (e.g. NSArray) doesn't use the buffer:
18/// [CFArray's NSFastEnumeration implementation](https://github.com/apple-oss-distributions/CF/blob/dc54c6bb1c1e5e0b9486c1d26dd5bef110b20bf3/CFArray.c#L618-L642)
19const BUF_SIZE: usize = 16;

21/// Track mutation mistakes when debug assertions are enabled (they should
22/// be made impossible by Rust at compile-time, so hence we don't need to
23/// track them in release mode).
24///
25/// This is set to `None` initially, but later loaded to `Some(_)` after
26/// the first enumeration.
27type MutationState = Option<c_ulong>;

29/// Helper type for doing fast enumeration.
30///
31/// See the following other implementations of this:
32/// - [Swift](https://github.com/apple/swift-corelibs-foundation/blob/2d23cf3dc07951ed2b988608d08d7a54cc53b26e/Darwin/Foundation-swiftoverlay/NSFastEnumeration.swift#L23)
33/// - [Clang](https://github.com/llvm/llvm-project/blob/28d85d207fc37b5593c17a25f687c91b7afda5b4/clang/lib/Frontend/Rewrite/RewriteModernObjC.cpp#L1653-L1850)
34#[derive(Debug, PartialEq)]
35struct FastEnumeratorHelper {
36    state: NSFastEnumerationState,
37    buf: [*mut AnyObject; BUF_SIZE],
38    // TODO: We could possibly optimize things a bit by doing
39    // `itemsPtr.add(1); items_count -= 1;` on every loop, instead of storing
40    // `current_item` - but it's not really defined whether we're allowed to
41    // mutate `itemsPtr`? All of the GNUStep code and the open sourced Apple
42    // code I've found would work under this scheme, and from a cursory GitHub
43    // search I only found the following that assume that `itemsPtr` is stable
44    // across invocations:
45    // https://github.com/iodefog/VipVideo/blob/52243b893f0004a6880f13c33104e2394b56b7f1/VipVideo/Helper/JSONKit.m#L759-L769
46    //
47    // Though they also assume that the given stack buffer has a stable
48    // address, which is something that would be prohibitively expensive to
49    // ensure, so don't think we should consider that one.
50    current_item: usize,
51    items_count: usize,
52}
53
54// SAFETY: Neither `FastEnumeratorHelper`, nor the inner enumeration state,
55// need to be bound to any specific thread (at least, we can generally assume
56// this for the enumerable types in Foundation - but `NSEnumerator` won't be
57// `Send`, and neither will its iterator types).
58unsafe impl Send for FastEnumeratorHelper {}

60// SAFETY: `FastEnumeratorHelper` is only mutable behind `&mut`, and as such
61// is safe to share with other threads.
62unsafe impl Sync for FastEnumeratorHelper {}

70#[cfg(not(feature = "unstable-mutation-return-null"))]
71#[cfg_attr(debug_assertions, track_caller)]
72fn items_ptr_null() -> ! {
73    panic!("`itemsPtr` was NULL, likely due to mutation during iteration");
74}

82impl FastEnumeratorHelper {
83    #[inline]
84    fn new() -> Self {
85        Self {
86            state: NSFastEnumerationState {
87                state: 0,
88                itemsPtr: ptr::null_mut(),
89                mutationsPtr: ptr::null_mut(),
90                extra: [0; 5],
91            },
92            buf: [ptr::null_mut(); BUF_SIZE],
93            current_item: 0,
94            items_count: 0,
95        }
96    }
97
98    #[inline]
99    const fn remaining_items_at_least(&self) -> usize {
100        self.items_count - self.current_item
101    }
102
103    /// Load the next array of items into the enumeration state.
104    ///
105    ///
106    /// # Safety
107    ///
108    /// The collection must be the same on each call.
109    #[inline]
110    unsafe fn load_next_items(&mut self, collection: &ProtocolObject<dyn NSFastEnumeration>) {
111        // SAFETY: The ptr comes from a slice, which is always non-null.
112        //
113        // The callee may choose to use this or not, see:
114        // <https://www.mikeash.com/pyblog/friday-qa-2010-04-16-implementing-fast-enumeration.html>
115        let buf_ptr = unsafe { NonNull::new_unchecked(self.buf.as_mut_ptr()) };
116        // SAFETY:
117        // - The state is mutable, and we don't touch the `state` and
118        //   `extra` fields.
119        // - The buffer is mutable and its length is correctly set.
120        // - The collection and state are guaranteed by the caller to match.
121        self.items_count = unsafe {
122            collection.countByEnumeratingWithState_objects_count(
123                NonNull::from(&mut self.state),
124                buf_ptr,
125                self.buf.len(),
126            )
127        };
128
129        // For unwind safety, we only set this _after_ the message send has
130        // completed, otherwise, if it unwinded, upon iterating again we might
131        // invalidly assume that the buffer was ready.
132        self.current_item = 0;
133    }
134
135    /// Get the next item from the given collection.
136    ///
137    /// We use a `ProtocolObject` instead of a generic, so that there is only
138    /// one instance of this function in the compilation unit (should improve
139    /// compilation speed).
140    ///
141    /// If `mutations_state` is `Some`, mutation tracking is activated.
142    ///
143    ///
144    /// # Safety
145    ///
146    /// The collection must be the same on each call.
147    #[inline]
148    #[track_caller]
149    unsafe fn next_from(
150        &mut self,
151        collection: &ProtocolObject<dyn NSFastEnumeration>,
152        mutations_state: Option<&mut MutationState>,
153    ) -> Option<NonNull<AnyObject>> {
154        // If we've exhausted the current array of items.
155        if self.current_item >= self.items_count {
156            // Get the next array of items.
157            //
158            // SAFETY: Upheld by caller.
159            unsafe { self.load_next_items(collection) };
160
161            // If the next array was also empty.
162            if self.items_count == 0 {
163                // We are done enumerating.
164                return None;
165            }
166
167            if mutations_state.is_some() && self.state.itemsPtr.is_null() {
168                #[cfg(feature = "unstable-mutation-return-null")]
169                return Some(unsafe { kCFNull });
170                #[cfg(not(feature = "unstable-mutation-return-null"))]
171                items_ptr_null();
172            }
173        }
174
175        if let Some(mutations_state) = mutations_state {
176            // If the mutation ptr is not set, we do nothing.
177            if let Some(ptr) = NonNull::new(self.state.mutationsPtr) {
178                // SAFETY:
179                // - The pointer is not NULL.
180                //
181                // - The enumerator is expected to give back a dereferenceable
182                //   pointer, that is alive for as long as the collection is
183                //   alive.
184                //
185                //   Note that iterating past the first returned `None` is not
186                //   tested by most Objective-C implementations, so it may
187                //   deallocate the mutations ptr in that case?
188                //
189                // - The enumeration should not be modifiable across threads,
190                //   so neither will this pointer be accessed from different
191                //   threads.
192                //
193                //   Note that this assumption is relatively likely to be
194                //   violated, but if that is the case, the program already
195                //   has UB, so then it is better that we detect it.
196                //
197                // - The value is an integer, so is always initialized.
198                //
199                //
200                // We do an unaligned read here since we have no guarantees
201                // about this pointer, and efficiency doesn't really matter.
202                let new_state = unsafe { ptr.as_ptr().read_unaligned() };
203                match *mutations_state {
204                    // On the first iteration, initialize the mutation state
205                    None => {
206                        *mutations_state = Some(new_state);
207                    }
208                    // On subsequent iterations, verify that the state hasn't
209                    // changed.
210                    Some(current_state) => {
211                        if current_state != new_state {
212                            #[cfg(feature = "unstable-mutation-return-null")]
213                            return Some(unsafe { kCFNull });
214                            #[cfg(not(feature = "unstable-mutation-return-null"))]
215                            mutation_detected();
216                        }
217                    }
218                }
219            }
220        }
221
222        // Compute a pointer to the current item.
223        //
224        // SAFETY: The index is checked above to be in bounds of the returned
225        // array, and the item pointer is checked to not be NULL.
226        let ptr = unsafe { self.state.itemsPtr.add(self.current_item) };
227        // Move to the next item.
228        self.current_item += 1;
229        // And read the current item.
230        //
231        // SAFETY:
232        // - The pointer is not NULL, it is guaranteed to have been set
233        //   by `countByEnumeratingWithState:objects:count:`.
234        // - The pointer is within the bounds of the returned array.
235        //
236        // Note: GNUStep may sometimes return unaligned pointers. TODO: Should
237        // we go back to using `read` on Apple systems for better performance?
238        let obj = unsafe { ptr.read_unaligned() };
239        // SAFETY: The returned array contains no NULL pointers.
240        Some(unsafe { NonNull::new_unchecked(obj) })
241    }
242}

249/// Internal helper trait to figure out the output type of something that
250/// implements `NSFastEnumeration`.
251///
252///
253/// # Safety
254///
255/// The associated `Item` type must be the type that is used in Objective-C's
256/// `for (type elem in collection) { stmts; }` enumeration, and the collection
257/// itself must not contain any lifetime parameter.
258pub(crate) unsafe trait FastEnumerationHelper: Message + NSFastEnumeration {

260
261    fn maybe_len(&self) -> Option<usize>;

264// Iterator implementations we _can't_ do:
265//
266//
267// # `ExactSizeIterator`
268//
269// This could probably be implemented correctly for the concrete class
270// `NSArray`, but not for arbitrary `NSArray` subclasses, which are always be
271// valid to convert to `NSArray`.
272//
273//
274// # `FusedIterator`
275//
276// Although most fast enumerators are probably fused, this is not guaranteed
277// by the documentation (actually, some may even use the last iteration as a
278// cleanup step, and do it twice, which we should perhaps protect against?).
279//
280//
281// # `DoubleEndedIterator`
282//
283// `NSArray` has `reverseObjectEnumerator`, but this is a separate object from
284// `objectEnumerator`, and as such it is not possible to enumerate it in the
285// backwards direction.
286
287// Explicit lifetime bound on `C` to future-proof against possible soundness
288// mistakes in the future.
289//
290// TODO: Consider adding `#[repr(C)]`, to allow LLVM to perform better
291// zero-initialization.
292#[derive(Debug, PartialEq)]
293pub(crate) struct IterUnchecked<'a, C: ?Sized + 'a> {
294    helper: FastEnumeratorHelper,
295    /// 'a and C are covariant.
296    collection: &'a C,
297    #[cfg(debug_assertions)]
298    mutations_state: MutationState,
299}

312impl<'a, C: FastEnumerationHelper> Iterator for IterUnchecked<'a, C> {
313    type Item = &'a C::Item;
314
315    #[inline]
316    #[track_caller]
317    fn next(&mut self) -> Option<&'a C::Item> {
318        #[cfg(debug_assertions)]
319        let mutations_state = Some(&mut self.mutations_state);
320        #[cfg(not(debug_assertions))]
321        let mutations_state = None;
322        // SAFETY: The collection is the same on each iteration.
323        //
324        // Passing the collection as immutable here is safe since it isn't
325        // mutated itself, and it is `UnsafeCell` anyhow.
326        let obj = unsafe {
327            self.helper
328                .next_from(ProtocolObject::from_ref(self.collection), mutations_state)?
329        };
330        // SAFETY: The lifetime is bound to the collection, and the type is
331        // correct.
332        //
333        // [Enumeration variables are externally retained][enum-retained], so
334        // it is okay to return a reference instead of retaining here.
335        //
336        // [enum-retained]: https://clang.llvm.org/docs/AutomaticReferenceCounting.html#fast-enumeration-iteration-variables
337        Some(unsafe { obj.cast::<C::Item>().as_ref() })
338    }
339
340    #[inline]
341    fn size_hint(&self) -> (usize, Option<usize>) {
342        (
343            self.helper.remaining_items_at_least(),
344            self.collection.maybe_len(),
345        )
346    }
347}

391// Note: I first considered having `IntoIter<T>` instead of `IntoIter<C>`, but
392// that would be unsound for `NSDictionary<K, V>`, since if the `V` has a
393// lifetime, that lifetime would have been erased (so we'd have to add an
394// `V: 'static` bound, which is easy to forget).

397// as well as allowing us to implement `FusedIterator` and `ExactSizeIterator`
398// when possible.
