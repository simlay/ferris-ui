18impl<T: Message> NSArray<T> {
19    pub fn from_vec(mut vec: Vec<Retained<T>>) -> Retained<Self> {
20        // We intentionally extract the length before we access the
21        // pointer as mutable, to not invalidate that mutable pointer.
22        let len = vec.len();
23        let ptr = util::retained_ptr_cast(vec.as_mut_ptr());
24        // SAFETY: We've consumed the `Retained<T>`s, which means that we can
25        // now safely take ownership (even if `T` is mutable).
26        unsafe { Self::initWithObjects_count(Self::alloc(), ptr, len) }
27        // The drop of `Vec` here would invalidate our mutable pointer,
28        // except for the fact that we're using `UnsafeCell` in `AnyObject`.
29    }
30
31    pub fn from_id_slice(slice: &[Retained<T>]) -> Retained<Self>
32    where
33        T: IsIdCloneable,
34    {
35        let len = slice.len();
36        let ptr = util::retained_ptr_cast_const(slice.as_ptr());
37        // SAFETY: Because of the `T: IsIdCloneable` bound, and since we
38        // take `&[Retained<T>]` (effectively `&Retained<T>`), we are allowed to give
39        // the slice to Objective-C, which will retain it internally.
40        //
41        // Faster version of:
42        //     Self::from_vec(slice.iter().map(|obj| obj.clone()).collect())
43        unsafe { Self::initWithObjects_count(Self::alloc(), ptr, len) }
44    }
45
46    pub fn from_slice(slice: &[&T]) -> Retained<Self>
47    where
48        T: IsRetainable,
49    {
50        let len = slice.len();
51        let ptr = util::ref_ptr_cast_const(slice.as_ptr());
52        // SAFETY: Because of the `T: IsRetainable` bound, we are allowed
53        // to give the slice to Objective-C, which will retain it
54        // internally.
55        //
56        // Faster version of:
57        //     Self::from_vec(slice.iter().map(|obj| obj.retain()).collect())
58        unsafe { Self::initWithObjects_count(Self::alloc(), ptr, len) }
59    }
60
61    #[doc(alias = "getObjects:range:")]
62    #[cfg(feature = "NSRange")]
63    pub fn to_vec(&self) -> Vec<&T> {
64        // SAFETY: The range is know to be in bounds
65        unsafe { self.objects_in_range_unchecked(0..self.len()) }
66    }
67
68    #[doc(alias = "getObjects:range:")]
69    #[cfg(feature = "NSRange")]
70    pub fn to_vec_retained(&self) -> Vec<Retained<T>>
71    where
72        T: IsIdCloneable,
73    {
74        // SAFETY: The objects are stored in the array
75        self.to_vec()
76            .into_iter()
77            .map(|obj| unsafe { util::collection_retain(obj) })
78            .collect()
79    }
80
81    // `fn into_vec(Retained<NSArray>) -> Vec<Retained<T>>` would not be safe, since
82    // the array itself is unconditionally `IsIdCloneable`, even when
83    // containing mutable elements, and hence we would be able to
84    // duplicate those.
85}

225impl<T: Message> NSArray<T> {
226    #[cfg(feature = "NSRange")]
227    unsafe fn objects_in_range_unchecked(&self, range: Range<usize>) -> Vec<&T> {
228        let range = crate::Foundation::NSRange::from(range);
229        let mut vec: Vec<core::ptr::NonNull<T>> = Vec::with_capacity(range.length);
230        unsafe {
231            self.getObjects_range(core::ptr::NonNull::new(vec.as_mut_ptr()).unwrap(), range);
232            vec.set_len(range.length);
233            core::mem::transmute(vec)
234        }
235    }
236
237    #[doc(alias = "getObjects:range:")]
238    #[cfg(feature = "NSRange")]
239    pub fn objects_in_range(&self, range: Range<usize>) -> Option<Vec<&T>> {
240        if range.end > self.len() {
241            return None;
242        }
243        // SAFETY: Just checked that the range is in bounds
244        Some(unsafe { self.objects_in_range_unchecked(range) })
245    }
246}

248impl<T: Message> NSMutableArray<T> {
249    #[doc(alias = "addObject:")]
250    pub fn push(&mut self, obj: Retained<T>) {
251        // SAFETY: We've consumed ownership of the object.
252        unsafe { self.addObject(&obj) }
253    }
254
255    #[doc(alias = "insertObject:atIndex:")]
256    pub fn insert(&mut self, index: usize, obj: Retained<T>) {
257        // TODO: Replace this check with catching the thrown NSRangeException
258        let len = self.len();
259        if index < len {
260            // SAFETY: We've consumed ownership of the object, and the
261            // index is checked to be in bounds.
262            unsafe { self.insertObject_atIndex(&obj, index) }
263        } else {
264            panic!(
265                "insertion index (is {}) should be <= len (is {})",
266                index, len
267            );
268        }
269    }
270
271    #[doc(alias = "replaceObjectAtIndex:withObject:")]
272    pub fn replace(&mut self, index: usize, obj: Retained<T>) -> Result<Retained<T>, Retained<T>> {
273        if let Some(old_obj) = self.get(index) {
274            // SAFETY: We remove the object from the array below.
275            let old_obj = unsafe { util::mutable_collection_retain_removed(old_obj) };
276            // SAFETY: The index is checked to be in bounds, and we've
277            // consumed ownership of the new object.
278            unsafe { self.replaceObjectAtIndex_withObject(index, &obj) };
279            Ok(old_obj)
280        } else {
281            Err(obj)
282        }
283    }
284
285    #[doc(alias = "removeObjectAtIndex:")]
286    pub fn remove(&mut self, index: usize) -> Option<Retained<T>> {
287        let obj = self.get(index)?;
288        // SAFETY: We remove the object from the array below.
289        let obj = unsafe { util::mutable_collection_retain_removed(obj) };
290        // SAFETY: The index is checked to be in bounds.
291        unsafe { self.removeObjectAtIndex(index) };
292        Some(obj)
293    }
294
295    #[doc(alias = "removeLastObject")]
296    pub fn pop(&mut self) -> Option<Retained<T>> {
297        let obj = self.last()?;
298        // SAFETY: We remove the object from the array below.
299        let obj = unsafe { util::mutable_collection_retain_removed(obj) };
300        // SAFETY: Just checked that there is an object.
301        unsafe { self.removeLastObject() };
302        Some(obj)
303    }

305    #[cfg(feature = "NSObjCRuntime")]
306    #[doc(alias = "sortUsingFunction:context:")]
307    pub fn sort_by<F: FnMut(&T, &T) -> core::cmp::Ordering>(&mut self, compare: F) {
308        // TODO: "C-unwind"
309        unsafe extern "C" fn compare_with_closure<T, F: FnMut(&T, &T) -> core::cmp::Ordering>(
310            obj1: core::ptr::NonNull<T>,
311            obj2: core::ptr::NonNull<T>,
312            context: *mut std::os::raw::c_void,
313        ) -> isize {
314            let context: *mut F = context.cast();
315            // Bring back a reference to the closure.
316            // Guaranteed to be unique, we gave `sortUsingFunction` unique is
317            // ownership, and that method only runs one function at a time.
318            let closure: &mut F = unsafe { context.as_mut().unwrap_unchecked() };
319
320            // SAFETY: The objects are guaranteed to be valid
321            let (obj1, obj2) = unsafe { (obj1.as_ref(), obj2.as_ref()) };
322
323            crate::Foundation::NSComparisonResult::from((*closure)(obj1, obj2)) as _
324        }
325
326        // Create function pointer
327        let f: unsafe extern "C" fn(_, _, _) -> _ = compare_with_closure::<T, F>;
328
329        // Grab a type-erased pointer to the closure (a pointer to stack).
330        let mut closure = compare;
331        let context: *mut F = &mut closure;
332
333        unsafe { self.sortUsingFunction_context(f, context.cast()) };
334        // Keep the closure alive until the function has run.
335        drop(closure);
336    }
