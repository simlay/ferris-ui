14/// The headers describe `initWithUUIDBytes:` and `getUUIDBytes:` as
15/// taking `uuid_t`, but something fishy is going on, in reality they
16/// expect a reference to these!
17///
18/// Hence we create this newtype to change the encoding.
19#[repr(transparent)]
20struct UuidBytes([u8; 16]);
21
22unsafe impl RefEncode for UuidBytes {
23    // Encoding depends on Foundation version, we hack it for now.
24    const ENCODING_REF: Encoding = if cfg!(target_arch = "aarch64") {
25        Encoding::String
26    } else {
27        Encoding::Array(16, &u8::ENCODING)
28    };
29}
30
31extern_methods!(
32    unsafe impl NSUUID {
33        #[method_id(initWithUUIDBytes:)]
34        fn initWithUUIDBytes(this: Allocated<Self>, bytes: &UuidBytes) -> Retained<Self>;
35
36        #[method(getUUIDBytes:)]
37        fn getUUIDBytes(&self, bytes: &mut UuidBytes);
38    }
39);
40
41impl NSUUID {
42    /// The 'nil UUID'.
43    pub fn nil() -> Retained<Self> {
44        Self::from_bytes([0; 16])
45    }
46
47    /// Create a new `NSUUID` from the given bytes.
48    ///
49    ///
50    /// # Example
51    ///
52    /// Create a new `NSUUID` from the `uuid` crate.
53    ///
54    /// ```ignore
55    /// use uuid::Uuid;
56    /// use objc2_foundation::NSUUID;
57    ///
58    /// let uuid: Uuid;
59    /// # uuid = todo!();
60    /// let obj = NSUUID::from_bytes(uuid.into_bytes());
61    /// assert_eq!(obj.as_bytes(), uuid.into_bytes());
62    /// ```
63    pub fn from_bytes(bytes: [u8; 16]) -> Retained<Self> {
64        let bytes = UuidBytes(bytes);
65        Self::initWithUUIDBytes(Self::alloc(), &bytes)
66    }
67
68    #[cfg(feature = "NSString")]
69    pub fn from_string(string: &crate::NSString) -> Option<Retained<Self>> {
70        Self::initWithUUIDString(Self::alloc(), string)
71    }
72
73    pub fn as_bytes(&self) -> [u8; 16] {
74        let mut bytes = UuidBytes([0; 16]);
75        self.getUUIDBytes(&mut bytes);
76        bytes.0
77    }
78}
