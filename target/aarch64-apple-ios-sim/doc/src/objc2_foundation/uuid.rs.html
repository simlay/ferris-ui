13impl NSUUID {
14    /// The 'nil UUID'.
15    pub fn nil() -> Retained<Self> {
16        Self::from_bytes([0; 16])
17    }
18
19    /// Create a new `NSUUID` from the given bytes.
20    ///
21    ///
22    /// # Example
23    ///
24    /// Create a new `NSUUID` from the `uuid` crate.
25    ///
26    /// ```ignore
27    /// use uuid::Uuid;
28    /// use objc2_foundation::NSUUID;
29    ///
30    /// let uuid: Uuid;
31    /// # uuid = todo!();
32    /// let obj = NSUUID::from_bytes(uuid.into_bytes());
33    /// assert_eq!(obj.as_bytes(), uuid.into_bytes());
34    /// ```
35    pub fn from_bytes(bytes: [u8; 16]) -> Retained<Self> {
36        Self::initWithUUIDBytes(Self::alloc(), &bytes)
37    }
38
39    #[cfg(feature = "NSString")]
40    pub fn from_string(string: &crate::NSString) -> Option<Retained<Self>> {
41        Self::initWithUUIDString(Self::alloc(), string)
42    }
43
44    /// Convert the UUID to an array.
45    pub fn as_bytes(&self) -> [u8; 16] {
46        let mut bytes = [0; 16];
47        self.getUUIDBytes(&mut bytes);
48        bytes
49    }
50}

52impl fmt::Display for NSUUID {
53    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
54        let string: Retained<NSObject> = unsafe { msg_send![self, UUIDString] };
55        // SAFETY: `UUIDString` returns `NSString`.
56        unsafe { util::display_string(&string, f) }
57    }
58}

66        unsafe { util::display_string(&string, f) }
67    }
