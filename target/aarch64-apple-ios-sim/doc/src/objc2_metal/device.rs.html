1use crate::MTLDevice;
2use objc2::rc::Retained;
3use objc2::runtime::ProtocolObject;
4use objc2_foundation::NSArray;
5
6/// Returns all Metal devices in the system.
7///
8/// On macOS and macCatalyst, this API will not cause the system to switch
9/// devices and leaves the decision about which GPU to use up to the
10/// application based on whatever criteria it deems appropriate.
11///
12/// On iOS, tvOS and visionOS, this API returns an array containing the same
13/// device that MTLCreateSystemDefaultDevice would have returned, or an empty
14/// array if it would have failed.
15#[inline]
16#[allow(unexpected_cfgs)]
17pub extern "C-unwind" fn MTLCopyAllDevices() -> Retained<NSArray<ProtocolObject<dyn MTLDevice>>> {
18    // MTLCopyAllDevices is always available on macOS and Mac Catalyst, but
19    // only available recently on iOS 18.0 / tvOS 18.0 / visionOS 2.0.
20    //
21    // Instead, we do the fallback to MTLCreateSystemDefaultDevice on those
22    // platforms that they do themselves on newer systems.
23    //
24    // TODO: Use something like <https://github.com/rust-lang/rfcs/pull/3750>
25    // to call the actual API when available.
26    #[cfg(any(target_os = "macos", target_env = "macabi"))]
27    {
28        extern "C-unwind" {
29            fn MTLCopyAllDevices() -> *mut NSArray<ProtocolObject<dyn MTLDevice>>;
30        }
31
32        let ret = unsafe { MTLCopyAllDevices() };
33        // SAFETY: Marked NS_RETURNS_RETAINED (and has `Copy` in the name).
34        unsafe { Retained::from_raw(ret) }
35            .expect("function was marked as returning non-null, but actually returned NULL")
36    }
37    #[cfg(not(any(target_os = "macos", target_env = "macabi")))]
38    {
39        let device = crate::MTLCreateSystemDefaultDevice();
40        let slice: &[_] = if let Some(device) = device.as_deref() {
41            &[device]
42        } else {
43            &[]
44        };
45        NSArray::from_slice(slice)
46    }
47}
