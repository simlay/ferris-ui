17/// A reference counted pointer type for Dispatch objects.
18///
19/// [`DispatchRetained`] strongly references or "retains" the given object
20/// `T`, and decrements the retain count or "releases" it again when dropped,
21/// thereby ensuring it will be deallocated at the right time.
22///
23/// The type `T` inside `DispatchRetained<T>` can be anything that implements
24/// [`DispatchObject`], i.e. any Dispatch object.
25///
26///
27/// # Comparison to other types
28///
29/// `DispatchRetained<T>` is equivalent to [`objc2::rc::Retained`], and can be
30/// converted to/from that when the `"objc2"` feature is enabled. Note though
31/// that this type uses the underlying Dispatch primitives `dispatch_retain` /
32/// `dispatch_release` instead of `objc_retain` / `objc_release` for
33/// performance, and to avoid depending on the Objective-C runtime if not
34/// needed.
35///
36/// You can also view `DispatchRetained<T>` as the Dispatch equivalent of
37/// [`std::sync::Arc`], that is, it is a thread-safe reference-counting smart
38/// pointer that allows cloning by bumping the reference count.
39///
40/// Unlike `Arc`, objects can be retained directly from a `&T` using
41/// [`DispatchObject::retain`] (for `Arc` you need `&Arc<T>`).
42///
43/// Weak references are not supported, for that you need to convert to
44/// `objc2::rc::Retained`.
45///
46#[cfg_attr(
47    not(feature = "objc2"),
48    doc = "[`objc2::rc::Retained`]: #objc2-not-available"
49)]
50#[cfg_attr(not(feature = "std"), doc = "[`std::sync::Arc`]: #std-not-enabled")]
51///
52///
53/// # Forwarding implementations
54///
55/// Since `DispatchRetained<T>` is a smart pointer, it [`Deref`]s to `T`.
56///
57/// It also forwards the implementation of a bunch of standard library traits
58/// such as [`PartialEq`], [`AsRef`], and so on, so that it becomes possible
59/// to use e.g. `DispatchRetained<DispatchQueue>` as-if it you had a
60/// `&DispatchQueue`. Note that having a `DispatchQueue` directly is not
61/// possible since dispatch objects cannot live on the stack, but instead must
62/// reside on the heap, and as such must be accessed behind a pointer or a
63/// reference.
64///
65///
66/// # Memory layout
67///
68/// This is guaranteed to have the same size and alignment as a pointer to the
69/// object, i.e. same as `*const T` or `dispatch_object_t`.
70///
71/// Additionally, it participates in the null-pointer optimization, that is,
72/// `Option<DispatchRetained<T>>` is guaranteed to have the same size as
73/// `DispatchRetained<T>`.
74#[repr(transparent)]
75#[doc(alias = "Retained")]
76#[doc(alias = "objc2::rc::Retained")]
77// TODO: Add `ptr::Thin` bound on `T` to allow for only extern types
78pub struct DispatchRetained<T: ?Sized> {
79    /// A pointer to the contained object. The pointer is always retained.
80    ///
81    /// It is important that this is `NonNull`, since we want to dereference
82    /// it later, and be able to use the null-pointer optimization.
83    ptr: NonNull<T>,
84    /// Necessary for dropck even though we never actually run T's destructor,
85    /// because it might have a `dealloc` that assumes that contained
86    /// references outlive the type.
87    ///
88    /// See <https://doc.rust-lang.org/nightly/nomicon/phantom-data.html>
89    item: PhantomData<T>,
90    /// Marks the type as !UnwindSafe. Later on we'll re-enable this.
91    ///
92    /// See <https://github.com/rust-lang/rust/issues/93367> for why this is
93    /// required.
94    notunwindsafe: PhantomData<&'static mut ()>,
95}

97// Same as `objc::rc::Retained`, `#[may_dangle]` does not apply here.
98impl<T: ?Sized> Drop for DispatchRetained<T> {
99    /// Releases the contained object.
100    #[doc(alias = "dispatch_release")]
101    #[doc(alias = "release")]
102    #[inline]
103    fn drop(&mut self) {
104        // SAFETY: The `ptr` is guaranteed to be valid and have at least one
105        // retain count.
106        unsafe { dispatch_release(self.ptr.cast()) };
107    }
108}

110impl<T: ?Sized + DispatchObject> DispatchRetained<T> {
111    /// Construct a `DispatchRetained` from a pointer that already has +1
112    /// retain count.
113    ///
114    /// This is useful when you have been given a pointer to an object from
115    /// some API that returns a retained pointer, and expects you to release
116    /// it. That is, an API that follows [the create rule].
117    ///
118    /// [the create rule]: https://developer.apple.com/library/archive/documentation/CoreFoundation/Conceptual/CFMemoryMgmt/Concepts/Ownership.html#//apple_ref/doc/uid/20001148-103029
119    ///
120    ///
121    /// # Safety
122    ///
123    /// The pointer must be a valid and live object according to Dispatch, and
124    /// it must be of type `T`.
125    ///
126    /// Additionally, you must ensure the given object pointer has +1 retain
127    /// count.
128    #[inline]
129    pub unsafe fn from_raw(ptr: NonNull<T>) -> Self {
130        Self {
131            ptr,
132            item: PhantomData,
133            notunwindsafe: PhantomData,
134        }
135    }
136
137    /// Retain the pointer and construct a [`DispatchRetained`] from it.
138    ///
139    /// This is useful when you have been given a pointer to an object from
140    /// some API that follows [the get rule], and you would like to keep it
141    /// around for longer than the current memory context.
142    ///
143    /// [the get rule]: https://developer.apple.com/library/archive/documentation/CoreFoundation/Conceptual/CFMemoryMgmt/Concepts/Ownership.html#//apple_ref/doc/uid/20001148-SW1
144    ///
145    ///
146    /// # Safety
147    ///
148    /// The pointer must be a valid and live object according to Dispatch, and
149    /// it must be of type `T`.
150    #[doc(alias = "dispatch_retain")]
151    #[inline]
152    pub unsafe fn retain(ptr: NonNull<T>) -> Self {
153        // SAFETY: The caller upholds that the pointer is valid.
154        unsafe { dispatch_retain(ptr.cast()) };
155
156        // SAFETY: We just retained the object, so it has +1 retain count.
157        // Validity of the pointer is upheld by the caller.
158        unsafe { Self::from_raw(ptr) }
159    }
160
161    /// Consumes the `DispatchRetained`, returning a raw pointer with +1
162    /// retain count.
163    ///
164    /// After calling this function, the caller is responsible for the memory
165    /// previously managed by the `DispatchRetained`.
166    ///
167    /// This is effectively the opposite of [`DispatchRetained::from_raw`].
168    ///
169    /// This is an associated method, and must be called as
170    /// `DispatchRetained::into_raw(obj)`.
171    #[inline]
172    pub fn into_raw(this: Self) -> NonNull<T> {
173        ManuallyDrop::new(this).ptr
174    }
175
176    /// Returns a raw pointer to the object.
177    ///
178    /// The pointer is valid for at least as long as the `DispatchRetained` is
179    /// held.
180    ///
181    /// This is an associated method, and must be called as
182    /// `DispatchRetained::as_ptr(&obj)`.
183    #[inline]
184    pub fn as_ptr(this: &Self) -> NonNull<T> {
185        this.ptr
186    }
187
188    /// Unchecked conversion to another Dispatch object.
189    ///
190    /// This is equivalent to a `cast` between two pointers.
191    ///
192    /// This is an associated method, and must be called as
193    /// `DispatchRetained::cast_unchecked(obj)`.
194    ///
195    ///
196    /// # Safety
197    ///
198    /// You must ensure that the object can be reinterpreted as the given
199    /// type.
200    ///
201    /// If `T` is not `'static`, you must ensure that `U` ensures that the
202    /// data contained by `T` is kept alive for as long as `U` lives.
203    ///
204    /// Additionally, you must ensure that any safety invariants that the new
205    /// type has are upheld.
206    #[inline]
207    // TODO: Add ?Sized bound
208    pub unsafe fn cast_unchecked<U: DispatchObject>(this: Self) -> DispatchRetained<U> {
209        // SAFETY: The object is forgotten, so we have +1 retain count.
210        //
211        // Caller verifies that the object is of the correct type.
212        unsafe { DispatchRetained::from_raw(Self::into_raw(this).cast()) }
213    }
214}

255#[cfg(feature = "objc2")]
256impl<T: ?Sized + DispatchObject + objc2::Message> From<objc2::rc::Retained<T>>
257    for DispatchRetained<T>
258{
259    /// Convert a [`objc2::rc::Retained`] into a [`DispatchRetained`].
260    ///
261    /// This only works if the type is a Dispatch object (implements the
262    /// [`DispatchObject`] trait).
263    ///
264    /// This conversion is cost-free.
265    #[inline]
266    fn from(obj: objc2::rc::Retained<T>) -> Self {
267        let ptr = objc2::rc::Retained::into_raw(obj);
268        let ptr = NonNull::new(ptr).unwrap();
269        // SAFETY: `T` is bound by `DispatchObject`, so we know that the type
270        // is a Dispatch object, and hence we know that it will respond to
271        // `dispatch_retain`/`dispatch_release`.
272        //
273        // Additionally, the pointer is valid and has +1 retain count, since
274        // we're passing it from `Retained::into_raw`.
275        unsafe { Self::from_raw(ptr) }
276    }
277}

280impl<T: ?Sized + DispatchObject + objc2::Message> From<DispatchRetained<T>>
281    for objc2::rc::Retained<T>
282{
283    /// Convert a [`DispatchRetained`] into a [`objc2::rc::Retained`].
284    ///
285    /// This conversion is cost-free, since Dispatch objects are fully
286    /// interoperable with Objective-C retain/release message sending.
287    #[inline]
288    fn from(obj: DispatchRetained<T>) -> Self {
289        let ptr = DispatchRetained::into_raw(obj);
290        // SAFETY: `T` is bound by `Message`, so we know that the type is an
291        // Objective-C object, and hence we know that it will respond to
292        // `objc_retain`, `objc_release` etc.
293        //
294        // Additionally, the pointer is valid and has +1 retain count, since
295        // we're passing it from `DispatchRetained::into_raw`.
296        unsafe { Self::from_raw(ptr.as_ptr()) }.unwrap()
297    }
298}
