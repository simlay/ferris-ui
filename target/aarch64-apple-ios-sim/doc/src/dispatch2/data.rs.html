24    /// Creates a dispatch data object with a copy of the given contiguous
25    /// buffer of memory.
26    #[cfg(feature = "block2")]
27    pub fn from_bytes(data: &[u8]) -> DispatchRetained<Self> {
28        // TODO: Autogenerate?
29        const DISPATCH_DATA_DESTRUCTOR_DEFAULT: crate::dispatch_block_t = ptr::null_mut();
30
31        let ptr = NonNull::new(data.as_ptr().cast_mut()).unwrap().cast();
32
33        // We don't care which queue ends up running the destructor.
34        let queue = None;
35
36        // SAFETY: Buffer pointer is valid for the given number of bytes.
37        //
38        // The destructor is DISPATCH_DATA_DESTRUCTOR_DEFAULT, which indicates
39        // the buffer should be copied, so it's safe to keep after the end of
40        // this function.
41        unsafe { Self::new(ptr, data.len(), queue, DISPATCH_DATA_DESTRUCTOR_DEFAULT) }
42    }

44    /// Creates a dispatch data object with a reference to the given
45    /// contiguous buffer of memory.
46    #[cfg(feature = "block2")]
47    pub fn from_static_bytes(data: &'static [u8]) -> DispatchRetained<Self> {
48        block2::global_block! {
49            static NOOP_BLOCK = || {}
50        }
51
52        let ptr = NonNull::new(data.as_ptr().cast_mut()).unwrap().cast();
53
54        // We don't care which queue ends up running the destructor.
55        let queue = None;
56
57        let destructor = (&*NOOP_BLOCK as *const block2::Block<_>).cast_mut();
58
59        // SAFETY: Buffer pointer is valid for the given number of bytes.
60        // Queue handle is valid, and the destructor is a NULL value which
61        // indicates the buffer should be copied.
62        unsafe { Self::new(ptr, data.len(), queue, destructor) }
63    }

65    /// Creates a dispatch data object with ownership of the given contiguous
66    /// buffer of memory.
67    #[cfg(feature = "alloc")]
68    #[cfg(feature = "block2")]
69    pub fn from_boxed(data: alloc::boxed::Box<[u8]>) -> DispatchRetained<Self> {
70        let data_len = data.len();
71        let raw = alloc::boxed::Box::into_raw(data);
72        let ptr = NonNull::new(raw).unwrap().cast();
73
74        let destructor = block2::RcBlock::new(move || {
75            // SAFETY: The fat pointer (plus size) was retrieved from
76            // `Box::into_raw()`, and its ownership was *not* consumed by
77            // dispatch_data_create().
78            let _ = unsafe { alloc::boxed::Box::<[u8]>::from_raw(raw) };
79        });
80        let destructor = block2::RcBlock::as_ptr(&destructor);
81
82        // We don't care which queue ends up running the destructor.
83        // Box<[u8]> is sendable, so it's fine for us to potentially pass it
84        // to a different thread.
85        let queue = None;
86
87        // SAFETY: Buffer pointer is valid for the given number of bytes.
88        //
89        // The destructor is valid and correctly destroys the buffer.
90        unsafe { Self::new(ptr, data_len, queue, destructor) }
91    }

93    /// Copy all the non-contiguous parts of the data into a contiguous
94    /// [`Vec`][std::vec::Vec].
95    ///
96    /// # Examples
97    ///
98    /// ```
99    /// use dispatch2::DispatchData;
100    ///
101    /// let data = DispatchData::from_bytes(b"foo");
102    /// assert_eq!(data.to_vec(), b"foo");
103    /// ```
104    #[cfg(feature = "alloc")]
105    #[cfg(feature = "block2")]
106    #[cfg(feature = "objc2")]
107    pub fn to_vec(&self) -> alloc::vec::Vec<u8> {
108        let contents = core::cell::RefCell::new(alloc::vec::Vec::new());
109        let block = block2::RcBlock::new(
110            |_region, _offset, buffer: NonNull<core::ffi::c_void>, size| {
111                // SAFETY: Dispatch guarantees that the slice is valid.
112                let buffer =
113                    unsafe { core::slice::from_raw_parts(buffer.cast::<u8>().as_ptr(), size) };
114                contents.borrow_mut().extend_from_slice(buffer);
115                1
116            },
117        );
118
119        let block = block2::RcBlock::as_ptr(&block);
120        // SAFETY: Transmute from return type `u8` to `bool` is safe, since we
121        // only ever return `1` / `true`.
122        // TODO: Fix the need for this in `block2`.
123        let block = unsafe {
124            core::mem::transmute::<
125                *mut block2::Block<
126                    dyn Fn(NonNull<DispatchData>, usize, NonNull<core::ffi::c_void>, usize) -> u8,
127                >,
128                *mut block2::Block<
129                    dyn Fn(NonNull<DispatchData>, usize, NonNull<core::ffi::c_void>, usize) -> bool,
130                >,
131            >(block)
132        };
133
134        // SAFETY: The block is implemented correctly.
135        unsafe { self.apply(block) };
136        contents.take()
137    }

180    // Test destruction, and that it still works when we add a finalizer to the data.
181    #[test]
182    fn with_finalizer() {
183        #[derive(Debug)]
184        struct State {
185            has_run_destructor: bool,
186            has_run_finalizer: bool,
187        }
188
189        let state = State {
190            has_run_destructor: false,
191            has_run_finalizer: false,
192        };
193        let pair = Arc::new((Mutex::new(state), Condvar::new()));
194
195        let pair2 = Arc::clone(&pair);
196        let destructor = RcBlock::new(move || {
197            let (lock, cvar) = &*pair2;
198            lock.lock().unwrap().has_run_destructor = true;
199            cvar.notify_one();
200        });
201
202        // SAFETY: The pointers are correct.
203        let data = unsafe {
204            let data = b"xyz";
205            DispatchData::new(
206                NonNull::new(data.as_ptr().cast_mut()).unwrap().cast(),
207                data.len(),
208                None,
209                RcBlock::as_ptr(&destructor),
210            )
211        };
212
213        let pair3 = Arc::clone(&pair);
214        data.set_finalizer(move || {
215            let (lock, cvar) = &*pair3;
216            lock.lock().unwrap().has_run_finalizer = true;
217            cvar.notify_one();
218        });
219
220        let (lock, cvar) = &*pair;
221        let lock = lock.lock().unwrap();
222
223        // Verify the destructor hasn't run yet.
224        let (lock, res) = cvar.wait_timeout(lock, Duration::from_millis(10)).unwrap();
225        assert!(res.timed_out());
226        assert!(!lock.has_run_destructor);
227        assert!(!lock.has_run_finalizer);
228
229        let data2 = data.clone();
230        drop(data);
231
232        // Still not yet, the second reference is still alive.
233        let (lock, res) = cvar.wait_timeout(lock, Duration::from_millis(10)).unwrap();
234        assert!(res.timed_out());
235        assert!(!lock.has_run_destructor);
236        assert!(!lock.has_run_finalizer);
237
238        let data3 = data2.concat(&DispatchData::from_bytes(b"foo"));
239        drop(data2);
240
241        // Still not yet, the reference is kept alive by the new data.
242        let (lock, res) = cvar.wait_timeout(lock, Duration::from_millis(10)).unwrap();
243        assert!(res.timed_out());
244        assert!(!lock.has_run_destructor);
245        assert!(!lock.has_run_finalizer);
246
247        drop(data3);
248
249        // Has run now!
250        let (lock, res) = cvar.wait_timeout(lock, Duration::from_millis(10)).unwrap();
251        assert!(!res.timed_out());
252        assert!(lock.has_run_destructor);
253        assert!(lock.has_run_finalizer);
254    }
