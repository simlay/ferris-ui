16impl UIApplication {
17    #[allow(clippy::needless_doctest_main)] // Useful to show a full example
18    /// The entry point to UIKit applications.
19    ///
20    /// Creates the application object and the application delegate and sets
21    /// up the event cycle.
22    ///
23    /// See [Apple's documentation][apple-doc] for more details.
24    ///
25    /// [apple-doc]: https://developer.apple.com/documentation/uikit/uiapplicationmain(_:_:_:_:)-1yub7
26    ///
27    /// # Example
28    ///
29    /// Create an application delegate and launch the application.
30    ///
31    /// ```no_run
32    /// use objc2::MainThreadMarker;
33    /// use objc2::rc::{Allocated, Retained};
34    /// use objc2::{define_class, msg_send, ClassType, Ivars, MainThreadOnly};
35    /// use objc2_foundation::{NSNotification, NSObject, NSObjectProtocol, NSString};
36    /// use objc2_ui_kit::{UIApplication, UIApplicationDelegate};
37    ///
38    /// define_class!(
39    ///     // SAFETY:
40    ///     // - `NSObject` does not have any subclassing requirements.
41    ///     // - `AppDelegate` does not implement `Drop`.
42    ///     #[unsafe(super(NSObject))]
43    ///     #[thread_kind = MainThreadOnly]
44    ///     #[derive(Debug)]
45    ///     struct AppDelegate {
46    ///         // Whatever state you want to store in your delegate.
47    ///     }
48    ///
49    ///     impl AppDelegate {
50    ///         // Called by `UIApplication::main`.
51    ///         #[unsafe(method_id(init))]
52    ///         fn init(this: Allocated<Self>) -> Retained<Self> {
53    ///             let this = this.set_ivars(Ivars::<Self> {});
54    ///             unsafe { msg_send![super(this), init] }
55    ///         }
56    ///     }
57    ///
58    ///     unsafe impl NSObjectProtocol for AppDelegate {}
59    ///
60    ///     unsafe impl UIApplicationDelegate for AppDelegate {
61    ///         #[unsafe(method(applicationDidFinishLaunching:))]
62    ///         fn did_finish_launching(&self, _notification: &NSNotification) {
63    ///             println!("did finish launching!");
64    ///
65    ///             // Do UI initialization in here, such as creating windows, views, etc.
66    ///         }
67    ///
68    ///         #[unsafe(method(applicationWillTerminate:))]
69    ///         fn will_terminate(&self, _notification: &NSNotification) {
70    ///             println!("will terminate!");
71    ///
72    ///             // Tear down your application state here.
73    ///         }
74    ///     }
75    /// );
76    ///
77    /// fn main() {
78    ///     let mtm = MainThreadMarker::new().unwrap();
79    ///     let delegate_class = NSString::from_class(AppDelegate::class());
80    ///     UIApplication::main(None, Some(&delegate_class), mtm);
81    /// }
82    /// ```
83    #[doc(alias = "UIApplicationMain")]
84    pub fn main(
85        principal_class_name: Option<&NSString>,
86        delegate_class_name: Option<&NSString>,
87        mtm: MainThreadMarker,
88    ) -> ! {
89        // UIApplicationMain must be called on the main thread.
90        let _ = mtm;
91
92        // NOTE: `UIApplicationMain` ignores `argc` and `argv`, so we choose
93        // to not expose those in our API.
94        // We pass correct values anyhow though, just to be certain.
95        let argc = unsafe { *_NSGetArgc() };
96        let argv = unsafe { NonNull::new(*_NSGetArgv()).unwrap().cast() };
97
98        // SAFETY: `argc` and `argv` are correct.
99        // `UIApplicationMain` is safely re-entrant, just weird to do so.
100        let _ret = unsafe { Self::__main(argc, argv, principal_class_name, delegate_class_name) };
101
102        // UIApplicationMain is documented to never return, so whatever we do
103        // here is just for show really.
104        #[cfg(feature = "std")]
105        {
106            std::process::exit(_ret as i32)
107        }
108        #[cfg(not(feature = "std"))]
109        {
110            unreachable!("UIApplicationMain should not have returned")
111        }
112    }
113}
