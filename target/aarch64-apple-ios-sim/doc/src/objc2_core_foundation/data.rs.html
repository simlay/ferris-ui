9impl CFData {
10    /// Creates a new `CFData` from a byte slice.
11    #[inline]
12    #[cfg(feature = "CFBase")]
13    #[doc(alias = "CFDataCreate")]
14    pub fn from_bytes(bytes: &[u8]) -> crate::CFRetained<Self> {
15        let len = bytes.len().try_into().expect("buffer too large");
16        unsafe { crate::CFDataCreate(None, bytes.as_ptr(), len) }.expect("failed creating CFData")
17    }
18
19    /// Creates a new `CFData` from a `'static` byte slice.
20    ///
21    /// This may be slightly more efficient than [`CFData::from_bytes`], as it
22    /// may be able to re-use the existing buffer (since we know it won't be
23    /// deallocated).
24    #[inline]
25    #[cfg(feature = "CFBase")]
26    #[doc(alias = "CFDataCreateWithBytesNoCopy")]
27    pub fn from_static_bytes(bytes: &'static [u8]) -> crate::CFRetained<Self> {
28        let len = bytes.len().try_into().expect("buffer too large");
29        // SAFETY: Same as `CFString::from_static_str`.
30        unsafe {
31            crate::CFDataCreateWithBytesNoCopy(None, bytes.as_ptr(), len, crate::kCFAllocatorNull)
32        }
33        .expect("failed creating CFData")
34    }
35
36    /// The number of bytes contained by the `CFData`.
37    #[inline]
38    #[doc(alias = "CFDataGetLength")]
39    pub fn len(&self) -> usize {
40        unsafe { CFDataGetLength(self) as _ }
41    }
42
43    /// Whether the `CFData` is empty.
44    #[inline]
45    pub fn is_empty(&self) -> bool {
46        self.len() == 0
47    }
48
49    /// The underlying bytes in the `CFData`.
50    ///
51    ///
52    /// # Safety
53    ///
54    /// The `CFData` must not be mutated for the lifetime of the returned
55    /// string. Consider using [`to_vec`] instead if this requirement is a bit
56    /// difficult to uphold.
57    ///
58    /// [`to_vec`]: Self::to_vec
59    #[inline]
60    #[doc(alias = "CFDataGetBytePtr")]
61    pub unsafe fn as_bytes_unchecked(&self) -> &[u8] {
62        let ptr = unsafe { CFDataGetBytePtr(self) };
63        if !ptr.is_null() {
64            // SAFETY: The pointer is valid, and caller ensures that the
65            // `CFData` is not mutated for the lifetime of it.
66            //
67            // Same as
68            unsafe { slice::from_raw_parts(ptr, self.len()) }
69        } else {
70            // The bytes pointer may be null for length zero
71            &[]
72        }
73    }
74
75    /// Copy the contents of the `CFData` into a new [`Vec`].
76    #[cfg(feature = "alloc")]
77    #[doc(alias = "CFDataGetBytePtr")]
78    pub fn to_vec(&self) -> Vec<u8> {
79        // NOTE: We don't do `Vec::from`, as that will call the allocator
80        // while the buffer is active, and we don't know if that allocator
81        // uses a CFMutableData under the hood (though very theoretical).
82
83        let mut vec = Vec::with_capacity(self.len());
84        // SAFETY: We've pre-allocated the Vec, so it won't call the allocator
85        // while the byte slice is alive (and hence it won't ).
86        vec.extend_from_slice(unsafe { self.as_bytes_unchecked() });
87        vec
88    }
89}
