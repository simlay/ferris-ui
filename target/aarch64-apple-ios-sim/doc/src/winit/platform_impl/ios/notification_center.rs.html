1use std::ptr::NonNull;

3use block2::RcBlock;
4use objc2::rc::Retained;
5use objc2_foundation::{NSNotification, NSNotificationCenter, NSNotificationName, NSObject};
6
7/// Observe the given notification.
8///
9/// This is used in Winit as an alternative to declaring an application delegate, as we want to
10/// give the user full control over those.
11pub fn create_observer(
12    center: &NSNotificationCenter,
13    name: &NSNotificationName,
14    handler: impl Fn(&NSNotification) + 'static,
15) -> Retained<NSObject> {
16    let block = RcBlock::new(move |notification: NonNull<NSNotification>| {
17        handler(unsafe { notification.as_ref() });
18    });
19    unsafe {
20        center.addObserverForName_object_queue_usingBlock(
21            Some(name),
22            None, // No sender filter
23            None, // No queue, run on posting thread (i.e. main thread)
24            &block,
25        )
26    }
27}
