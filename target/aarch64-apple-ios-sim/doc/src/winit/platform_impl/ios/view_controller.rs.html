72impl WinitViewController {
73    pub(crate) fn set_prefers_status_bar_hidden(&self, val: bool) {
74        self.ivars().prefers_status_bar_hidden.set(val);
75        self.setNeedsStatusBarAppearanceUpdate();
76    }
77
78    pub(crate) fn set_preferred_status_bar_style(&self, val: StatusBarStyle) {
79        let val = match val {
80            StatusBarStyle::Default => UIStatusBarStyle::Default,
81            StatusBarStyle::LightContent => UIStatusBarStyle::LightContent,
82            StatusBarStyle::DarkContent => UIStatusBarStyle::DarkContent,
83        };
84        self.ivars().preferred_status_bar_style.set(val);
85        self.setNeedsStatusBarAppearanceUpdate();
86    }
87
88    pub(crate) fn set_prefers_home_indicator_auto_hidden(&self, val: bool) {
89        self.ivars().prefers_home_indicator_auto_hidden.set(val);
90        let os_capabilities = app_state::os_capabilities();
91        if os_capabilities.home_indicator_hidden {
92            self.setNeedsUpdateOfHomeIndicatorAutoHidden();
93        } else {
94            os_capabilities.home_indicator_hidden_err_msg("ignoring")
95        }
96    }
97
98    pub(crate) fn set_preferred_screen_edges_deferring_system_gestures(&self, val: ScreenEdge) {
99        let val = {
100            assert_eq!(val.bits() & !ScreenEdge::ALL.bits(), 0, "invalid `ScreenEdge`");
101            UIRectEdge(val.bits().into())
102        };
103        self.ivars().preferred_screen_edges_deferring_system_gestures.set(val);
104        let os_capabilities = app_state::os_capabilities();
105        if os_capabilities.defer_system_gestures {
106            self.setNeedsUpdateOfScreenEdgesDeferringSystemGestures();
107        } else {
108            os_capabilities.defer_system_gestures_err_msg("ignoring")
109        }
110    }
111
112    pub(crate) fn set_supported_interface_orientations(
113        &self,
114        mtm: MainThreadMarker,
115        valid_orientations: ValidOrientations,
116    ) {
117        let mask = match (valid_orientations, UIDevice::currentDevice(mtm).userInterfaceIdiom()) {
118            (ValidOrientations::LandscapeAndPortrait, UIUserInterfaceIdiom::Phone) => {
119                UIInterfaceOrientationMask::AllButUpsideDown
120            },
121            (ValidOrientations::LandscapeAndPortrait, _) => UIInterfaceOrientationMask::All,
122            (ValidOrientations::Landscape, _) => UIInterfaceOrientationMask::Landscape,
123            (ValidOrientations::Portrait, UIUserInterfaceIdiom::Phone) => {
124                UIInterfaceOrientationMask::Portrait
125            },
126            (ValidOrientations::Portrait, _) => {
127                UIInterfaceOrientationMask::Portrait
128                    | UIInterfaceOrientationMask::PortraitUpsideDown
129            },
130        };
131        self.ivars().supported_orientations.set(mask);
132        #[allow(deprecated)]
133        UIViewController::attemptRotationToDeviceOrientation(mtm);
134    }
135
136    pub(crate) fn new(
137        mtm: MainThreadMarker,
138        window_attributes: &WindowAttributes,
139        view: &UIView,
140    ) -> Retained<Self> {
141        // These are set properly below, we just to set them to something in the meantime.
142        let this = mtm.alloc().set_ivars(ViewControllerState {
143            prefers_status_bar_hidden: Cell::new(false),
144            preferred_status_bar_style: Cell::new(UIStatusBarStyle::Default),
145            prefers_home_indicator_auto_hidden: Cell::new(false),
146            supported_orientations: Cell::new(UIInterfaceOrientationMask::All),
147            preferred_screen_edges_deferring_system_gestures: Cell::new(UIRectEdge::empty()),
148        });
149        let this: Retained<Self> = unsafe { msg_send_id![super(this), init] };
150
151        this.set_prefers_status_bar_hidden(
152            window_attributes.platform_specific.prefers_status_bar_hidden,
153        );
154
155        this.set_preferred_status_bar_style(
156            window_attributes.platform_specific.preferred_status_bar_style,
157        );
158
159        this.set_supported_interface_orientations(
160            mtm,
161            window_attributes.platform_specific.valid_orientations,
162        );
163
164        this.set_prefers_home_indicator_auto_hidden(
165            window_attributes.platform_specific.prefers_home_indicator_hidden,
166        );
167
168        this.set_preferred_screen_edges_deferring_system_gestures(
169            window_attributes.platform_specific.preferred_screen_edges_deferring_system_gestures,
170        );
171
172        this.setView(Some(view));
173
174        this
175    }
