8#[derive(Clone, Copy, PartialEq, Eq)]
9struct Isa(*const ffi::Class);
10
11impl Isa {
12    #[allow(unused_unsafe)]
13    fn is_global(self) -> bool {
14        ptr::eq(
15            unsafe { ptr::addr_of!(ffi::_NSConcreteGlobalBlock) },
16            self.0,
17        )
18    }
19
20    #[allow(unused_unsafe)]
21    fn is_stack(self) -> bool {
22        ptr::eq(unsafe { ptr::addr_of!(ffi::_NSConcreteStackBlock) }, self.0)
23    }
24}
25
26impl Debug for Isa {
27    fn fmt(&self, f: &mut Formatter<'_>) -> Result<(), Error> {
28        if self.is_global() {
29            f.write_str("_NSConcreteGlobalBlock")
30        } else if self.is_stack() {
31            f.write_str("_NSConcreteStackBlock")
32        } else {
33            write!(f, "{:?} (likely _NSConcreteMallocBlock)", self.0)
34        }
35    }
36}

38pub(crate) fn debug_block_header(header: &BlockHeader, f: &mut DebugStruct<'_, '_>) {
39    f.field("isa", &Isa(header.isa));
40    f.field("flags", &header.flags);
41    f.field("reserved", &header.reserved);
42    f.field("invoke", &header.invoke);
43    f.field(
44        "descriptor",
45        &BlockDescriptorHelper {
46            has_copy_dispose: header.flags.has(BlockFlags::BLOCK_HAS_COPY_DISPOSE),
47            has_signature: header.flags.has(BlockFlags::BLOCK_HAS_SIGNATURE),
48            descriptor: header.descriptor,
49        },
50    );
51}

53#[derive(Clone, Copy)]
54struct BlockDescriptorHelper {
55    has_copy_dispose: bool,
56    has_signature: bool,
57    descriptor: BlockDescriptorPtr,
58}
59
60impl Debug for BlockDescriptorHelper {
61    fn fmt(&self, f: &mut Formatter<'_>) -> Result<(), Error> {
62        if unsafe { self.descriptor.basic }.is_null() {
63            return f.write_str("(null)");
64        }
65
66        let mut f = f.debug_struct("BlockDescriptor");
67
68        let header = unsafe { self.descriptor.basic.as_ref().unwrap() };
69
70        f.field("reserved", &header.reserved);
71        f.field("size", &header.size);
72
73        match (self.has_copy_dispose, self.has_signature) {
74            (false, false) => {}
75            (true, false) => {
76                let descriptor = unsafe { self.descriptor.with_copy_dispose.as_ref().unwrap() };
77                f.field("copy", &descriptor.copy);
78                f.field("dispose", &descriptor.dispose);
79            }
80            (false, true) => {
81                let descriptor = unsafe { self.descriptor.with_signature.as_ref().unwrap() };
82                f.field(
83                    "encoding",
84                    &if descriptor.encoding.is_null() {
85                        None
86                    } else {
87                        Some(unsafe { CStr::from_ptr(descriptor.encoding) })
88                    },
89                );
90            }
91            (true, true) => {
92                let descriptor = unsafe {
93                    self.descriptor
94                        .with_copy_dispose_signature
95                        .as_ref()
96                        .unwrap()
97                };
98                f.field("copy", &descriptor.copy);
99                f.field("dispose", &descriptor.dispose);
100                f.field(
101                    "encoding",
102                    &if descriptor.encoding.is_null() {
103                        None
104                    } else {
105                        Some(unsafe { CStr::from_ptr(descriptor.encoding) })
106                    },
107                );
108            }
109        }
110
111        f.finish()
112    }
113}

119    #[test]
120    #[allow(unused_unsafe)]
121    fn test_isa() {
122        let isa = Isa(unsafe { ptr::addr_of!(ffi::_NSConcreteGlobalBlock) });
123        assert!(isa.is_global());
124        assert!(!isa.is_stack());
125        let isa = Isa(unsafe { ptr::addr_of!(ffi::_NSConcreteStackBlock) });
126        assert!(!isa.is_global());
127        assert!(isa.is_stack());
128        let isa = Isa(unsafe { ptr::addr_of!(ffi::private::_NSConcreteMallocBlock) });
129        assert!(!isa.is_global());
130        assert!(!isa.is_stack());
131        let isa = Isa(ptr::null());
132        assert!(!isa.is_global());
133        assert!(!isa.is_stack());
134    }
