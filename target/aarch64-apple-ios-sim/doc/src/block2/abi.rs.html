17/// Block descriptor flags.
18#[repr(transparent)]
19#[derive(Clone, Copy, PartialEq, Eq)]
20pub(crate) struct BlockFlags(pub(crate) c_int);
21
22impl BlockFlags {
23    pub(crate) const EMPTY: Self = Self(0);
24
25    /// Note: Not public ABI.
26    const BLOCK_DEALLOCATING: Self = Self(0x0001);
27
28    /// Note: Not public ABI.
29    const BLOCK_REFCOUNT_MASK: Self = Self(if cfg!(feature = "gnustep-1-7") {
30        // Mask for the reference count in byref structure's flags field. The low
31        // 3 bytes are reserved for the reference count, the top byte for the flags.
32        0x00ffffff
33    } else if cfg!(any(feature = "compiler-rt", feature = "unstable-objfw")) {
34        0xffff
35    } else if cfg!(target_vendor = "apple") {
36        0xfffe // runtime
37    } else {
38        0
39    });
40
41    /// Note: Not public ABI.
42    const BLOCK_INLINE_LAYOUT_STRING: Self = Self(1 << 21);
43
44    /// Note: Not public ABI.
45    const BLOCK_SMALL_DESCRIPTOR: Self = Self(1 << 22);
46
47    pub(crate) const BLOCK_IS_NOESCAPE: Self = Self(1 << 23);
48
49    /// Note: Not public ABI.
50    const BLOCK_NEEDS_FREE: Self = Self(1 << 24);
51
52    /// The block descriptor contains copy and dispose helpers.
53    pub(crate) const BLOCK_HAS_COPY_DISPOSE: Self = Self(1 << 25);
54
55    /// Helpers have C++ code.
56    #[doc(alias = "BLOCK_HAS_CXX_OBJ")]
57    pub(crate) const BLOCK_HAS_CTOR: Self = Self(1 << 26);
58
59    /// Note: Not public ABI.
60    const BLOCK_IS_GC: Self = Self(1 << 27);
61
62    /// Block is stored in global memory and does not need to be copied.
63    pub(crate) const BLOCK_IS_GLOBAL: Self = Self(1 << 28);
64
65    /// Block function uses a calling convention that returns a structure via a
66    /// pointer passed in by the caller.
67    ///
68    /// match (BLOCK_USE_STRET, BLOCK_HAS_SIGNATURE) {
69    ///     (false, false) => 10.6.ABI, no signature field available
70    ///     (true, false)  => 10.6.ABI, no signature field available
71    ///     (false, true)  => ABI.2010.3.16, regular calling convention, presence of signature field
72    ///     (true, true)   => ABI.2010.3.16, stret calling convention, presence of signature field,
73    /// }
74    ///
75    /// See <https://clang.llvm.org/docs/Block-ABI-Apple.html#high-level>
76    #[doc(alias = "BLOCK_USE_SRET")]
77    #[doc(alias = "BLOCK_HAS_DESCRIPTOR")]
78    pub(crate) const BLOCK_USE_STRET: Self = Self(1 << 29);
79
80    /// Block has an Objective-C type encoding.
81    pub(crate) const BLOCK_HAS_SIGNATURE: Self = Self(1 << 30);
82
83    /// Note: Not public ABI.
84    const BLOCK_HAS_EXTENDED_LAYOUT: Self = Self(1 << 31);
85
86    /// `const` version of [`PartialEq`].
87    pub(crate) const fn equals(self, other: Self) -> bool {
88        self.0 == other.0
89    }
90
91    /// `const` version of [`BitOr`]: adds the flags together.
92    pub(crate) const fn union(self, other: Self) -> Self {
93        Self(self.0 | other.0)
94    }
95
96    /// `const` version of [`BitAnd`]: only keeps the common flags.
97    pub(crate) const fn intersect(self, other: Self) -> Self {
98        Self(self.0 & other.0)
99    }
100
101    /// Returns `true` if and only if all the flags from `other` are enabled,
102    /// i.e. `self & other == other`.
103    pub(crate) const fn has(self, other: Self) -> bool {
104        self.intersect(other).equals(other)
105    }
106}
107
108/// See [`BlockFlags::union`].
109impl BitOr for BlockFlags {
110    type Output = Self;
111
112    fn bitor(self, other: Self) -> Self {
113        self.union(other)
114    }
115}
116
117/// See [`BlockFlags::intersect`].
118impl BitAnd for BlockFlags {
119    type Output = Self;
120
121    fn bitand(self, other: Self) -> Self {
122        self.intersect(other)
123    }
124}

126impl fmt::Debug for BlockFlags {
127    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
128        let mut f = f.debug_struct("BlockFlags");
129        f.field("value", &format!("{:032b}", self.0));
130
131        macro_rules! test_flags {
132            {$(
133                $(#[$m:meta])?
134                $name:ident: $flag:ident
135            );* $(;)?} => ($(
136                $(#[$m])?
137                f.field(stringify!($name), &self.has(Self::$flag));
138            )*)
139        }
140        test_flags! {
141            #[cfg(target_vendor = "apple")]
142            deallocating: BLOCK_DEALLOCATING;
143            #[cfg(target_vendor = "apple")]
144            inline_layout_string: BLOCK_INLINE_LAYOUT_STRING;
145            #[cfg(target_vendor = "apple")]
146            small_descriptor: BLOCK_SMALL_DESCRIPTOR;
147            #[cfg(target_vendor = "apple")]
148            is_noescape: BLOCK_IS_NOESCAPE;
149            #[cfg(target_vendor = "apple")]
150            needs_free: BLOCK_NEEDS_FREE;
151            has_copy_dispose: BLOCK_HAS_COPY_DISPOSE;
152            has_ctor: BLOCK_HAS_CTOR;
153            #[cfg(target_vendor = "apple")]
154            is_gc: BLOCK_IS_GC;
155            is_global: BLOCK_IS_GLOBAL;
156            use_stret: BLOCK_USE_STRET;
157            has_signature: BLOCK_HAS_SIGNATURE;
158            #[cfg(target_vendor = "apple")]
159            has_extended_layout: BLOCK_HAS_EXTENDED_LAYOUT;
160        }
161
162        f.field("over_referenced", &self.has(Self::BLOCK_REFCOUNT_MASK));
163        f.field(
164            "reference_count",
165            &((*self & Self::BLOCK_REFCOUNT_MASK).0 >> 1),
166        );
167
168        f.finish_non_exhaustive()
169    }
170}

202/// The expected header of every block.
203#[repr(C)]
204#[doc(alias = "__block_literal")]
205#[doc(alias = "Block_layout")]
206#[doc(alias = "Block_basic")]
207#[allow(missing_debug_implementations)]
208#[derive(Clone, Copy)]
209pub struct BlockHeader {
210    /// Class pointer.
211    ///
212    /// Always initialised to &_NSConcreteStackBlock for blocks that are
213    /// created on the stack or &_NSConcreteGlobalBlock for blocks that are
214    /// created in global storage.
215    pub isa: *const Class,
216    /// Flags.
217    ///
218    /// See the `BlockFlags` enumerated type for possible values.
219    ///
220    /// Contains reference count in Apple's and ObjFW's runtime.
221    #[doc(alias = "Block_flags")]
222    pub(crate) flags: BlockFlags,
223    /// Reserved.
224    ///
225    /// Initialized to 0 by the compiler, but is said to be uninitialized in
226    /// the specification.
227    ///
228    /// Used for the reference count in GNUStep's and WinObjC's runtime.
229    #[doc(alias = "Block_size")]
230    pub(crate) reserved: MaybeUninit<c_int>,
231    /// The function that implements the block.
232    ///
233    /// The first parameter is a pointer to this structure, the subsequent
234    /// parameters are the block's explicit parameters.
235    ///
236    /// If the BLOCK_USE_SRET & BLOCK_HAS_SIGNATURE flag is set, there is an
237    /// additional hidden parameter, which is a pointer to the space on the
238    /// stack allocated to hold the return value.
239    pub invoke: Option<unsafe extern "C-unwind" fn()>,
240    /// The block's descriptor.
241    pub(crate) descriptor: BlockDescriptorPtr,
242}
243
244/// The type of this is:
245/// ```pseudo-code
246/// match (BLOCK_HAS_COPY_DISPOSE, BLOCK_HAS_SIGNATURE) {
247///     (false, false) => BlockDescriptor,
248///     (true, false) => BlockDescriptorCopyDispose,
249///     (false, true) => BlockDescriptorSignature,
250///     (true, true) => BlockDescriptorCopyDisposeSignature,
251/// }
252/// ```
253///
254/// Since all of these start with `BlockDescriptor`, it is always safe to
255/// use the `basic` field.
256//
257// Note: We use an union on top of the pointer, since otherwise the descriptor
258// would be forced to have a greater size than is actually required.
259#[repr(C)]
260#[derive(Clone, Copy)]
261pub(crate) union BlockDescriptorPtr {
262    pub(crate) basic: *const BlockDescriptor,
263    pub(crate) with_copy_dispose: *const BlockDescriptorCopyDispose,
264    pub(crate) with_signature: *const BlockDescriptorSignature,
265    pub(crate) with_copy_dispose_signature: *const BlockDescriptorCopyDisposeSignature,
266}

280/// Block descriptor that contains copy and dispose operations.

282/// Requires BLOCK_HAS_COPY_DISPOSE.

285#[doc(alias = "Block_descriptor_2")]
286#[derive(Clone, Copy, Debug)]
287pub(crate) struct BlockDescriptorCopyDispose {
288    /// Reserved for future use. Currently always 0.
289    pub(crate) reserved: c_ulong,
290    /// Size of the block.
291    pub(crate) size: c_ulong,
292
293    /// Helper to copy the block if it contains nontrivial copy operations.
294    ///
295    /// This may be NULL since macOS 11.0.1 in Apple's runtime, but this
296    /// should not be relied on.
297    pub(crate) copy: Option<unsafe extern "C-unwind" fn(dst: *mut c_void, src: *const c_void)>,
298    /// Helper to destroy the block after being copied.
299    ///
300    /// This may be NULL since macOS 11.0.1 in Apple's runtime, but this
301    /// should not be relied on.
302    pub(crate) dispose: Option<unsafe extern "C-unwind" fn(src: *mut c_void)>,
303}
