8/// Type for block class ISAs.
9///
10/// This will likely become an extern type in the future.
11#[repr(C)]
12#[allow(missing_debug_implementations)]
13pub struct Class {
14    /// The size probably doesn't really matter here, as we only ever use the
15    /// classes behind pointers, but let's import it with the correct size to
16    /// be sure.
17    ///
18    /// This applies with the compiler-rt runtime and with Apple's runtime.
19    #[cfg(not(any(feature = "gnustep-1-7", feature = "unstable-objfw")))]
20    _priv: [*mut c_void; 32],
21
22    /// The size of this is unknown, so let's use a ZST so the compiler
23    /// doesn't assume anything about the size.
24    #[cfg(any(feature = "gnustep-1-7", feature = "unstable-objfw"))]
25    _priv: [u8; 0],
26
27    /// Mark as `!Send + !Sync + !Unpin` and as mutable behind shared
28    /// references (`!Freeze`).
29    ///
30    /// Same as `objc2::ffi::OpaqueData`.
31    _opaque: UnsafeCell<PhantomData<(*const UnsafeCell<()>, PhantomPinned)>>,
32}
33
34// Use `extern "C-unwind"`, runtime functions may call external routines.
35extern "C-unwind" {
36    /// Class ISA used for global blocks.
37    pub static _NSConcreteGlobalBlock: Class;
38
39    /// Class ISA used for stack blocks.
40    pub static _NSConcreteStackBlock: Class;
41
42    /// Copy/retain a block.
43    ///
44    /// When called on a:
45    /// - Global block: Does nothing.
46    /// - Stack block: `memmove`s the block to a new heap allocation, calls
47    ///   the copy helper, and returns the new malloc block.
48    /// - Malloc block: Increments the retain count.
49    ///
50    /// Returns `NULL` on allocation failure.
51    #[doc(alias = "Block_copy")]
52    pub fn _Block_copy(block: *const c_void) -> *mut c_void;
53
54    /// Release a block.
55    ///
56    /// When called on a:
57    /// - Global block: Does nothing.
58    /// - Stack block: Does nothing.
59    /// - Malloc block: Decrements the retain count, and if it reaches zero,
60    ///   calls the dispose helper and frees the underlying storage.
61    #[doc(alias = "Block_release")]
62    pub fn _Block_release(block: *const c_void);
63
64    /// Copy a block field or `__block` variable from one location to another.
65    ///
66    /// Called by C compilers to clone fields inside copy helper routines, and
67    /// to handle memory management of `__block` marked variables.
68    pub fn _Block_object_assign(dest_addr: *mut c_void, object: *const c_void, flags: c_int);
69
70    /// Dispose an object previously copied using `_Block_object_assign`.
71    ///
72    /// Called by C compilers to drop fields inside dispose helper routines,
73    /// and handle memory management of `__block` marked variables.
74    pub fn _Block_object_dispose(object: *const c_void, flags: c_int);
75}

103
104        // Returns a string describing the block's GC layout.
105        // This uses the GC skip/scan encoding.
106        // May return NULL.
107        // macOS 10.7

109        pub fn _Block_layout(block: *mut c_void) -> *const c_char;
110
111        // Returns a string describing the block's layout.
112        // This uses the "extended layout" form described above.
113        // May return NULL.
114        // macOS 10.8
115        #[cfg(any(doc, target_vendor = "apple"))]
116        pub fn _Block_extended_layout(block: *mut c_void) -> *const c_char;
117
118        // Callable only from the ARR weak subsystem while in exclusion zone
119        // macOS 10.7
120        #[cfg(any(doc, target_vendor = "apple"))]
121        pub fn _Block_tryRetain(block: *const c_void) -> bool;
122
123        // Callable only from the ARR weak subsystem while in exclusion zone
124        // macOS 10.7
125        #[cfg(any(doc, target_vendor = "apple"))]
126        pub fn _Block_isDeallocating(block: *const c_void) -> bool;
127
128        // indicates whether block was compiled with compiler that sets the ABI
129        // related metadata bits
130        // macOS 10.7
131        #[cfg(any(doc, target_vendor = "apple", feature = "gnustep-1-7"))]
132        pub fn _Block_has_signature(block: *mut c_void) -> bool;
133
134        // Returns a string describing the block's parameter and return types.
135        // The encoding scheme is the same as Objective-C @encode.
136        // Returns NULL for blocks compiled with some compilers.
137        // macOS 10.7
138        #[cfg(any(doc, target_vendor = "apple", feature = "gnustep-1-7"))]
139        pub fn _Block_signature(block: *mut c_void) -> *const c_char;
140    }
